86  /**
87   * A renderer that handles the drawing of bars for a bar plot where
88   * each bar has a high and low value.  This renderer is for use with the
89   * {@link CategoryPlot} class.  The example shown here is generated by the
90   * {@code IntervalBarChartDemo1.java} program included in the JFreeChart
91   * Demo Collection:
92   * <br><br>
93   * <img src="../../../../../images/IntervalBarRendererSample.png"
94   * alt="IntervalBarRendererSample.png">
95   */
96  public class IntervalBarRenderer extends BarRenderer {
97  
98      /** For serialization. */
99      private static final long serialVersionUID = -5068857361615528725L;
100 
101     /**
102      * Constructs a new renderer.
103      */
104     public IntervalBarRenderer() {
105         super();
106     }
107 
108     /**
109      * Returns the range of values from the specified dataset.  For this
110      * renderer, this is equivalent to calling
111      * {@code findRangeBounds(dataset, true)}.
112      *
113      * @param dataset  the dataset ({@code null} permitted).
114      *
115      * @return The range (or {@code null} if the dataset is
116      *         {@code null} or empty).
117      */
118     @Override
119     public Range findRangeBounds(CategoryDataset dataset) {
120         return findRangeBounds(dataset, true);
121     }

123     /**
124      * Draws the bar for a single (series, category) data item.
125      *
126      * @param g2  the graphics device.
127      * @param state  the renderer state.
128      * @param dataArea  the data area.
129      * @param plot  the plot.
130      * @param domainAxis  the domain axis.
131      * @param rangeAxis  the range axis.
132      * @param dataset  the dataset.
133      * @param row  the row index (zero-based).
134      * @param column  the column index (zero-based).
135      * @param pass  the pass index.
136      */
137     @Override
138     public void drawItem(Graphics2D g2, CategoryItemRendererState state,
139             Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
140             ValueAxis rangeAxis, CategoryDataset dataset, int row, int column,
141             int pass) {
142 
143          if (dataset instanceof IntervalCategoryDataset) {
144              IntervalCategoryDataset d = (IntervalCategoryDataset) dataset;
145              drawInterval(g2, state, dataArea, plot, domainAxis, rangeAxis,
146                      d, row, column);
147          } else {
148              super.drawItem(g2, state, dataArea, plot, domainAxis, rangeAxis,
149                      dataset, row, column, pass);
150          }
151 
152      }
153 
154      /**
155       * Draws a single interval.
156       *
157       * @param g2  the graphics device.
158       * @param state  the renderer state.
159       * @param dataArea  the data plot area.
160       * @param plot  the plot.
161       * @param domainAxis  the domain axis.
162       * @param rangeAxis  the range axis.
163       * @param dataset  the data.
164       * @param row  the row index (zero-based).
165       * @param column  the column index (zero-based).
166       */
167      protected void drawInterval(Graphics2D g2, CategoryItemRendererState state,
168             Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
169             ValueAxis rangeAxis, IntervalCategoryDataset dataset, int row,
170                                  int column) {
171 
172         int visibleRow = state.getVisibleSeriesIndex(row);
173         if (visibleRow < 0) {
174             return;
175         }
176 
177         PlotOrientation orientation = plot.getOrientation();
178         double rectX = 0.0;
179         double rectY = 0.0;
180 
181         RectangleEdge rangeAxisLocation = plot.getRangeAxisEdge();
182 
183         // Y0
184         Number value0 = dataset.getEndValue(row, column);
185         if (value0 == null) {
186             return;
187         }
188         double java2dValue0 = rangeAxis.valueToJava2D(value0.doubleValue(),
189                 dataArea, rangeAxisLocation);
190 
191         // Y1
192         Number value1 = dataset.getStartValue(row, column);
193         if (value1 == null) {
194             return;
195         }
196         double java2dValue1 = rangeAxis.valueToJava2D(
197                 value1.doubleValue(), dataArea, rangeAxisLocation);
198 
199         if (java2dValue1 < java2dValue0) {
200             double temp = java2dValue1;
201             java2dValue1 = java2dValue0;
202             java2dValue0 = temp;
203         }
204 
205         // BAR WIDTH
206         double rectWidth = state.getBarWidth();
207 
208         // BAR HEIGHT
209         double rectHeight = Math.abs(java2dValue1 - java2dValue0);
210 
211         RectangleEdge barBase = RectangleEdge.LEFT;
212         if (orientation == PlotOrientation.HORIZONTAL) {
213             // BAR Y
214             rectX = java2dValue0;
215             rectY = calculateBarW0(getPlot(), orientation, dataArea, 
216                     domainAxis, state, visibleRow, column);
217             rectHeight = state.getBarWidth();
218             rectWidth = Math.abs(java2dValue1 - java2dValue0);
219             barBase = RectangleEdge.LEFT;
220         } else if (orientation.isVertical()) {
221             // BAR X
222             rectX = calculateBarW0(getPlot(), orientation, dataArea, 
223                     domainAxis, state, visibleRow, column);
224             rectY = java2dValue0;
225             barBase = RectangleEdge.BOTTOM;
226         }
227         Rectangle2D bar = new Rectangle2D.Double(rectX, rectY, rectWidth,
228                 rectHeight);
229         BarPainter painter = getBarPainter();
230         if (state.getElementHinting()) {
231             beginElementGroup(g2, dataset.getRowKey(row), 
232                     dataset.getColumnKey(column));
233         }
234         if (getShadowsVisible()) {
235             painter.paintBarShadow(g2, this, row, column, bar, barBase, false);
236         }
237         getBarPainter().paintBar(g2, this, row, column, bar, barBase);
238         if (state.getElementHinting()) {
239             endElementGroup(g2);
240         }
241 
242         CategoryItemLabelGenerator generator = getItemLabelGenerator(row,
243                 column);
244         if (generator != null && isItemLabelVisible(row, column)) {
245             drawItemLabel(g2, dataset, row, column, plot, generator, bar,
246                     false);
247         }
248 
249         // add an item entity, if this information is being collected
250         EntityCollection entities = state.getEntityCollection();
251         if (entities != null) {
252             addItemEntity(entities, dataset, row, column, bar);
253         }
254 
255     }
