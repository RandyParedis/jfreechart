41  package org.jfree.chart;
42  
43  import java.util.Map;
44  
45  /**
46   * Special rendering hints that can be used internally by JFreeChart or by
47   * specialised implementations of the {@code Graphics2D} API.  For example,
48   * JFreeSVG's {@code SVGGraphics2D} class, will use the 
49   * {@code KEY_BEGIN_ELEMENT} and {@code KEY_END_ELEMENT} hints to drive the 
50   * output content.
51   * 
52   * @since 1.0.18
53   */
54  public final class ChartHints {
55  
56      private ChartHints() {
57          // no need to instantiate this    
58      }
59      
60      /**
61       * The key for a hint to signal the beginning of an element.  The value
62       * should be a string containing the element id or, alternatively, a Map 
63       * containing the 'id' (String) and 'ref' (String in JSON format).
64       */
65      public static final Key KEY_BEGIN_ELEMENT = new ChartHints.Key(0);
66      
67      /**
68       * The key for a hint that ends an element.
69       */
70      public static final Key KEY_END_ELEMENT = new ChartHints.Key(1);
71      
72      /**
73       * A key for rendering hints that can be used with JFreeChart (in 
74       * addition to the regular Java2D rendering hints).
75       */
76      public static class Key extends java.awt.RenderingHints.Key {
77  
78          /**
79           * Creates a new key.
80           * 
81           * @param privateKey  the private key. 
82           */
83          public Key(int privateKey) {
84              super(privateKey);    
85          }
86      
87          /**
88           * Returns {@code true} if {@code val} is a value that is
89           * compatible with this key, and {@code false} otherwise.
90           * 
91           * @param val  the value.
92           * 
93           * @return A boolean. 
94           */
95          @Override
96          public boolean isCompatibleValue(Object val) {
97              switch (intKey()) {
98                  case 0:
99                      return val == null || val instanceof String 
100                             || val instanceof Map;
101                 case 1:
102                     return val == null || val instanceof Object;
103                 default:
104                     throw new RuntimeException("Not possible!");
105             }
106         }
107     }
108     
109 }
