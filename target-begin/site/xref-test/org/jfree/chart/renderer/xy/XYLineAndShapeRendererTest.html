182         assertTrue(r1.equals(r2));
183

185         assertFalse(r1.equals(r2));
186         r2.setSeriesLinesVisible(0, false);
187         assertTrue(r1.equals(r2));
188 
189         legendShape.setRect(4.0, 3.0, 2.0, 1.0);
190         assertFalse(r1.equals(r2));

192         assertTrue(r1.equals(r2));
193

265     /**
266      * A check for the datasetIndex and seriesIndex fields in the LegendItem
267      * returned by the getLegendItem() method.
268      */
269     @Test
270     public void testGetLegendItemSeriesIndex() {
271         XYSeriesCollection d1 = new XYSeriesCollection();
272         XYSeries s1 = new XYSeries("S1");
273         s1.add(1.0, 1.1);
274         XYSeries s2 = new XYSeries("S2");
275         s2.add(1.0, 1.1);
276         d1.addSeries(s1);
277         d1.addSeries(s2);
278 
279         XYSeriesCollection d2 = new XYSeriesCollection();
280         XYSeries s3 = new XYSeries("S3");
281         s3.add(1.0, 1.1);
282         XYSeries s4 = new XYSeries("S4");
283         s4.add(1.0, 1.1);
284         XYSeries s5 = new XYSeries("S5");
285         s5.add(1.0, 1.1);
286         d2.addSeries(s3);
287         d2.addSeries(s4);
288         d2.addSeries(s5);
289 
290         XYLineAndShapeRenderer r = new XYLineAndShapeRenderer();
291         XYPlot plot = new XYPlot(d1, new NumberAxis("x"),
292                 new NumberAxis("y"), r);
293         plot.setDataset(1, d2);
294         /*JFreeChart chart =*/ new JFreeChart(plot);
295         LegendItem li = r.getLegendItem(1, 2);
296         assertEquals("S5", li.getLabel());
297         assertEquals(1, li.getDatasetIndex());
298         assertEquals(2, li.getSeriesIndex());
299     }
