42  package org.jfree.data.time;
43  
44  import org.jfree.chart.date.MonthConstants;
45  import static org.junit.Assert.assertEquals;
46  
47  import org.junit.Test;
48  
49  /**
50   * Tests for the {@link MovingAverage} class.
51   */
52  public class MovingAverageTest {
53  
54      private static final double EPSILON = 0.0000000001;
55  
56      /**
57       * A test for the values calculated from a time series.
58       */
59      @Test
60      public void test1() {
61          TimeSeries source = createDailyTimeSeries1();
62          TimeSeries maverage = MovingAverage.createMovingAverage(source, 
63                  "Moving Average", 3, 3);
64  
65          // the moving average series has 7 items, the first three
66          // days (11, 12, 13 August are skipped)
67          assertEquals(7, maverage.getItemCount());
68          double value = maverage.getValue(0).doubleValue();
69          assertEquals(14.1, value, EPSILON);
70          value = maverage.getValue(1).doubleValue();
71          assertEquals(13.4, value, EPSILON);
72          value = maverage.getValue(2).doubleValue();
73          assertEquals(14.433333333333, value, EPSILON);
74          value = maverage.getValue(3).doubleValue();
75          assertEquals(14.933333333333, value, EPSILON);
76          value = maverage.getValue(4).doubleValue();
77          assertEquals(19.8, value, EPSILON);
78          value = maverage.getValue(5).doubleValue();
79          assertEquals(15.25, value, EPSILON);
80          value = maverage.getValue(6).doubleValue();
81          assertEquals(12.5, value, EPSILON);
82      }
83  
84      /**
85       * Creates a sample series.
86       *
87       * @return A sample series.
88       */
89      private TimeSeries createDailyTimeSeries1() {
90          TimeSeries series = new TimeSeries("Series 1");
91          series.add(new Day(11, MonthConstants.AUGUST, 2003), 11.2);
92          series.add(new Day(13, MonthConstants.AUGUST, 2003), 13.8);
93          series.add(new Day(17, MonthConstants.AUGUST, 2003), 14.1);
94          series.add(new Day(18, MonthConstants.AUGUST, 2003), 12.7);
95          series.add(new Day(19, MonthConstants.AUGUST, 2003), 16.5);
96          series.add(new Day(20, MonthConstants.AUGUST, 2003), 15.6);
97          series.add(new Day(25, MonthConstants.AUGUST, 2003), 19.8);
98          series.add(new Day(27, MonthConstants.AUGUST, 2003), 10.7);
99          series.add(new Day(28, MonthConstants.AUGUST, 2003), 14.3);
100         return series;
101     }
