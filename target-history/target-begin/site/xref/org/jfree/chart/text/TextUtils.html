79      /**
80       * Creates a {@link TextBlock} from a {@code String}.  Line breaks
81       * are added where the {@code String} contains '\n' characters.
82       *
83       * @param text  the text.
84       * @param font  the font.
85       * @param paint  the paint.
86       *
87       * @return A text block.
88       */
89      public static TextBlock createTextBlock(String text, Font font, 
90              Paint paint) {
91          if (text == null) {
92              throw new IllegalArgumentException("Null 'text' argument.");
93          }
94          TextBlock result = new TextBlock();
95          String input = text;
96          boolean moreInputToProcess = (text.length() > 0);
97          int start = 0;
98          while (moreInputToProcess) {
99              int index = input.indexOf("\n");
100             if (index > start) {
101                 String line = input.substring(start, index);
102                 if (index < input.length() - 1) {
103                     result.addLine(line, font, paint);
104                     input = input.substring(index + 1);
105                 }
106                 else {
107                     moreInputToProcess = false;
108                 }
109             }
110             else if (index == start) {
111                 if (index < input.length() - 1) {
112                     input = input.substring(index + 1);
113                 }
114                 else {
115                     moreInputToProcess = false;
116                 }
117             }
118             else {
119                 result.addLine(input, font, paint);
120                 moreInputToProcess = false;
121             }
122         }
123         return result;
124     }

144     /**
145      * Creates a new text block from the given string, breaking the
146      * text into lines so that the {@code maxWidth} value is
147      * respected.
148      *
149      * @param text  the text.
150      * @param font  the font.
151      * @param paint  the paint.
152      * @param maxWidth  the maximum width for each line.
153      * @param maxLines  the maximum number of lines.
154      * @param measurer  the text measurer.
155      *
156      * @return A text block.
157      */
158     public static TextBlock createTextBlock(String text, Font font,
159             Paint paint, float maxWidth, int maxLines, TextMeasurer measurer) {
160 
161         TextBlock result = new TextBlock();
162         BreakIterator iterator = BreakIterator.getLineInstance();
163         iterator.setText(text);
164         int current = 0;
165         int lines = 0;
166         int length = text.length();
167         while (current < length && lines < maxLines) {
168             int next = nextLineBreak(text, current, maxWidth, iterator,
169                     measurer);
170             if (next == BreakIterator.DONE) {
171                 result.addLine(text.substring(current), font, paint);
172                 return result;
173             } else if (next == current) {
174                 next++; // we must take one more character or we'll loop forever
175             }
176             result.addLine(text.substring(current, next), font, paint);
177             lines++;
178             current = next;
179             while (current < text.length()&& text.charAt(current) == '\n') {
180                 current++;
181             }
182         }
183         if (current < length) {
184             TextLine lastLine = result.getLastLine();
185             TextFragment lastFragment = lastLine.getLastTextFragment();
186             String oldStr = lastFragment.getText();
187             String newStr = "...";
188             if (oldStr.length() > 3) {
189                 newStr = oldStr.substring(0, oldStr.length() - 3) + "...";
190             }
191 
192             lastLine.removeFragment(lastFragment);
193             TextFragment newFragment = new TextFragment(newStr,
194                     lastFragment.getFont(), lastFragment.getPaint());
195             lastLine.addFragment(newFragment);
196         }
197         return result;
198     }

200     /**
201      * Returns the character index of the next line break.  If the next
202      * character is wider than {@code width]} this method will return
203      * {@code start} - the caller should check for this case.
204      *
205      * @param text  the text ({@code null} not permitted).
206      * @param start  the start index.
207      * @param width  the target display width.
208      * @param iterator  the word break iterator.
209      * @param measurer  the text measurer.
210      *
211      * @return The index of the next line break.
212      */
213     private static int nextLineBreak(String text, int start, float width, 
214             BreakIterator iterator, TextMeasurer measurer) {
215 
216         // this method is (loosely) based on code in JFreeReport's
217         // TextParagraph class
218         int current = start;
219         int end;
220         float x = 0.0f;
221         boolean firstWord = true;
222         int newline = text.indexOf('\n', start);
223         if (newline < 0) {
224             newline = Integer.MAX_VALUE;
225         }
226         while (((end = iterator.following(current)) != BreakIterator.DONE)) {
227             x += measurer.getStringWidth(text, current, end);
228             if (x > width) {
229                 if (firstWord) {
230                     while (measurer.getStringWidth(text, start, end) > width) {
231                         end--;
232                         if (end <= start) {
233                             return end;
234                         }
235                     }
236                     return end;
237                 }
238                 else {
239                     end = iterator.previous();
240                     return end;
241                 }
242             }
243             else {
244                 if (end > newline) {
245                     return newline;
246                 }
247             }
248             // we found at least one word that fits ...
249             firstWord = false;
250             current = end;
251         }
252         return BreakIterator.DONE;
253     }

255     /**
256      * Returns the bounds for the specified text.
257      *
258      * @param text  the text ({@code null} permitted).
259      * @param g2  the graphics context (not {@code null}).
260      * @param fm  the font metrics (not {@code null}).
261      *
262      * @return The text bounds ({@code null} if the {@code text}
263      *         argument is {@code null}).
264      */
265     public static Rectangle2D getTextBounds(String text, Graphics2D g2, 
266             FontMetrics fm) {
267 
268         Rectangle2D bounds;
269         if (TextUtils.useFontMetricsGetStringBounds) {
270             bounds = fm.getStringBounds(text, g2);
271             // getStringBounds() can return incorrect height for some Unicode
272             // characters...see bug parade 6183356, let's replace it with
273             // something correct
274             LineMetrics lm = fm.getFont().getLineMetrics(text,
275                     g2.getFontRenderContext());
276             bounds.setRect(bounds.getX(), bounds.getY(), bounds.getWidth(),
277                     lm.getHeight());
278         }
279         else {
280             double width = fm.stringWidth(text);
281             double height = fm.getHeight();
282             bounds = new Rectangle2D.Double(0.0, -fm.getAscent(), width,
283                     height);
284         }
285         return bounds;
286     }

289     /**
290      * Returns the bounds of an aligned string.
291      * 
292      * @param text  the string ({@code null} not permitted).
293      * @param g2  the graphics target ({@code null} not permitted).
294      * @param x  the x-coordinate.
295      * @param y  the y-coordinate.
296      * @param anchor  the anchor point that will be aligned to 
297      *     {@code (x, y)} ({@code null} not permitted).
298      * 
299      * @return The text bounds (never {@code null}).
300      * 
301      * @since 1.3
302      */
303     public static Rectangle2D calcAlignedStringBounds(String text,
304             Graphics2D g2, float x, float y, TextAnchor anchor) {
305 
306         Rectangle2D textBounds = new Rectangle2D.Double();
307         float[] adjust = deriveTextBoundsAnchorOffsets(g2, text, anchor,
308                 textBounds);
309         // adjust text bounds to match string position
310         textBounds.setRect(x + adjust[0], y + adjust[1] + adjust[2],
311             textBounds.getWidth(), textBounds.getHeight());
312         return textBounds;
313     }

315     /**
316      * Draws a string such that the specified anchor point is aligned to the
317      * given (x, y) location.
318      *
319      * @param text  the text.
320      * @param g2  the graphics device.
321      * @param x  the x coordinate (Java 2D).
322      * @param y  the y coordinate (Java 2D).
323      * @param anchor  the anchor location.
324      *
325      * @return The text bounds (adjusted for the text position).
326      */
327     public static Rectangle2D drawAlignedString(String text, Graphics2D g2, 
328             float x, float y, TextAnchor anchor) {
329 
330         Rectangle2D textBounds = new Rectangle2D.Double();
331         float[] adjust = deriveTextBoundsAnchorOffsets(g2, text, anchor,
332                 textBounds);
333         // adjust text bounds to match string position
334         textBounds.setRect(x + adjust[0], y + adjust[1] + adjust[2],
335             textBounds.getWidth(), textBounds.getHeight());
336         if (!drawStringsWithFontAttributes) {
337             g2.drawString(text, x + adjust[0], y + adjust[1]);
338         } else {
339             AttributedString as = new AttributedString(text, 
340                     g2.getFont().getAttributes());
341             g2.drawString(as.getIterator(), x + adjust[0], y + adjust[1]);
342         }
343         return textBounds;
344     }

346     /**
347      * A utility method that calculates the anchor offsets for a string.
348      * Normally, the (x, y) coordinate for drawing text is a point on the
349      * baseline at the left of the text string.  If you add these offsets to
350      * (x, y) and draw the string, then the anchor point should coincide with
351      * the (x, y) point.
352      *
353      * @param g2  the graphics device (not {@code null}).
354      * @param text  the text.
355      * @param anchor  the anchor point.
356      * @param textBounds  the text bounds (if not {@code null}, this
357      *                    object will be updated by this method to match the
358      *                    string bounds).
359      *
360      * @return  The offsets.
361      */
362     private static float[] deriveTextBoundsAnchorOffsets(Graphics2D g2,
363             String text, TextAnchor anchor, Rectangle2D textBounds) {
364 
365         float[] result = new float[3];
366         FontRenderContext frc = g2.getFontRenderContext();
367         Font f = g2.getFont();
368         FontMetrics fm = g2.getFontMetrics(f);
369         Rectangle2D bounds = TextUtils.getTextBounds(text, g2, fm);
370         LineMetrics metrics = f.getLineMetrics(text, frc);
371         float ascent = metrics.getAscent();
372         result[2] = -ascent;
373         float halfAscent = ascent / 2.0f;
374         float descent = metrics.getDescent();
375         float leading = metrics.getLeading();
376         float xAdj = 0.0f;
377         float yAdj = 0.0f;
378 
379         if (anchor.isHorizontalCenter()) {
380             xAdj = (float) -bounds.getWidth() / 2.0f;
381         }
382         else if (anchor.isRight()) {
383             xAdj = (float) -bounds.getWidth();
384         }
385 
386         if (anchor.isTop()) {
387             yAdj = -descent - leading + (float) bounds.getHeight();
388         }
389         else if (anchor.isHalfAscent()) {
390             yAdj = halfAscent;
391         }
392         else if (anchor.isVerticalCenter()) {
393             yAdj = -descent - leading + (float) (bounds.getHeight() / 2.0);
394         }
395         else if (anchor.isBaseline()) {
396             yAdj = 0.0f;
397         }
398         else if (anchor.isBottom()) {
399             yAdj = -metrics.getDescent() - metrics.getLeading();
400         }
401         if (textBounds != null) {
402             textBounds.setRect(bounds);
403         }
404         result[0] = xAdj;
405         result[1] = yAdj;
406         return result;
407 
408     }

427     /**
428      * A utility method for drawing rotated text.
429      * <P>
430      * A common rotation is -Math.PI/2 which draws text 'vertically' (with the
431      * top of the characters on the left).
432      *
433      * @param text  the text.
434      * @param g2  the graphics device.
435      * @param textX  the x-coordinate for the text (before rotation).
436      * @param textY  the y-coordinate for the text (before rotation).
437      * @param angle  the angle of the (clockwise) rotation (in radians).
438      * @param rotateX  the point about which the text is rotated.
439      * @param rotateY  the point about which the text is rotated.
440      */
441     public static void drawRotatedString(String text, Graphics2D g2,
442             float textX, float textY, 
443             double angle, float rotateX, float rotateY) {
444 
445         if ((text == null) || (text.equals(""))) {
446             return;
447         }
448         if (angle == 0.0) {
449             drawAlignedString(text, g2, textX, textY, TextAnchor.BASELINE_LEFT);
450             return;
451         }
452         
453         AffineTransform saved = g2.getTransform();
454         AffineTransform rotate = AffineTransform.getRotateInstance(
455                 angle, rotateX, rotateY);
456         g2.transform(rotate);
457 
458         if (useDrawRotatedStringWorkaround) {
459             // workaround for JDC bug ID 4312117 and others...
460             TextLayout tl = new TextLayout(text, g2.getFont(),
461                     g2.getFontRenderContext());
462             tl.draw(g2, textX, textY);
463         }
464         else {
465             if (!drawStringsWithFontAttributes) {
466                 g2.drawString(text, textX, textY);
467             } else {
468                 AttributedString as = new AttributedString(text, 
469                         g2.getFont().getAttributes());
470                 g2.drawString(as.getIterator(), textX, textY);
471             }
472         }
473         g2.setTransform(saved);
474 
475     }

478      * Draws a string that is aligned by one anchor point and rotated about
479      * another anchor point.
480      *
481      * @param text  the text.
482      * @param g2  the graphics device.
483      * @param x  the x-coordinate for positioning the text.
484      * @param y  the y-coordinate for positioning the text.
485      * @param textAnchor  the text anchor.
486      * @param angle  the rotation angle.
487      * @param rotationX  the x-coordinate for the rotation anchor point.
488      * @param rotationY  the y-coordinate for the rotation anchor point.
489      */
490     public static void drawRotatedString(String text, Graphics2D g2, 
491             float x, float y, TextAnchor textAnchor, 
492             double angle, float rotationX, float rotationY) {
493 
494         if (text == null || text.equals("")) {
495             return;
496         }
497         if (angle == 0.0) {
498             drawAlignedString(text, g2, x, y, textAnchor);
499         } else {
500             float[] textAdj = deriveTextBoundsAnchorOffsets(g2, text, 
501                     textAnchor);
502             drawRotatedString(text, g2, x + textAdj[0], y + textAdj[1], angle,
503                     rotationX, rotationY);
504         }
505     }

510      *
511      * @param text  the text.
512      * @param g2  the graphics device.
513      * @param x  the x-coordinate for positioning the text.

570     /**
571      * A utility method that calculates the anchor offsets for a string.
572      * Normally, the (x, y) coordinate for drawing text is a point on the
573      * baseline at the left of the text string.  If you add these offsets to
574      * (x, y) and draw the string, then the anchor point should coincide with
575      * the (x, y) point.
576      *
577      * @param g2  the graphics device (not {@code null}).
578      * @param text  the text.
579      * @param anchor  the anchor point.
580      *
581      * @return  The offsets.
582      */
583     private static float[] deriveTextBoundsAnchorOffsets(Graphics2D g2,
584             String text, TextAnchor anchor) {
585 
586         float[] result = new float[2];
587         FontRenderContext frc = g2.getFontRenderContext();
588         Font f = g2.getFont();
589         FontMetrics fm = g2.getFontMetrics(f);
590         Rectangle2D bounds = getTextBounds(text, g2, fm);
591         LineMetrics metrics = f.getLineMetrics(text, frc);
592         float ascent = metrics.getAscent();
593         float halfAscent = ascent / 2.0f;
594         float descent = metrics.getDescent();
595         float leading = metrics.getLeading();
596         float xAdj = 0.0f;
597         float yAdj = 0.0f;
598 
599         if (anchor.isHorizontalCenter()) {
600             xAdj = (float) -bounds.getWidth() / 2.0f;
601         }
602         else if (anchor.isRight()) {
603             xAdj = (float) -bounds.getWidth();
604         }
605 
606         if (anchor.isTop()) {
607             yAdj = -descent - leading + (float) bounds.getHeight();
608         }
609         else if (anchor.isHalfAscent()) {
610             yAdj = halfAscent;
611         }
612         else if (anchor.isVerticalCenter()) {
613             yAdj = -descent - leading + (float) (bounds.getHeight() / 2.0);
614         }
615         else if (anchor.isBaseline()) {
616             yAdj = 0.0f;
617         }
618         else if (anchor.isBottom()) {
619             yAdj = -metrics.getDescent() - metrics.getLeading();
620         }
621         result[0] = xAdj;
622         result[1] = yAdj;
623         return result;
624 
625     }

627     /**
628      * A utility method that calculates the rotation anchor offsets for a
629      * string.  These offsets are relative to the text starting coordinate
630      * ({@code BASELINE_LEFT}).
631      *
632      * @param g2  the graphics device.
633      * @param text  the text.
634      * @param anchor  the anchor point.
635      *
636      * @return The offsets.
637      */
638     private static float[] deriveRotationAnchorOffsets(Graphics2D g2,
639             String text, TextAnchor anchor) {
640 
641         float[] result = new float[2];
642         FontRenderContext frc = g2.getFontRenderContext();
643         LineMetrics metrics = g2.getFont().getLineMetrics(text, frc);
644         FontMetrics fm = g2.getFontMetrics();
645         Rectangle2D bounds = TextUtils.getTextBounds(text, g2, fm);
646         float ascent = metrics.getAscent();
647         float halfAscent = ascent / 2.0f;
648         float descent = metrics.getDescent();
649         float leading = metrics.getLeading();
650         float xAdj = 0.0f;
651         float yAdj = 0.0f;
652 
653         if (anchor.isLeft()) {
654             xAdj = 0.0f;
655         }
656         else if (anchor.isHorizontalCenter()) {
657             xAdj = (float) bounds.getWidth() / 2.0f;
658         }
659         else if (anchor.isRight()) {
660             xAdj = (float) bounds.getWidth();
661         }
662 
663         if (anchor.isTop()) {
664             yAdj = descent + leading - (float) bounds.getHeight();
665         }
666         else if (anchor.isVerticalCenter()) {
667             yAdj = descent + leading - (float) (bounds.getHeight() / 2.0);
668         }
669         else if (anchor.isHalfAscent()) {
670             yAdj = -halfAscent;
671         }
672         else if (anchor.isBaseline()) {
673             yAdj = 0.0f;
674         }
675         else if (anchor.isBottom()) {
676             yAdj = metrics.getDescent() + metrics.getLeading();
677         }
678         result[0] = xAdj;
679         result[1] = yAdj;
680         return result;
681 
682     }

686      * specified rotation has been applied.
687      *
688      * @param text  the text ({@code null} permitted).
689      * @param g2  the graphics device.
690      * @param textX  the x coordinate for the text.
691      * @param textY  the y coordinate for the text.
692      * @param angle  the angle.
693      * @param rotateX  the x coordinate for the rotation point.
694      * @param rotateY  the y coordinate for the rotation point.
695      *
696      * @return The bounds ({@code null} if {@code text} is
697      *         {@code null} or has zero length).
698      */
699     public static Shape calculateRotatedStringBounds(String text, Graphics2D g2,
700             float textX, float textY, double angle, float rotateX, 
701             float rotateY) {
702 
703         if ((text == null) || (text.equals(""))) {
704             return null;
705         }
706         FontMetrics fm = g2.getFontMetrics();
707         Rectangle2D bounds = TextUtils.getTextBounds(text, g2, fm);
708         AffineTransform translate = AffineTransform.getTranslateInstance(
709                 textX, textY);
710         Shape translatedBounds = translate.createTransformedShape(bounds);
711         AffineTransform rotate = AffineTransform.getRotateInstance(
712                 angle, rotateX, rotateY);
713         Shape result = rotate.createTransformedShape(translatedBounds);
714         return result;
715 
716     }
