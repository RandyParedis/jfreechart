41  package org.jfree.chart.util;
42  
43  import java.awt.GradientPaint;
44  import java.awt.geom.Rectangle2D;
45  import java.awt.Shape;
46  import org.jfree.chart.ui.GradientPaintTransformer;
47  
48  /**
49   * Transforms a {@code GradientPaint} to range over the width of a target 
50   * shape.  The orientation of the resulting {@code GradientPaint}
51   * depend on the coordinates of the original paint:
52   *
53   * <ul>
54   * <li> If the original paint starts at 0,0 and ends at a point 0, y != 0,
55   * the resulting paint will have a vertical orientation.
56   * <li> If the original paint starts at 0,0 and ends at a point x !=0, 0,
57   * the resulting paint will have a horizontal orientation.
58   * <li> If the original paint starts at 0,0 and ends at a point x != 0, y != 0,
59   * the resulting paint will have a diagonal orientation from the upper left to
60   * the lower right edge. Lines of equal color will have a 45 ∞ angle,
61   * pointing upwards from left to right.
62   * <li> If the original paint starts at a point x != 0, y != 0,
63   * the resulting paint will have a diagonal orientation from the lower left to
64   * the upper right edge. Lines of equal color will have a 45 ∞ angle,
65   * pointing downwards from left to right.
66   * </ul>
67   * <p>In all cases, the cyclic flag of the original paint will be taken into 
68   * account.</p>
69   *
70   * @author Peter Kolb
71   * @since 1.0.17
72   */
73  public class DirectionalGradientPaintTransformer 
74          implements GradientPaintTransformer {
75      
76      /**
77       * Default constructor.
78       */
79      public DirectionalGradientPaintTransformer() {
80          super();    
81      }
82      
83      /**
84       * Transforms a {@code GradientPaint} instance to fit some target 
85       * shape.
86       * 
87       * @param paint  the original paint (not {@code null}).
88       * @param target  the reference area (not {@code null}).
89       * 
90       * @return A transformed paint.
91       */
92      @Override
93      public GradientPaint transform(GradientPaint paint, Shape target) {
94          //get the coordinates of the original GradientPaint
95          final double px1 = paint.getPoint1().getX();
96          final double py1 = paint.getPoint1().getY();
97          final double px2 = paint.getPoint2().getX();
98          final double py2 = paint.getPoint2().getY();
99          //get the coordinates of the shape that is to be filled
100         final Rectangle2D bounds = target.getBounds();
101         final float bx = (float)bounds.getX();
102         final float by = (float)bounds.getY();
103         final float bw = (float)bounds.getWidth();
104         final float bh = (float)bounds.getHeight();
105         //reserve variables to store the coordinates of the resulting GradientPaint
106         float rx1, ry1, rx2, ry2;
107         if (px1 == 0 && py1 == 0) {
108             //start point is upper left corner
109             rx1 = bx;
110             ry1 = by;
111             if (px2 != 0.0f && py2 != 0.0f) {
112                 //end point is lower right corner --> diagonal gradient
113                 float offset = (paint.isCyclic()) ? (bw + bh) / 4.0f 
114                         : (bw + bh) / 2.0f ;
115                 rx2 = bx + offset;
116                 ry2 = by + offset;
117             }
118             else {
119                 //end point is either lower left corner --> vertical gradient
120                 //or end point is upper right corner --> horizontal gradient
121                 rx2 = (px2 == 0) ? rx1 : (paint.isCyclic() ? (rx1 + bw / 2.0f) 
122                         : (rx1 + bw));
123                 ry2 = (py2 == 0) ? ry1 : (paint.isCyclic() ? (ry1 + bh / 2.0f) 
124                         : (ry1 + bh));
125             }
126         }
127         else {
128             //start point is lower left right corner --> diagonal gradient
129             rx1 = bx;
130             ry1 = by + bh;
131             float offset = (paint.isCyclic()) ? (bw + bh) / 4.0f 
132                     : (bw + bh) / 2.0f;
133             rx2 = bx + offset;
134             ry2 = by + bh - offset;
135         }
136         return new GradientPaint(rx1, ry1, paint.getColor1(), rx2, ry2, 
137                 paint.getColor2(), paint.isCyclic());
138     }
