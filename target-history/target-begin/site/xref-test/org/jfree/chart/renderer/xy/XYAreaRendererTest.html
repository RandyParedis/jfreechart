146     /**
147      * Confirm that cloning works.
148      */
149     @Test
150     public void testCloning() throws CloneNotSupportedException {
151         XYAreaRenderer r1 = new XYAreaRenderer();
152         Rectangle2D rect1 = new Rectangle2D.Double(1.0, 2.0, 3.0, 4.0);
153         r1.setLegendArea(rect1);
154         XYAreaRenderer r2 = (XYAreaRenderer) r1.clone();
155         assertTrue(r1 != r2);
156         assertTrue(r1.getClass() == r2.getClass());
157         assertTrue(r1.equals(r2));
158 
159         // check independence
160         rect1.setRect(4.0, 3.0, 2.0, 1.0);
161         assertFalse(r1.equals(r2));
162         r2.setLegendArea(new Rectangle2D.Double(4.0, 3.0, 2.0, 1.0));
163         assertTrue(r1.equals(r2));
164     }

219     /**
220      * A check for the datasetIndex and seriesIndex fields in the LegendItem
221      * returned by the getLegendItem() method.
222      */
223     @Test
224     public void testGetLegendItemSeriesIndex() {
225         XYSeriesCollection d1 = new XYSeriesCollection();
226         XYSeries s1 = new XYSeries("S1");
227         s1.add(1.0, 1.1);
228         XYSeries s2 = new XYSeries("S2");
229         s2.add(1.0, 1.1);
230         d1.addSeries(s1);
231         d1.addSeries(s2);
232 
233         XYSeriesCollection d2 = new XYSeriesCollection();
234         XYSeries s3 = new XYSeries("S3");
235         s3.add(1.0, 1.1);
236         XYSeries s4 = new XYSeries("S4");
237         s4.add(1.0, 1.1);
238         XYSeries s5 = new XYSeries("S5");
239         s5.add(1.0, 1.1);
240         d2.addSeries(s3);
241         d2.addSeries(s4);
242         d2.addSeries(s5);
243 
244         XYAreaRenderer r = new XYAreaRenderer();
245         XYPlot plot = new XYPlot(d1, new NumberAxis("x"),
246                 new NumberAxis("y"), r);
247         plot.setDataset(1, d2);
248         /*JFreeChart chart =*/ new JFreeChart(plot);
249         LegendItem li = r.getLegendItem(1, 2);
250         assertEquals("S5", li.getLabel());
251         assertEquals(1, li.getDatasetIndex());
252         assertEquals(2, li.getSeriesIndex());
253     }
