29  package org.jfree.chart.ui;
30  
31  import java.awt.geom.Rectangle2D;
32  
33  /**
34   * A utility class for aligning rectangles.
35   */
36  public final class Align {
37  
38      /** Center alignment. */
39      public static final int CENTER = 0x00;
40  
41      /** Top alignment. */
42      public static final int TOP = 0x01;
43  
44      /** Bottom alignment. */
45      public static final int BOTTOM = 0x02;
46  
47      /** Left alignment. */
48      public static final int LEFT = 0x04;
49  
50      /** Right alignment. */
51      public static final int RIGHT = 0x08;
52  
53      /** Top/Left alignment. */
54      public static final int TOP_LEFT = TOP | LEFT;
55  
56      /** Top/Right alignment. */
57      public static final int TOP_RIGHT = TOP | RIGHT;
58  
59      /** Bottom/Left alignment. */
60      public static final int BOTTOM_LEFT = BOTTOM | LEFT;
61  
62      /** Bottom/Right alignment. */
63      public static final int BOTTOM_RIGHT = BOTTOM | RIGHT;
64  
65      /** Horizontal fit. */
66      public static final int FIT_HORIZONTAL = LEFT | RIGHT;
67  
68      /** Vertical fit. */
69      public static final int FIT_VERTICAL = TOP | BOTTOM;
70  
71      /** Complete fit. */
72      public static final int FIT = FIT_HORIZONTAL | FIT_VERTICAL;
73  
74      /** North alignment (same as TOP). */
75      public static final int NORTH = TOP;
76  
77      /** South alignment (same as BOTTOM). */
78      public static final int SOUTH = BOTTOM;
79  
80      /** West alignment (same as LEFT). */
81      public static final int WEST = LEFT;
82  
83      /** East alignment (same as RIGHT). */
84      public static final int EAST = RIGHT;
85  
86      /** North/West alignment (same as TOP_LEFT). */
87      public static final int NORTH_WEST = NORTH | WEST;
88  
89      /** North/East alignment (same as TOP_RIGHT). */
90      public static final int NORTH_EAST = NORTH | EAST;
91  
92      /** South/West alignment (same as BOTTOM_LEFT). */
93      public static final int SOUTH_WEST = SOUTH | WEST;
94  
95      /** South/East alignment (same as BOTTOM_RIGHT). */
96      public static final int SOUTH_EAST = SOUTH | EAST;
97  
98      /**
99       * Private constructor.
100      */
101     private Align() { 
102         super();
103     }
104     
105     /**
106      * Aligns one rectangle ({@code rect}) relative to another rectangle ({@code frame}).
107      *
108      * @param rect  the rectangle to be aligned ({@code null} not permitted).
109      * @param frame  the reference frame ({@code null} not permitted).
110      * @param align  the alignment code.
111      */
112     public static void align(Rectangle2D rect, Rectangle2D frame, int align) {
113 
114         double x = frame.getCenterX() - rect.getWidth() / 2.0;
115         double y = frame.getCenterY() - rect.getHeight() / 2.0;
116         double w = rect.getWidth();
117         double h = rect.getHeight();
118 
119         if ((align & FIT_VERTICAL) == FIT_VERTICAL) {
120             h = frame.getHeight();
121         }
122 
123         if ((align & FIT_HORIZONTAL) == FIT_HORIZONTAL) {
124             w = frame.getWidth();
125         }
126 
127         if ((align & TOP) == TOP) {
128             y = frame.getMinY();
129         }
130 
131         if ((align & BOTTOM) == BOTTOM) {
132             y = frame.getMaxY() - h;
133         }
134 
135         if ((align & LEFT) == LEFT) {
136             x = frame.getX();
137         }
138 
139         if ((align & RIGHT) == RIGHT) {
140             x = frame.getMaxX() - w;
141         }
142 
143         rect.setRect(x, y, w, h);
144 
145     }
146 
147 }
