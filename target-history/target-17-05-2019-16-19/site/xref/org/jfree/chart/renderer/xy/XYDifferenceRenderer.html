120  * A renderer for an {@link XYPlot} that highlights the differences between two

350     /**
351      * Draws the visual representation of a single data item.
352      *
353      * @param g2  the graphics device.
354      * @param state  the renderer state.
355      * @param dataArea  the area within which the data is being drawn.
356      * @param info  collects information about the drawing.
357      * @param plot  the plot (can be used to obtain standard color
358      *              information etc).
359      * @param domainAxis  the domain (horizontal) axis.
360      * @param rangeAxis  the range (vertical) axis.
361      * @param dataset  the dataset.
362      * @param series  the series index (zero-based).
363      * @param item  the item index (zero-based).
364      * @param crosshairState  crosshair information for the plot
365      *                        ({@code null} permitted).
366      * @param pass  the pass index.
367      */
368     @Override
369     public void drawItem(Graphics2D g2, XYItemRendererState state,
370             Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
371             ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
372             int series, int item, CrosshairState crosshairState, int pass) {
373 
374         if (pass == 0) {
375             drawItemPass0(g2, dataArea, info, plot, domainAxis, rangeAxis,
376                     dataset, series, item, crosshairState);
377         }
378         else if (pass == 1) {
379             drawItemPass1(g2, dataArea, info, plot, domainAxis, rangeAxis,
380                     dataset, series, item, crosshairState);
381         }
382 
383     }

385     /**
386      * Draws the visual representation of a single data item, first pass.
387      *
388      * @param x_graphics  the graphics device.
389      * @param x_dataArea  the area within which the data is being drawn.
390      * @param x_info  collects information about the drawing.
391      * @param x_plot  the plot (can be used to obtain standard color
392      *                information etc).
393      * @param x_domainAxis  the domain (horizontal) axis.
394      * @param x_rangeAxis  the range (vertical) axis.
395      * @param x_dataset  the dataset.
396      * @param x_series  the series index (zero-based).
397      * @param x_item  the item index (zero-based).
398      * @param x_crosshairState  crosshair information for the plot
399      *                          ({@code null} permitted).
400      */
401     protected void drawItemPass0(Graphics2D x_graphics,
402                                  Rectangle2D x_dataArea,
403                                  PlotRenderingInfo x_info,
404                                  XYPlot x_plot,
405                                  ValueAxis x_domainAxis,
406                                  ValueAxis x_rangeAxis,
407                                  XYDataset x_dataset,
408                                  int x_series,
409                                  int x_item,
410                                  CrosshairState x_crosshairState) {
411 
412         if (!((0 == x_series) && (0 == x_item))) {
413             return;
414         }
415 
416         boolean b_impliedZeroSubtrahend = (1 == x_dataset.getSeriesCount());
417 
418         // check if either series is a degenerate case (i.e. less than 2 points)
419         if (isEitherSeriesDegenerate(x_dataset, b_impliedZeroSubtrahend)) {
420             return;
421         }
422 
423         // check if series are disjoint (i.e. domain-spans do not overlap)
424         if (!b_impliedZeroSubtrahend && areSeriesDisjoint(x_dataset)) {
425             return;
426         }
427 
428         // polygon definitions
429         LinkedList l_minuendXs    = new LinkedList();
430         LinkedList l_minuendYs    = new LinkedList();
431         LinkedList l_subtrahendXs = new LinkedList();
432         LinkedList l_subtrahendYs = new LinkedList();
433         LinkedList l_polygonXs    = new LinkedList();
434         LinkedList l_polygonYs    = new LinkedList();
435 
436         // state
437         int l_minuendItem      = 0;
438         int l_minuendItemCount = x_dataset.getItemCount(0);
439         Double l_minuendCurX   = null;
440         Double l_minuendNextX  = null;
441         Double l_minuendCurY   = null;
442         Double l_minuendNextY  = null;
443         double l_minuendMaxY   = Double.NEGATIVE_INFINITY;
444         double l_minuendMinY   = Double.POSITIVE_INFINITY;
445 
446         int l_subtrahendItem      = 0;
447         int l_subtrahendItemCount = 0; // actual value set below
448         Double l_subtrahendCurX   = null;
449         Double l_subtrahendNextX  = null;
450         Double l_subtrahendCurY   = null;
451         Double l_subtrahendNextY  = null;
452         double l_subtrahendMaxY   = Double.NEGATIVE_INFINITY;
453         double l_subtrahendMinY   = Double.POSITIVE_INFINITY;
454 
455         // if a subtrahend is not specified, assume it is zero
456         if (b_impliedZeroSubtrahend) {
457             l_subtrahendItem      = 0;
458             l_subtrahendItemCount = 2;
459             l_subtrahendCurX      = new Double(x_dataset.getXValue(0, 0));
460             l_subtrahendNextX     = new Double(x_dataset.getXValue(0,
461                     (l_minuendItemCount - 1)));
462             l_subtrahendCurY      = new Double(0.0);
463             l_subtrahendNextY     = new Double(0.0);
464             l_subtrahendMaxY      = 0.0;
465             l_subtrahendMinY      = 0.0;
466 
467             l_subtrahendXs.add(l_subtrahendCurX);
468             l_subtrahendYs.add(l_subtrahendCurY);
469         }
470         else {
471             l_subtrahendItemCount = x_dataset.getItemCount(1);
472         }
473 
474         boolean b_minuendDone           = false;
475         boolean b_minuendAdvanced       = true;
476         boolean b_minuendAtIntersect    = false;
477         boolean b_minuendFastForward    = false;
478         boolean b_subtrahendDone        = false;
479         boolean b_subtrahendAdvanced    = true;
480         boolean b_subtrahendAtIntersect = false;
481         boolean b_subtrahendFastForward = false;
482         boolean b_colinear              = false;
483 
484         boolean b_positive;
485 
486         // coordinate pairs
487         double l_x1 = 0.0, l_y1 = 0.0; // current minuend point
488         double l_x2 = 0.0, l_y2 = 0.0; // next minuend point
489         double l_x3 = 0.0, l_y3 = 0.0; // current subtrahend point
490         double l_x4 = 0.0, l_y4 = 0.0; // next subtrahend point
491 
492         // fast-forward through leading tails
493         boolean b_fastForwardDone = false;
494         while (!b_fastForwardDone) {
495             // get the x and y coordinates
496             l_x1 = x_dataset.getXValue(0, l_minuendItem);
497             l_y1 = x_dataset.getYValue(0, l_minuendItem);
498             l_x2 = x_dataset.getXValue(0, l_minuendItem + 1);
499             l_y2 = x_dataset.getYValue(0, l_minuendItem + 1);
500 
501             l_minuendCurX  = new Double(l_x1);
502             l_minuendCurY  = new Double(l_y1);
503             l_minuendNextX = new Double(l_x2);
504             l_minuendNextY = new Double(l_y2);
505 
506             if (b_impliedZeroSubtrahend) {
507                 l_x3 = l_subtrahendCurX.doubleValue();
508                 l_y3 = l_subtrahendCurY.doubleValue();
509                 l_x4 = l_subtrahendNextX.doubleValue();
510                 l_y4 = l_subtrahendNextY.doubleValue();
511             }
512             else {
513                 l_x3 = x_dataset.getXValue(1, l_subtrahendItem);
514                 l_y3 = x_dataset.getYValue(1, l_subtrahendItem);
515                 l_x4 = x_dataset.getXValue(1, l_subtrahendItem + 1);
516                 l_y4 = x_dataset.getYValue(1, l_subtrahendItem + 1);
517 
518                 l_subtrahendCurX  = new Double(l_x3);
519                 l_subtrahendCurY  = new Double(l_y3);
520                 l_subtrahendNextX = new Double(l_x4);
521                 l_subtrahendNextY = new Double(l_y4);
522             }
523 
524             if (l_x2 <= l_x3) {
525                 // minuend needs to be fast forwarded
526                 l_minuendItem++;
527                 b_minuendFastForward = true;
528                 continue;
529             }
530 
531             if (l_x4 <= l_x1) {
532                 // subtrahend needs to be fast forwarded
533                 l_subtrahendItem++;
534                 b_subtrahendFastForward = true;
535                 continue;
536             }
537 
538             // check if initial polygon needs to be clipped
539             if ((l_x3 < l_x1) && (l_x1 < l_x4)) {
540                 // project onto subtrahend
541                 double l_slope   = (l_y4 - l_y3) / (l_x4 - l_x3);
542                 l_subtrahendCurX = l_minuendCurX;
543                 l_subtrahendCurY = new Double((l_slope * l_x1)
544                         + (l_y3 - (l_slope * l_x3)));
545 
546                 l_subtrahendXs.add(l_subtrahendCurX);
547                 l_subtrahendYs.add(l_subtrahendCurY);
548             }
549 
550             if ((l_x1 < l_x3) && (l_x3 < l_x2)) {
551                 // project onto minuend
552                 double l_slope = (l_y2 - l_y1) / (l_x2 - l_x1);
553                 l_minuendCurX  = l_subtrahendCurX;
554                 l_minuendCurY  = new Double((l_slope * l_x3)
555                         + (l_y1 - (l_slope * l_x1)));
556 
557                 l_minuendXs.add(l_minuendCurX);
558                 l_minuendYs.add(l_minuendCurY);
559             }
560 
561             l_minuendMaxY    = l_minuendCurY.doubleValue();
562             l_minuendMinY    = l_minuendCurY.doubleValue();
563             l_subtrahendMaxY = l_subtrahendCurY.doubleValue();
564             l_subtrahendMinY = l_subtrahendCurY.doubleValue();
565 
566             b_fastForwardDone = true;
567         }
568 
569         // start of algorithm
570         while (!b_minuendDone && !b_subtrahendDone) {
571             if (!b_minuendDone && !b_minuendFastForward && b_minuendAdvanced) {
572                 l_x1 = x_dataset.getXValue(0, l_minuendItem);
573                 l_y1 = x_dataset.getYValue(0, l_minuendItem);
574                 l_minuendCurX = new Double(l_x1);
575                 l_minuendCurY = new Double(l_y1);
576 
577                 if (!b_minuendAtIntersect) {
578                     l_minuendXs.add(l_minuendCurX);
579                     l_minuendYs.add(l_minuendCurY);
580                 }
581 
582                 l_minuendMaxY = Math.max(l_minuendMaxY, l_y1);
583                 l_minuendMinY = Math.min(l_minuendMinY, l_y1);
584 
585                 l_x2 = x_dataset.getXValue(0, l_minuendItem + 1);
586                 l_y2 = x_dataset.getYValue(0, l_minuendItem + 1);
587                 l_minuendNextX = new Double(l_x2);
588                 l_minuendNextY = new Double(l_y2);
589             }
590 
591             // never updated the subtrahend if it is implied to be zero
592             if (!b_impliedZeroSubtrahend && !b_subtrahendDone
593                     && !b_subtrahendFastForward && b_subtrahendAdvanced) {
594                 l_x3 = x_dataset.getXValue(1, l_subtrahendItem);
595                 l_y3 = x_dataset.getYValue(1, l_subtrahendItem);
596                 l_subtrahendCurX = new Double(l_x3);
597                 l_subtrahendCurY = new Double(l_y3);
598 
599                 if (!b_subtrahendAtIntersect) {
600                     l_subtrahendXs.add(l_subtrahendCurX);
601                     l_subtrahendYs.add(l_subtrahendCurY);
602                 }
603 
604                 l_subtrahendMaxY = Math.max(l_subtrahendMaxY, l_y3);
605                 l_subtrahendMinY = Math.min(l_subtrahendMinY, l_y3);
606 
607                 l_x4 = x_dataset.getXValue(1, l_subtrahendItem + 1);
608                 l_y4 = x_dataset.getYValue(1, l_subtrahendItem + 1);
609                 l_subtrahendNextX = new Double(l_x4);
610                 l_subtrahendNextY = new Double(l_y4);
611             }
612 
613             // deassert b_*FastForward (only matters for 1st time through loop)
614             b_minuendFastForward    = false;
615             b_subtrahendFastForward = false;
616 
617             Double l_intersectX = null;
618             Double l_intersectY = null;
619             boolean b_intersect = false;
620 
621             b_minuendAtIntersect    = false;
622             b_subtrahendAtIntersect = false;
623 
624             // check for intersect
625             if ((l_x2 == l_x4) && (l_y2 == l_y4)) {
626                 // check if line segments are colinear
627                 if ((l_x1 == l_x3) && (l_y1 == l_y3)) {
628                     b_colinear = true;
629                 }
630                 else {
631                     // the intersect is at the next point for both the minuend
632                     // and subtrahend
633                     l_intersectX = new Double(l_x2);
634                     l_intersectY = new Double(l_y2);
635 
636                     b_intersect             = true;
637                     b_minuendAtIntersect    = true;
638                     b_subtrahendAtIntersect = true;
639                  }
640             }
641             else {
642                 // compute common denominator
643                 double l_denominator = ((l_y4 - l_y3) * (l_x2 - l_x1))
644                         - ((l_x4 - l_x3) * (l_y2 - l_y1));
645 
646                 // compute common deltas
647                 double l_deltaY = l_y1 - l_y3;
648                 double l_deltaX = l_x1 - l_x3;
649 
650                 // compute numerators
651                 double l_numeratorA = ((l_x4 - l_x3) * l_deltaY)
652                         - ((l_y4 - l_y3) * l_deltaX);
653                 double l_numeratorB = ((l_x2 - l_x1) * l_deltaY)
654                         - ((l_y2 - l_y1) * l_deltaX);
655 
656                 // check if line segments are colinear
657                 if ((0 == l_numeratorA) && (0 == l_numeratorB)
658                         && (0 == l_denominator)) {
659                     b_colinear = true;
660                 }
661                 else {
662                     // check if previously colinear
663                     if (b_colinear) {
664                         // clear colinear points and flag
665                         l_minuendXs.clear();
666                         l_minuendYs.clear();
667                         l_subtrahendXs.clear();
668                         l_subtrahendYs.clear();
669                         l_polygonXs.clear();
670                         l_polygonYs.clear();
671 
672                         b_colinear = false;
673 
674                         // set new starting point for the polygon
675                         boolean b_useMinuend = ((l_x3 <= l_x1)
676                                 && (l_x1 <= l_x4));
677                         l_polygonXs.add(b_useMinuend ? l_minuendCurX
678                                 : l_subtrahendCurX);
679                         l_polygonYs.add(b_useMinuend ? l_minuendCurY
680                                 : l_subtrahendCurY);
681                     }
682                 }
683 
684                 // compute slope components
685                 double l_slopeA = l_numeratorA / l_denominator;
686                 double l_slopeB = l_numeratorB / l_denominator;
687 
688                 // test if both grahphs have a vertical rise at the same x-value
689                 boolean b_vertical = (l_x1 == l_x2) && (l_x3 == l_x4) && (l_x2 == l_x4);
690 
691                 // check if the line segments intersect
692                 if (((0 < l_slopeA) && (l_slopeA <= 1) && (0 < l_slopeB)
693                         && (l_slopeB <= 1))|| b_vertical) {
694 
695                     // compute the point of intersection
696                     double l_xi;
697                     double l_yi;
698                     if(b_vertical){
699                         b_colinear = false;
700                         l_xi = l_x2;
701                         l_yi = l_x4;
702                     }
703                     else{
704                         l_xi = l_x1 + (l_slopeA * (l_x2 - l_x1));
705                         l_yi = l_y1 + (l_slopeA * (l_y2 - l_y1));
706                     }
707 
708                     l_intersectX            = new Double(l_xi);
709                     l_intersectY            = new Double(l_yi);
710                     b_intersect             = true;
711                     b_minuendAtIntersect    = ((l_xi == l_x2)
712                             && (l_yi == l_y2));
713                     b_subtrahendAtIntersect = ((l_xi == l_x4)
714                             && (l_yi == l_y4));
715 
716                     // advance minuend and subtrahend to intesect
717                     l_minuendCurX    = l_intersectX;
718                     l_minuendCurY    = l_intersectY;
719                     l_subtrahendCurX = l_intersectX;
720                     l_subtrahendCurY = l_intersectY;
721                 }
722             }
723 
724             if (b_intersect) {
725                 // create the polygon
726                 // add the minuend's points to polygon
727                 l_polygonXs.addAll(l_minuendXs);
728                 l_polygonYs.addAll(l_minuendYs);
729 
730                 // add intersection point to the polygon
731                 l_polygonXs.add(l_intersectX);
732                 l_polygonYs.add(l_intersectY);
733 
734                 // add the subtrahend's points to the polygon in reverse
735                 Collections.reverse(l_subtrahendXs);
736                 Collections.reverse(l_subtrahendYs);
737                 l_polygonXs.addAll(l_subtrahendXs);
738                 l_polygonYs.addAll(l_subtrahendYs);
739 
740                 // create an actual polygon
741                 b_positive = (l_subtrahendMaxY <= l_minuendMaxY)
742                         && (l_subtrahendMinY <= l_minuendMinY);
743                 createPolygon(x_graphics, x_dataArea, x_plot, x_domainAxis,
744                         x_rangeAxis, b_positive, l_polygonXs, l_polygonYs);
745 
746                 // clear the point vectors
747                 l_minuendXs.clear();
748                 l_minuendYs.clear();
749                 l_subtrahendXs.clear();
750                 l_subtrahendYs.clear();
751                 l_polygonXs.clear();
752                 l_polygonYs.clear();
753 
754                 // set the maxY and minY values to intersect y-value
755                 double l_y       = l_intersectY.doubleValue();
756                 l_minuendMaxY    = l_y;
757                 l_subtrahendMaxY = l_y;
758                 l_minuendMinY    = l_y;
759                 l_subtrahendMinY = l_y;
760 
761                 // add interection point to new polygon
762                 l_polygonXs.add(l_intersectX);
763                 l_polygonYs.add(l_intersectY);
764             }
765 
766             // advance the minuend if needed
767             if (l_x2 <= l_x4) {
768                 l_minuendItem++;
769                 b_minuendAdvanced = true;
770             }
771             else {
772                 b_minuendAdvanced = false;
773             }
774 
775             // advance the subtrahend if needed
776             if (l_x4 <= l_x2) {
777                 l_subtrahendItem++;
778                 b_subtrahendAdvanced = true;
779             }
780             else {
781                 b_subtrahendAdvanced = false;
782             }
783 
784             b_minuendDone    = (l_minuendItem == (l_minuendItemCount - 1));
785             b_subtrahendDone = (l_subtrahendItem == (l_subtrahendItemCount
786                     - 1));
787         }
788 
789         // check if the final polygon needs to be clipped
790         if (b_minuendDone && (l_x3 < l_x2) && (l_x2 < l_x4)) {
791             // project onto subtrahend
792             double l_slope    = (l_y4 - l_y3) / (l_x4 - l_x3);
793             l_subtrahendNextX = l_minuendNextX;
794             l_subtrahendNextY = new Double((l_slope * l_x2)
795                     + (l_y3 - (l_slope * l_x3)));
796         }
797 
798         if (b_subtrahendDone && (l_x1 < l_x4) && (l_x4 < l_x2)) {
799             // project onto minuend
800             double l_slope = (l_y2 - l_y1) / (l_x2 - l_x1);
801             l_minuendNextX = l_subtrahendNextX;
802             l_minuendNextY = new Double((l_slope * l_x4)
803                     + (l_y1 - (l_slope * l_x1)));
804         }
805 
806         // consider last point of minuend and subtrahend for determining
807         // positivity
808         l_minuendMaxY    = Math.max(l_minuendMaxY,
809                 l_minuendNextY.doubleValue());
810         l_subtrahendMaxY = Math.max(l_subtrahendMaxY,
811                 l_subtrahendNextY.doubleValue());
812         l_minuendMinY    = Math.min(l_minuendMinY,
813                 l_minuendNextY.doubleValue());
814         l_subtrahendMinY = Math.min(l_subtrahendMinY,
815                 l_subtrahendNextY.doubleValue());
816 
817         // add the last point of the minuned and subtrahend
818         l_minuendXs.add(l_minuendNextX);
819         l_minuendYs.add(l_minuendNextY);
820         l_subtrahendXs.add(l_subtrahendNextX);
821         l_subtrahendYs.add(l_subtrahendNextY);
822 
823         // create the polygon
824         // add the minuend's points to polygon
825         l_polygonXs.addAll(l_minuendXs);
826         l_polygonYs.addAll(l_minuendYs);
827 
828         // add the subtrahend's points to the polygon in reverse
829         Collections.reverse(l_subtrahendXs);
830         Collections.reverse(l_subtrahendYs);
831         l_polygonXs.addAll(l_subtrahendXs);
832         l_polygonYs.addAll(l_subtrahendYs);
833 
834         // create an actual polygon
835         b_positive = (l_subtrahendMaxY <= l_minuendMaxY)
836                 && (l_subtrahendMinY <= l_minuendMinY);
837         createPolygon(x_graphics, x_dataArea, x_plot, x_domainAxis,
838                 x_rangeAxis, b_positive, l_polygonXs, l_polygonYs);
839     }

967     /**
968      * Determines if a dataset is degenerate.  A degenerate dataset is a
969      * dataset where either series has less than two (2) points.
970      *
971      * @param x_dataset  the dataset.
972      * @param x_impliedZeroSubtrahend  if false, do not check the subtrahend
973      *
974      * @return true if the dataset is degenerate.
975      */
976     private boolean isEitherSeriesDegenerate(XYDataset x_dataset,
977             boolean x_impliedZeroSubtrahend) {
978 
979         if (x_impliedZeroSubtrahend) {
980             return (x_dataset.getItemCount(0) < 2);
981         }
982 
983         return ((x_dataset.getItemCount(0) < 2)
984                 || (x_dataset.getItemCount(1) < 2));

987     /**
988      * Determines if the two (2) series are disjoint.
989      * Disjoint series do not overlap in the domain space.
990      *
991      * @param x_dataset  the dataset.
992      *
993      * @return true if the dataset is degenerate.
994      */
995     private boolean areSeriesDisjoint(XYDataset x_dataset) {
996 
997         int l_minuendItemCount = x_dataset.getItemCount(0);
998         double l_minuendFirst  = x_dataset.getXValue(0, 0);
999         double l_minuendLast   = x_dataset.getXValue(0, l_minuendItemCount - 1);
1000 
1001         int l_subtrahendItemCount = x_dataset.getItemCount(1);
1002         double l_subtrahendFirst  = x_dataset.getXValue(1, 0);
1003         double l_subtrahendLast   = x_dataset.getXValue(1,
1004                 l_subtrahendItemCount - 1);
1005 
1006         return ((l_minuendLast < l_subtrahendFirst)
1007                 || (l_subtrahendLast < l_minuendFirst));

1010     /**
1011      * Draws the visual representation of a polygon
1012      *
1013      * @param x_graphics  the graphics device.
1014      * @param x_dataArea  the area within which the data is being drawn.
1015      * @param x_plot  the plot (can be used to obtain standard color
1016      *                information etc).
1017      * @param x_domainAxis  the domain (horizontal) axis.
1018      * @param x_rangeAxis  the range (vertical) axis.
1019      * @param x_positive  indicates if the polygon is positive (true) or
1020      *                    negative (false).
1021      * @param x_xValues  a linked list of the x values (expects values to be
1022      *                   of type Double).
1023      * @param x_yValues  a linked list of the y values (expects values to be
1024      *                   of type Double).
1025      */
1026     private void createPolygon (Graphics2D x_graphics,
1027                                 Rectangle2D x_dataArea,
1028                                 XYPlot x_plot,
1029                                 ValueAxis x_domainAxis,
1030                                 ValueAxis x_rangeAxis,
1031                                 boolean x_positive,
1032                                 LinkedList x_xValues,
1033                                 LinkedList x_yValues) {
1034 
1035         PlotOrientation l_orientation      = x_plot.getOrientation();
1036         RectangleEdge l_domainAxisLocation = x_plot.getDomainAxisEdge();
1037         RectangleEdge l_rangeAxisLocation  = x_plot.getRangeAxisEdge();
1038 
1039         Object[] l_xValues = x_xValues.toArray();
1040         Object[] l_yValues = x_yValues.toArray();
1041 
1042         GeneralPath l_path = new GeneralPath();
1043 
1044         if (PlotOrientation.VERTICAL == l_orientation) {
1045             double l_x = x_domainAxis.valueToJava2D((
1046                     (Double) l_xValues[0]).doubleValue(), x_dataArea,
1047                     l_domainAxisLocation);
1048             if (this.roundXCoordinates) {
1049                 l_x = Math.rint(l_x);
1050             }
1051 
1052             double l_y = x_rangeAxis.valueToJava2D((
1053                     (Double) l_yValues[0]).doubleValue(), x_dataArea,
1054                     l_rangeAxisLocation);
1055 
1056             l_path.moveTo((float) l_x, (float) l_y);
1057             for (int i = 1; i < l_xValues.length; i++) {
1058                 l_x = x_domainAxis.valueToJava2D((
1059                         (Double) l_xValues[i]).doubleValue(), x_dataArea,
1060                         l_domainAxisLocation);
1061                 if (this.roundXCoordinates) {
1062                     l_x = Math.rint(l_x);
1063                 }
1064 
1065                 l_y = x_rangeAxis.valueToJava2D((
1066                         (Double) l_yValues[i]).doubleValue(), x_dataArea,
1067                         l_rangeAxisLocation);
1068                 l_path.lineTo((float) l_x, (float) l_y);
1069             }
1070             l_path.closePath();
1071         }
1072         else {
1073             double l_x = x_domainAxis.valueToJava2D((
1074                     (Double) l_xValues[0]).doubleValue(), x_dataArea,
1075                     l_domainAxisLocation);
1076             if (this.roundXCoordinates) {
1077                 l_x = Math.rint(l_x);
1078             }
1079 
1080             double l_y = x_rangeAxis.valueToJava2D((
1081                     (Double) l_yValues[0]).doubleValue(), x_dataArea,
1082                     l_rangeAxisLocation);
1083 
1084             l_path.moveTo((float) l_y, (float) l_x);
1085             for (int i = 1; i < l_xValues.length; i++) {
1086                 l_x = x_domainAxis.valueToJava2D((
1087                         (Double) l_xValues[i]).doubleValue(), x_dataArea,
1088                         l_domainAxisLocation);
1089                 if (this.roundXCoordinates) {
1090                     l_x = Math.rint(l_x);
1091                 }
1092 
1093                 l_y = x_rangeAxis.valueToJava2D((
1094                         (Double) l_yValues[i]).doubleValue(), x_dataArea,
1095                         l_rangeAxisLocation);
1096                 l_path.lineTo((float) l_y, (float) l_x);
1097             }
1098             l_path.closePath();
1099         }
1100 
1101         if (l_path.intersects(x_dataArea)) {
1102             x_graphics.setPaint(x_positive ? getPositivePaint()
1103                     : getNegativePaint());
1104             x_graphics.fill(l_path);
1105         }
1106     }
