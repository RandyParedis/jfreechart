86  /**
87   * A renderer that draws a circle at each data point with a diameter that is
88   * determined by the z-value in the dataset (the renderer requires the dataset
89   * to be an instance of {@link XYZDataset}.  The example shown here
90   * is generated by the {@code XYBubbleChartDemo1.java} program
91   * included in the JFreeChart demo collection:
92   * <br><br>
93   * <img src="../../../../../images/XYBubbleRendererSample.png"
94   * alt="XYBubbleRendererSample.png">
95   */
96  public class XYBubbleRenderer extends AbstractXYItemRenderer
97          implements XYItemRenderer, PublicCloneable {
98

100     public static final long serialVersionUID = -5221991598674249125L;

102     /**
103      * A constant to specify that the bubbles drawn by this renderer should be
104      * scaled on both axes (see {@link #XYBubbleRenderer(int)}).
105      */
106     public static final int SCALE_ON_BOTH_AXES = 0;
107 
108     /**
109      * A constant to specify that the bubbles drawn by this renderer should be
110      * scaled on the domain axis (see {@link #XYBubbleRenderer(int)}).
111      */
112     public static final int SCALE_ON_DOMAIN_AXIS = 1;
113 
114     /**
115      * A constant to specify that the bubbles drawn by this renderer should be
116      * scaled on the range axis (see {@link #XYBubbleRenderer(int)}).
117      */
118     public static final int SCALE_ON_RANGE_AXIS = 2;
119 
120     /** Controls how the width and height of the bubble are scaled. */
121     private int scaleType;

123     /**
124      * Constructs a new renderer.
125      */
126     public XYBubbleRenderer() {
127         this(SCALE_ON_BOTH_AXES);
128     }

130     /**
131      * Constructs a new renderer with the specified type of scaling.
132      *
133      * @param scaleType  the type of scaling (must be one of:
134      *        {@link #SCALE_ON_BOTH_AXES}, {@link #SCALE_ON_DOMAIN_AXIS},
135      *        {@link #SCALE_ON_RANGE_AXIS}).
136      */
137     public XYBubbleRenderer(int scaleType) {
138         super();
139         if (scaleType < 0 || scaleType > 2) {
140             throw new IllegalArgumentException("Invalid 'scaleType'.");
141         }
142         this.scaleType = scaleType;
143         setDefaultLegendShape(new Ellipse2D.Double(-4.0, -4.0, 8.0, 8.0));
144     }

152     public int getScaleType() {

156     /**
157      * Draws the visual representation of a single data item.
158      *
159      * @param g2  the graphics device.
160      * @param state  the renderer state.
161      * @param dataArea  the area within which the data is being drawn.
162      * @param info  collects information about the drawing.
163      * @param plot  the plot (can be used to obtain standard color
164      *              information etc).
165      * @param domainAxis  the domain (horizontal) axis.
166      * @param rangeAxis  the range (vertical) axis.
167      * @param dataset  the dataset (an {@link XYZDataset} is expected).
168      * @param series  the series index (zero-based).
169      * @param item  the item index (zero-based).
170      * @param crosshairState  crosshair information for the plot
171      *                        ({@code null} permitted).
172      * @param pass  the pass index.
173      */
174     @Override
175     public void drawItem(Graphics2D g2, XYItemRendererState state,
176             Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
177             ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
178             int series, int item, CrosshairState crosshairState, int pass) {
179 
180         // return straight away if the item is not visible
181         if (!getVisibility().getItemVisible(series, item)) {
182             return;
183         }
184 
185         PlotOrientation orientation = plot.getOrientation();
186 
187         // get the data point...
188         double x = dataset.getXValue(series, item);
189         double y = dataset.getYValue(series, item);
190         double z = Double.NaN;
191         if (dataset instanceof XYZDataset) {
192             XYZDataset xyzData = (XYZDataset) dataset;
193             z = xyzData.getZValue(series, item);
194         }
195         if (!Double.isNaN(z)) {
196             RectangleEdge domainAxisLocation = plot.getDomainAxisEdge();
197             RectangleEdge rangeAxisLocation = plot.getRangeAxisEdge();
198             double transX = domainAxis.valueToJava2D(x, dataArea,
199                     domainAxisLocation);
200             double transY = rangeAxis.valueToJava2D(y, dataArea,
201                     rangeAxisLocation);
202 
203             double transDomain;
204             double transRange;
205             double zero;
206 
207             switch(getScaleType()) {
208                 case SCALE_ON_DOMAIN_AXIS:
209                     zero = domainAxis.valueToJava2D(0.0, dataArea,
210                             domainAxisLocation);
211                     transDomain = domainAxis.valueToJava2D(z, dataArea,
212                             domainAxisLocation) - zero;
213                     transRange = transDomain;
214                     break;
215                 case SCALE_ON_RANGE_AXIS:
216                     zero = rangeAxis.valueToJava2D(0.0, dataArea,
217                             rangeAxisLocation);
218                     transRange = zero - rangeAxis.valueToJava2D(z, dataArea,
219                             rangeAxisLocation);
220                     transDomain = transRange;
221                     break;
222                 default:
223                     double zero1 = domainAxis.valueToJava2D(0.0, dataArea,
224                             domainAxisLocation);
225                     double zero2 = rangeAxis.valueToJava2D(0.0, dataArea,
226                             rangeAxisLocation);
227                     transDomain = domainAxis.valueToJava2D(z, dataArea,
228                             domainAxisLocation) - zero1;
229                     transRange = zero2 - rangeAxis.valueToJava2D(z, dataArea,
230                             rangeAxisLocation);
231             }
232             transDomain = Math.abs(transDomain);
233             transRange = Math.abs(transRange);
234             Ellipse2D circle = null;
235             if (orientation == PlotOrientation.VERTICAL) {
236                 circle = new Ellipse2D.Double(transX - transDomain / 2.0,
237                         transY - transRange / 2.0, transDomain, transRange);
238             }
239             else if (orientation == PlotOrientation.HORIZONTAL) {
240                 circle = new Ellipse2D.Double(transY - transRange / 2.0,
241                         transX - transDomain / 2.0, transRange, transDomain);
242             } else {
243                 throw new IllegalStateException();
244             }
245             g2.setPaint(getItemPaint(series, item));
246             g2.fill(circle);
247             g2.setStroke(getItemOutlineStroke(series, item));
248             g2.setPaint(getItemOutlinePaint(series, item));
249             g2.draw(circle);
250 
251             if (isItemLabelVisible(series, item)) {
252                 if (orientation == PlotOrientation.VERTICAL) {
253                     drawItemLabel(g2, orientation, dataset, series, item,
254                             transX, transY, false);
255                 }
256                 else if (orientation == PlotOrientation.HORIZONTAL) {
257                     drawItemLabel(g2, orientation, dataset, series, item,
258                             transY, transX, false);
259                 }
260             }
261 
262             // add an entity if this info is being collected
263             if (info != null) {
264                 EntityCollection entities 
265                         = info.getOwner().getEntityCollection();
266                 if (entities != null && circle.intersects(dataArea)) {
267                     addEntity(entities, circle, dataset, series, item,
268                             circle.getCenterX(), circle.getCenterY());
269                 }
270             }
271 
272             int datasetIndex = plot.indexOf(dataset);
273             updateCrosshairValues(crosshairState, x, y, datasetIndex,
274                     transX, transY, orientation);
275         }
276 
277     }

279     /**
280      * Returns a legend item for the specified series.  The default method
281      * is overridden so that the legend displays circles for all series.
282      *
283      * @param datasetIndex  the dataset index (zero-based).
284      * @param series  the series index (zero-based).
285      *
286      * @return A legend item for the series.
287      */
288     @Override
289     public LegendItem getLegendItem(int datasetIndex, int series) {
290         LegendItem result = null;
291         XYPlot plot = getPlot();
292         if (plot == null) {
293             return null;
294         }
295 
296         XYDataset dataset = plot.getDataset(datasetIndex);
297         if (dataset != null) {
298             if (getVisibility().getItemVisible(series, 0)) {
299                 String label = getLegendItemLabelGenerator().generateLabel(
300                         dataset, series);
301                 String description = label;
302                 String toolTipText = null;
303                 if (getLegendItemToolTipGenerator() != null) {
304                     toolTipText = getLegendItemToolTipGenerator().generateLabel(
305                             dataset, series);
306                 }
307                 String urlText = null;
308                 if (getLegendItemURLGenerator() != null) {
309                     urlText = getLegendItemURLGenerator().generateLabel(
310                             dataset, series);
311                 }
312                 Shape shape = lookupLegendShape(series);
313                 Paint paint = lookupSeriesPaint(series);
314                 Paint outlinePaint = lookupSeriesOutlinePaint(series);
315                 Stroke outlineStroke = lookupSeriesOutlineStroke(series);
316                 result = new LegendItem(label, description, toolTipText,
317                         urlText, shape, paint, outlineStroke, outlinePaint);
318                 result.setLabelFont(lookupLegendTextFont(series));
319                 Paint labelPaint = lookupLegendTextPaint(series);
320                 if (labelPaint != null) {
321                     result.setLabelPaint(labelPaint);
322                 }
323                 result.setDataset(dataset);
324                 result.setDatasetIndex(datasetIndex);
325                 result.setSeriesKey(dataset.getSeriesKey(series));
326                 result.setSeriesIndex(series);
327             }
328         }
329         return result;
330     }
