36  package org.jfree.chart.util;
37  
38  /**
39   * A utility class for checking method arguments.
40   */
41  public class Args {
42  
43      /**
44       * Throws an {@code IllegalArgumentException} if the supplied 
45       * {@code param} is {@code null}.
46       *
47       * @param param  the parameter to check ({@code null} permitted).
48       * @param name  the name of the parameter (to use in the exception message
49       *     if {@code param} is {@code null}).
50       *
51       * @throws IllegalArgumentException  if {@code param} is 
52       *     {@code null}.
53       */
54      public static void nullNotPermitted(Object param, String name) {
55          if (param == null) {
56              throw new IllegalArgumentException("Null '" + name + "' argument.");
57          }
58      }
59      
60      /**
61       * Throws an {@code IllegalArgumentException} if {@code value} is negative.
62       * 
63       * @param value  the value.
64       * @param name  the parameter name (for use in the exception message).
65       */
66      public static void requireNonNegative(int value, String name) {
67          if (value < 0) {
68              throw new IllegalArgumentException("Require '" + name + "' (" 
69                      + value + ") to be non-negative.");
70          }
71      }
72      
73      /**
74       * Checks that the value falls within the specified range and, if it does
75       * not, throws an {@code IllegalArgumentException}.
76       * 
77       * @param value  the value.
78       * @param name  the parameter name.
79       * @param lowerBound  the lower bound of the permitted range.
80       * @param upperBound  the upper bound fo the permitted range.
81       */
82      public static void requireInRange(int value, String name, 
83              int lowerBound, int upperBound) {
84          if (value < lowerBound) {
85              throw new IllegalArgumentException("Require '" + name + "' (" 
86                      + value + ") to be in the range " + lowerBound + " to " 
87                      + upperBound);
88          }
89      }
90  }
