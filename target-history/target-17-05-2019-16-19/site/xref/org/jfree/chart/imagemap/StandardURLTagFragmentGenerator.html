43  package org.jfree.chart.imagemap;
44  
45  /**
46   * Generates URLs using the HTML href attribute for image map area tags.
47   */
48  public class StandardURLTagFragmentGenerator
49          implements URLTagFragmentGenerator {
50  
51      /**
52       * Creates a new instance.
53       */
54      public StandardURLTagFragmentGenerator() {
55          super();
56      }
57  
58      /**
59       * Generates a URL string to go in an HTML image map.
60       *
61       * @param urlText  the URL text (fully escaped).
62       *
63       * @return The formatted text
64       */
65      @Override
66      public String generateURLFragment(String urlText) {
67          // the URL text should already have been escaped by the URL generator
68          return " href=\"" + urlText + "\"";
69      }
70  
71  }
