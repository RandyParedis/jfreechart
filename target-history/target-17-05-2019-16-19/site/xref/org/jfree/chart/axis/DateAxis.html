171 /**
172  * The base class for axes that display dates.  You will find it easier to
173  * understand how this axis works if you bear in mind that it really
174  * displays/measures integer (or long) data, where the integers are
175  * milliseconds since midnight, 1-Jan-1970.  When displaying tick labels, the
176  * millisecond values are converted back to dates using a {@code DateFormat} 
177  * instance.
178  * <P>
179  * You can also create a {@link org.jfree.chart.axis.Timeline} and supply in
180  * the constructor to create an axis that only contains certain domain values.
181  * For example, this allows you to create a date axis that only contains
182  * working days.
183  */
184 public class DateAxis extends ValueAxis implements Cloneable, Serializable {

622     /**
623      * Sets the minimum date visible on the axis and sends an
624      * {@link AxisChangeEvent} to all registered listeners.  If
625      * {@code date} is on or after the current maximum date for
626      * the axis, the maximum date will be shifted to preserve the current
627      * length of the axis.
628      *
629      * @param date  the date ({@code null} not permitted).
630      *
631      * @see #getMinimumDate()
632      * @see #setMaximumDate(Date)
633      */
634     public void setMinimumDate(Date date) {
635         Args.nullNotPermitted(date, "date");
636         // check the new minimum date relative to the current maximum date
637         Date maxDate = getMaximumDate();
638         long maxMillis = maxDate.getTime();
639         long newMinMillis = date.getTime();
640         if (maxMillis <= newMinMillis) {
641             Date oldMin = getMinimumDate();
642             long length = maxMillis - oldMin.getTime();
643             maxDate = new Date(newMinMillis + length);
644         }
645         setRange(new DateRange(date, maxDate), true, false);
646         fireChangeEvent();
647     }

671      * Sets the maximum date visible on the axis and sends an
672      * {@link AxisChangeEvent} to all registered listeners.  If
673      * {@code maximumDate} is on or before the current minimum date for
674      * the axis, the minimum date will be shifted to preserve the current
675      * length of the axis.

677      * @param maximumDate  the date ({@code null} not permitted).
678      *
679      * @see #getMinimumDate()
680      * @see #setMinimumDate(Date)

682     public void setMaximumDate(Date maximumDate) {
683         Args.nullNotPermitted(maximumDate, "maximumDate");
684         // check the new maximum date relative to the current minimum date
685         Date minDate = getMinimumDate();
686         long minMillis = minDate.getTime();
687         long newMaxMillis = maximumDate.getTime();
688         if (minMillis >= newMaxMillis) {
689             Date oldMax = getMaximumDate();
690             long length = oldMax.getTime() - minMillis;
691             minDate = new Date(newMaxMillis - length);
692         }
693         setRange(new DateRange(minDate, maximumDate), true, false);
694         fireChangeEvent();

744      *
745      * @param value  the date to be plotted.
746      * @param area  the rectangle (in Java2D space) where the data is to be
747      *              plotted.
748      * @param edge  the axis location.
749      *
750      * @return The coordinate corresponding to the supplied data value.

752     @Override

754             RectangleEdge edge) {
755 
756         value = this.timeline.toTimelineValue((long) value);
757 
758         DateRange range = (DateRange) getRange();
759         double axisMin = this.timeline.toTimelineValue(range.getLowerMillis());
760         double axisMax = this.timeline.toTimelineValue(range.getUpperMillis());
761         double result = 0.0;

763             double minX = area.getX();
764             double maxX = area.getMaxX();
765             if (isInverted()) {
766                 result = maxX + ((value - axisMin) / (axisMax - axisMin))
767                          * (minX - maxX);
768             }
769             else {
770                 result = minX + ((value - axisMin) / (axisMax - axisMin))
771                          * (maxX - minX);

779                          * (maxY - minY));
780             }
781             else {
782                 result = maxY - (((value - axisMin) / (axisMax - axisMin))
783                          * (maxY - minY));
784             }
785         }

880     protected Date previousStandardDate(Date date, DateTickUnit unit) {
881 
882         int milliseconds;
883         int seconds;
884         int minutes;
885         int hours;
886         int days;
887         int months;
888         int years;
889 
890         Calendar calendar = Calendar.getInstance(this.timeZone, this.locale);
891         calendar.setTime(date);
892         int count = unit.getMultiple();
893         int current = calendar.get(unit.getCalendarField());
894         int value = count * (current / count);
895 
896         if (DateTickUnitType.MILLISECOND.equals(unit.getUnitType())) {
897             years = calendar.get(Calendar.YEAR);
898             months = calendar.get(Calendar.MONTH);
899             days = calendar.get(Calendar.DATE);
900             hours = calendar.get(Calendar.HOUR_OF_DAY);
901             minutes = calendar.get(Calendar.MINUTE);
902             seconds = calendar.get(Calendar.SECOND);
903             calendar.set(years, months, days, hours, minutes, seconds);
904             calendar.set(Calendar.MILLISECOND, value);
905             Date mm = calendar.getTime();
906             if (mm.getTime() >= date.getTime()) {
907                 calendar.set(Calendar.MILLISECOND, value - count);
908                 mm = calendar.getTime();
909             }
910             return mm;
911         }
912         else if (DateTickUnitType.SECOND.equals(unit.getUnitType())) {
913             years = calendar.get(Calendar.YEAR);
914             months = calendar.get(Calendar.MONTH);
915             days = calendar.get(Calendar.DATE);
916             hours = calendar.get(Calendar.HOUR_OF_DAY);
917             minutes = calendar.get(Calendar.MINUTE);
918             if (this.tickMarkPosition == DateTickMarkPosition.START) {
919                 milliseconds = 0;
920             }
921             else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
922                 milliseconds = 500;
923             }
924             else {
925                 milliseconds = 999;
926             }
927             calendar.set(Calendar.MILLISECOND, milliseconds);
928             calendar.set(years, months, days, hours, minutes, value);
929             Date dd = calendar.getTime();
930             if (dd.getTime() >= date.getTime()) {
931                 calendar.set(Calendar.SECOND, value - count);
932                 dd = calendar.getTime();
933             }
934             return dd;
935         }
936         else if (DateTickUnitType.MINUTE.equals(unit.getUnitType())) {
937             years = calendar.get(Calendar.YEAR);
938             months = calendar.get(Calendar.MONTH);
939             days = calendar.get(Calendar.DATE);
940             hours = calendar.get(Calendar.HOUR_OF_DAY);
941             if (this.tickMarkPosition == DateTickMarkPosition.START) {
942                 seconds = 0;
943             }
944             else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
945                 seconds = 30;
946             }
947             else {
948                 seconds = 59;
949             }
950             calendar.clear(Calendar.MILLISECOND);
951             calendar.set(years, months, days, hours, value, seconds);
952             Date d0 = calendar.getTime();
953             if (d0.getTime() >= date.getTime()) {
954                 calendar.set(Calendar.MINUTE, value - count);
955                 d0 = calendar.getTime();
956             }
957             return d0;
958         }
959         else if (DateTickUnitType.HOUR.equals(unit.getUnitType())) {
960             years = calendar.get(Calendar.YEAR);
961             months = calendar.get(Calendar.MONTH);
962             days = calendar.get(Calendar.DATE);
963             if (this.tickMarkPosition == DateTickMarkPosition.START) {
964                 minutes = 0;
965                 seconds = 0;
966             }
967             else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
968                 minutes = 30;
969                 seconds = 0;
970             }
971             else {
972                 minutes = 59;
973                 seconds = 59;
974             }
975             calendar.clear(Calendar.MILLISECOND);
976             calendar.set(years, months, days, value, minutes, seconds);
977             Date d1 = calendar.getTime();
978             if (d1.getTime() >= date.getTime()) {
979                 calendar.set(Calendar.HOUR_OF_DAY, value - count);
980                 d1 = calendar.getTime();
981             }
982             return d1;
983         }
984         else if (DateTickUnitType.DAY.equals(unit.getUnitType())) {
985             years = calendar.get(Calendar.YEAR);
986             months = calendar.get(Calendar.MONTH);
987             if (this.tickMarkPosition == DateTickMarkPosition.START) {
988                 hours = 0;
989             }
990             else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
991                 hours = 12;
992             }
993             else {
994                 hours = 23;
995             }
996             calendar.clear(Calendar.MILLISECOND);
997             calendar.set(years, months, value, hours, 0, 0);
998             // long result = calendar.getTimeInMillis();
999                 // won't work with JDK 1.3
1000             Date d2 = calendar.getTime();
1001             if (d2.getTime() >= date.getTime()) {
1002                 calendar.set(Calendar.DATE, value - count);
1003                 d2 = calendar.getTime();
1004             }
1005             return d2;
1006         }
1007         else if (DateTickUnitType.MONTH.equals(unit.getUnitType())) {
1008             value = count * ((current + 1) / count) - 1;
1009             years = calendar.get(Calendar.YEAR);
1010             calendar.clear(Calendar.MILLISECOND);
1011             calendar.set(years, value, 1, 0, 0, 0);
1012             Month month = new Month(calendar.getTime(), this.timeZone,
1013                     this.locale);
1014             Date standardDate = calculateDateForPosition(
1015                     month, this.tickMarkPosition);
1016             long millis = standardDate.getTime();
1017             if (millis >= date.getTime()) {
1018                 for (int i = 0; i < count; i++) {
1019                     month = (Month) month.previous();
1020                 }
1021                 // need to peg the month in case the time zone isn't the
1022                 // default - see bug 2078057
1023                 month.peg(Calendar.getInstance(this.timeZone));
1024                 standardDate = calculateDateForPosition(
1025                         month, this.tickMarkPosition);
1026             }
1027             return standardDate;
1028         }
1029         else if (DateTickUnitType.YEAR.equals(unit.getUnitType())) {
1030             if (this.tickMarkPosition == DateTickMarkPosition.START) {
1031                 months = 0;
1032                 days = 1;
1033             }
1034             else if (this.tickMarkPosition == DateTickMarkPosition.MIDDLE) {
1035                 months = 6;
1036                 days = 1;
1037             }
1038             else {
1039                 months = 11;
1040                 days = 31;
1041             }
1042             calendar.clear(Calendar.MILLISECOND);
1043             calendar.set(value, months, days, 0, 0, 0);
1044             Date d3 = calendar.getTime();
1045             if (d3.getTime() >= date.getTime()) {
1046                 calendar.set(Calendar.YEAR, value - count);
1047                 d3 = calendar.getTime();
1048             }
1049             return d3;
1050         }
1051         return null;
1052     }

1250     protected void autoAdjustRange() {
1251 
1252         Plot plot = getPlot();
1253 
1254         if (plot == null) {
1255             return;  // no plot, no data
1256         }
1257 
1258         if (plot instanceof ValueAxisPlot) {
1259             ValueAxisPlot vap = (ValueAxisPlot) plot;
1260 
1261             Range r = vap.getDataRange(this);
1262             if (r == null) {
1263                 r = new DateRange();
1264             }
1265 
1266             long upper = this.timeline.toTimelineValue(
1267                     (long) r.getUpperBound());
1268             long lower;
1269             long fixedAutoRange = (long) getFixedAutoRange();
1270             if (fixedAutoRange > 0.0) {
1271                 lower = upper - fixedAutoRange;
1272             }
1273             else {
1274                 lower = this.timeline.toTimelineValue((long) r.getLowerBound());
1275                 double range = upper - lower;
1276                 long minRange = (long) getAutoRangeMinimumSize();
1277                 if (range < minRange) {
1278                     long expand = (long) (minRange - range) / 2;
1279                     upper = upper + expand;
1280                     lower = lower - expand;
1281                 }
1282                 upper = upper + (long) (range * getUpperMargin());
1283                 lower = lower - (long) (range * getLowerMargin());
1284             }
1285 
1286             upper = this.timeline.toMillisecond(upper);
1287             lower = this.timeline.toMillisecond(lower);
1288             DateRange dr = new DateRange(new Date(lower), new Date(upper));
1289             setRange(dr, false, false);
1290         }
