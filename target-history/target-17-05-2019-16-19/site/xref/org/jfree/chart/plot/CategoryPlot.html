577     /**
578      * Creates a new plot.
579      *
580      * @param dataset  the dataset ({@code null} permitted).
581      * @param domainAxis  the domain axis ({@code null} permitted).
582      * @param rangeAxis  the range axis ({@code null} permitted).
583      * @param renderer  the item renderer ({@code null} permitted).
584      *
585      */
586     public CategoryPlot(CategoryDataset dataset, CategoryAxis domainAxis,
587             ValueAxis rangeAxis, CategoryItemRenderer renderer) {
588 
589         super();
590 
591         this.orientation = PlotOrientation.VERTICAL;
592 
593         // allocate storage for dataset, axes and renderers
594         this.domainAxes = new HashMap<Integer, CategoryAxis>();
595         this.domainAxisLocations = new HashMap<Integer, AxisLocation>();
596         this.rangeAxes = new HashMap<Integer, ValueAxis>();
597         this.rangeAxisLocations = new HashMap<Integer, AxisLocation>();
598 
599         this.datasetToDomainAxesMap = new TreeMap();
600         this.datasetToRangeAxesMap = new TreeMap();
601 
602         this.renderers = new HashMap<Integer, CategoryItemRenderer>();
603 
604         this.datasets = new HashMap<Integer, CategoryDataset>();
605         this.datasets.put(0, dataset);
606         if (dataset != null) {
607             dataset.addChangeListener(this);
608         }
609 
610         this.axisOffset = RectangleInsets.ZERO_INSETS;
611         this.domainAxisLocations.put(0, AxisLocation.BOTTOM_OR_LEFT);
612         this.rangeAxisLocations.put(0, AxisLocation.TOP_OR_LEFT);
613 
614         this.renderers.put(0, renderer);
615         if (renderer != null) {
616             renderer.setPlot(this);
617             renderer.addChangeListener(this);
618         }
619 
620         this.domainAxes.put(0, domainAxis);
621         mapDatasetToDomainAxis(0, 0);
622         if (domainAxis != null) {
623             domainAxis.setPlot(this);
624             domainAxis.addChangeListener(this);
625         }

1414     /**
1415      * This method is used to perform argument checking on the list of
1416      * axis indices passed to mapDatasetToDomainAxes() and
1417      * mapDatasetToRangeAxes().
1418      *
1419      * @param indices  the list of indices ({@code null} permitted).
1420      */
1421     private void checkAxisIndices(List indices) {
1422         // axisIndices can be:
1423         // 1.  null;
1424         // 2.  non-empty, containing only Integer objects that are unique.
1425         if (indices == null) {
1426             return;  // OK
1427         }
1428         int count = indices.size();
1429         if (count == 0) {
1430             throw new IllegalArgumentException("Empty list not permitted.");
1431         }
1432         HashSet set = new HashSet();
1433         for (int i = 0; i < count; i++) {
1434             Object item = indices.get(i);
1435             if (!(item instanceof Integer)) {
1436                 throw new IllegalArgumentException(
1437                         "Indices must be Integer instances.");
1438             }
1439             if (set.contains(item)) {
1440                 throw new IllegalArgumentException("Indices must be unique.");
1441             }
1442             set.add(item);
1443         }
1444     }
