84  /**
85   * A calendar week.  All years are considered to have 53 weeks, numbered from 1
86   * to 53, although in many cases the 53rd week is empty.  Most of the time, the
87   * 1st week of the year *begins* in the previous calendar year, but it always
88   * finishes in the current year (this behaviour matches the workings of the
89   * {@code GregorianCalendar} class).
90   * <P>
91   * This class is immutable, which is a requirement for all
92   * {@link RegularTimePeriod} subclasses.
93   */
94  public class Week extends RegularTimePeriod implements Serializable {

125     /**
126      * Creates a time period representing the week in the specified year.
127      *
128      * @param week  the week (1 to 53).
129      * @param year  the year (1900 to 9999).
130      */
131     public Week(int week, int year) {
132         if ((week < FIRST_WEEK_IN_YEAR) && (week > LAST_WEEK_IN_YEAR)) {
133             throw new IllegalArgumentException(
134                     "The 'week' argument must be in the range 1 - 53.");
135         }
136         this.week = (byte) week;
137         this.year = (short) year;
138         peg(Calendar.getInstance());
139     }

172     /**
173      * Creates a time period for the week in which the specified date/time
174      * falls, calculated relative to the specified time zone.
175      *
176      * @param time  the date/time ({@code null} not permitted).
177      * @param zone  the time zone ({@code null} not permitted).
178      * @param locale  the locale ({@code null} not permitted).
179      *
180      * @since 1.0.7
181      */
182     public Week(Date time, TimeZone zone, Locale locale) {
183         Args.nullNotPermitted(time, "time");
184         Args.nullNotPermitted(zone, "zone");
185         Args.nullNotPermitted(locale, "locale");
186         Calendar calendar = Calendar.getInstance(zone, locale);
187         calendar.setTime(time);
188 
189         // sometimes the last few days of the year are considered to fall in
190         // the *first* week of the following year.  Refer to the Javadocs for
191         // GregorianCalendar.
192         int tempWeek = calendar.get(Calendar.WEEK_OF_YEAR);
193         if (tempWeek == 1
194                 && calendar.get(Calendar.MONTH) == Calendar.DECEMBER) {
195             this.week = 1;
196             this.year = (short) (calendar.get(Calendar.YEAR) + 1);
197         }
198         else {
199             this.week = (byte) Math.min(tempWeek, LAST_WEEK_IN_YEAR);
200             int yyyy = calendar.get(Calendar.YEAR);
201             // alternatively, sometimes the first few days of the year are
202             // considered to fall in the *last* week of the previous year...
203             if (calendar.get(Calendar.MONTH) == Calendar.JANUARY
204                     && this.week >= 52) {
205                 yyyy--;
206             }
207             this.year = (short) yyyy;
208         }
209         peg(calendar);
210     }

283     /**
284      * Returns the week preceding this one.  This method will return
285      * {@code null} for some lower limit on the range of weeks (currently
286      * week 1, 1900).  For week 1 of any year, the previous week is always week
287      * 53, but week 53 may not contain any days (you should check for this).
288      *
289      * @return The preceding week (possibly {@code null}).
290      */
291     @Override
292     public RegularTimePeriod previous() {
293 
294         Week result;
295         if (this.week != FIRST_WEEK_IN_YEAR) {
296             result = new Week(this.week - 1, this.year);
297         }
298         else {
299             // we need to work out if the previous year has 52 or 53 weeks...
300             if (this.year > 1900) {
301                 int yy = this.year - 1;
302                 Calendar prevYearCalendar = Calendar.getInstance();
303                 prevYearCalendar.set(yy, Calendar.DECEMBER, 31);
304                 result = new Week(prevYearCalendar.getActualMaximum(
305                         Calendar.WEEK_OF_YEAR), yy);
306             }
307             else {
308                 result = null;
309             }
310         }
311         return result;
312 
313     }

315     /**
316      * Returns the week following this one.  This method will return
317      * {@code null} for some upper limit on the range of weeks (currently
318      * week 53, 9999).  For week 52 of any year, the following week is always
319      * week 53, but week 53 may not contain any days (you should check for
320      * this).
321      *
322      * @return The following week (possibly {@code null}).
323      */
324     @Override
325     public RegularTimePeriod next() {
326 
327         Week result;
328         if (this.week < 52) {
329             result = new Week(this.week + 1, this.year);
330         }
331         else {
332             Calendar calendar = Calendar.getInstance();
333             calendar.set(this.year, Calendar.DECEMBER, 31);
334             int actualMaxWeek
335                 = calendar.getActualMaximum(Calendar.WEEK_OF_YEAR);
336             if (this.week < actualMaxWeek) {
337                 result = new Week(this.week + 1, this.year);
338             }
339             else {
340                 if (this.year < 9999) {
341                     result = new Week(FIRST_WEEK_IN_YEAR, this.year + 1);
342                 }
343                 else {
344                     result = null;
345                 }
346             }
347         }
348         return result;
349 
350     }

373     @Override

375         Calendar c = (Calendar) calendar.clone();
376         c.clear();
377         c.set(Calendar.YEAR, this.year);
378         c.set(Calendar.WEEK_OF_YEAR, this.week);
379         c.set(Calendar.DAY_OF_WEEK, c.getFirstDayOfWeek());
380         c.set(Calendar.HOUR, 0);
381         c.set(Calendar.MINUTE, 0);
382         c.set(Calendar.SECOND, 0);
383         c.set(Calendar.MILLISECOND, 0);
384         return c.getTimeInMillis();
385     }

400         Calendar c = (Calendar) calendar.clone();
401         c.clear();
402         c.set(Calendar.YEAR, this.year);
403         c.set(Calendar.WEEK_OF_YEAR, this.week + 1);
404         c.set(Calendar.DAY_OF_WEEK, c.getFirstDayOfWeek());
405         c.set(Calendar.HOUR, 0);

407         c.set(Calendar.SECOND, 0);

409         return c.getTimeInMillis() - 1;

471     /**
472      * Returns an integer indicating the order of this Week object relative to
473      * the specified object:
474      *
475      * negative == before, zero == same, positive == after.
476      *
477      * @param o1  the object to compare.
478      *
479      * @return negative == before, zero == same, positive == after.
480      */
481     @Override
482     public int compareTo(Object o1) {
483 
484         int result;
485 
486         // CASE 1 : Comparing to another Week object
487         // --------------------------------------------
488         if (o1 instanceof Week) {
489             Week w = (Week) o1;
490             result = this.year - w.getYear().getYear();
491             if (result == 0) {
492                 result = this.week - w.getWeek();
493             }
494         }
495 
496         // CASE 2 : Comparing to another TimePeriod object
497         // -----------------------------------------------
498         else if (o1 instanceof RegularTimePeriod) {
499             // more difficult case - evaluate later...
500             result = 0;
501         }
502 
503         // CASE 3 : Comparing to a non-TimePeriod object
504         // ---------------------------------------------
505         else {
506             // consider time periods to be ordered after general objects
507             result = 1;
508         }
509 
510         return result;
511 
512     }

514     /**
515      * Parses the string argument as a week.
516      * <P>
517      * This method is required to accept the format "YYYY-Wnn".  It will also
518      * accept "Wnn-YYYY". Anything else, at the moment, is a bonus.
519      *
520      * @param s  string to parse.
521      *
522      * @return {@code null} if the string is not parseable, the week
523      *         otherwise.
524      */
525     public static Week parseWeek(String s) {
526 
527         Week result = null;
528         if (s != null) {
529 
530             // trim whitespace from either end of the string
531             s = s.trim();
532 
533             int i = Week.findSeparator(s);
534             if (i != -1) {
535                 String s1 = s.substring(0, i).trim();
536                 String s2 = s.substring(i + 1, s.length()).trim();
537 
538                 Year y = Week.evaluateAsYear(s1);
539                 int w;
540                 if (y != null) {
541                     w = Week.stringToWeek(s2);
542                     if (w == -1) {
543                         throw new TimePeriodFormatException(
544                                 "Can't evaluate the week.");
545                     }
546                     result = new Week(w, y);
547                 }
548                 else {
549                     y = Week.evaluateAsYear(s2);
550                     if (y != null) {
551                         w = Week.stringToWeek(s1);
552                         if (w == -1) {
553                             throw new TimePeriodFormatException(
554                                     "Can't evaluate the week.");
555                         }
556                         result = new Week(w, y);
557                     }
558                     else {
559                         throw new TimePeriodFormatException(
560                                 "Can't evaluate the year.");
561                     }
562                 }
563 
564             }
565             else {
566                 throw new TimePeriodFormatException(
567                         "Could not find separator.");
568             }
569 
570         }
571         return result;
572 
573     }

575     /**
576      * Finds the first occurrence of ' ', '-', ',' or '.'
577      *
578      * @param s  the string to parse.
579      *
580      * @return {@code -1} if none of the characters was found, the
581      *      index of the first occurrence otherwise.
582      */
583     private static int findSeparator(String s) {
584 
585         int result = s.indexOf('-');
586         if (result == -1) {
587             result = s.indexOf(',');
588         }
589         if (result == -1) {
590             result = s.indexOf(' ');
591         }
592         if (result == -1) {
593             result = s.indexOf('.');
594         }
595         return result;
596     }

620     /**
621      * Converts a string to a week.
622      *
623      * @param s  the string to parse.
624      * @return {@code -1} if the string does not contain a week number,
625      *         the number of the week otherwise.
626      */
627     private static int stringToWeek(String s) {
628 
629         int result = -1;
630         s = s.replace('W', ' ');
631         s = s.trim();
632         try {
633             result = Integer.parseInt(s);
634             if ((result < 1) || (result > LAST_WEEK_IN_YEAR)) {
635                 result = -1;
636             }
637         }
638         catch (NumberFormatException e) {
639             // suppress
640         }
641         return result;
642 
643     }
