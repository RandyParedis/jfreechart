132     /**
133      * Constructs a new instance, based on a particular date/time and time zone.
134      *
135      * @param time  the date/time ({@code null} not permitted).
136      * @param zone  the time zone ({@code null} not permitted).
137      * @param locale  the locale ({@code null} not permitted).
138      */
139     public Day(Date time, TimeZone zone, Locale locale) {
140         Args.nullNotPermitted(time, "time");
141         Args.nullNotPermitted(zone, "zone");
142         Args.nullNotPermitted(locale, "locale");
143         Calendar calendar = Calendar.getInstance(zone, locale);
144         calendar.setTime(time);
145         int d = calendar.get(Calendar.DAY_OF_MONTH);
146         int m = calendar.get(Calendar.MONTH) + 1;
147         int y = calendar.get(Calendar.YEAR);
148         this.serialDate = SerialDate.createInstance(d, m, y);
149         peg(calendar);
150     }

236     /**
237      * Returns the day preceding this one.
238      *
239      * @return The day preceding this one.
240      */
241     @Override
242     public RegularTimePeriod previous() {
243         Day result;
244         int serial = this.serialDate.toSerial();
245         if (serial > SerialDate.SERIAL_LOWER_BOUND) {
246             SerialDate yesterday = SerialDate.createInstance(serial - 1);
247             return new Day(yesterday);
248         }
249         else {
250             result = null;
251         }
252         return result;
253     }

264         Day result;

266         if (serial < SerialDate.SERIAL_UPPER_BOUND) {
267             SerialDate tomorrow = SerialDate.createInstance(serial + 1);

286     /**
287      * Returns the first millisecond of the day, evaluated using the supplied
288      * calendar (which determines the time zone).
289      *
290      * @param calendar  calendar to use ({@code null} not permitted).
291      *
292      * @return The start of the day as milliseconds since 01-01-1970.
293      *
294      * @throws NullPointerException if {@code calendar} is
295      *     {@code null}.
296      */
297     @Override
298     public long getFirstMillisecond(Calendar calendar) {
299         int year = this.serialDate.getYYYY();
300         int month = this.serialDate.getMonth();
301         int day = this.serialDate.getDayOfMonth();
302         calendar.clear();
303         calendar.set(year, month - 1, day, 0, 0, 0);
304         calendar.set(Calendar.MILLISECOND, 0);
305         return calendar.getTimeInMillis();
306     }

308     /**
309      * Returns the last millisecond of the day, evaluated using the supplied
310      * calendar (which determines the time zone).
311      *
312      * @param calendar  calendar to use ({@code null} not permitted).
313      *
314      * @return The end of the day as milliseconds since 01-01-1970.
315      *
316      * @throws NullPointerException if {@code calendar} is
317      *     {@code null}.
318      */
319     @Override
320     public long getLastMillisecond(Calendar calendar) {
321         int year = this.serialDate.getYYYY();
322         int month = this.serialDate.getMonth();
323         int day = this.serialDate.getDayOfMonth();
324         calendar.clear();
325         calendar.set(year, month - 1, day, 23, 59, 59);
326         calendar.set(Calendar.MILLISECOND, 999);
327         return calendar.getTimeInMillis();
328     }

332      * true if the target is a Day instance or a SerialDate instance
333      * representing the same day as this object. In all other cases,
334      * returns false.

340     @Override
341     public boolean equals(Object obj) {
342         if (obj == this) {
343             return true;
344         }
345         if (!(obj instanceof Day)) {
346             return false;
347         }
348         Day that = (Day) obj;

369     /**
370      * Returns an integer indicating the order of this Day object relative to
371      * the specified object:
372      *
373      * negative == before, zero == same, positive == after.
374      *
375      * @param o1  the object to compare.
376      *
377      * @return negative == before, zero == same, positive == after.
378      */
379     @Override
380     public int compareTo(Object o1) {
381         int result;
382 
383         // CASE 1 : Comparing to another Day object
384         // ----------------------------------------
385         if (o1 instanceof Day) {
386             Day d = (Day) o1;
387             result = -d.getSerialDate().compare(this.serialDate);
388         }
389 
390         // CASE 2 : Comparing to another TimePeriod object
391         // -----------------------------------------------
392         else if (o1 instanceof RegularTimePeriod) {
393             // more difficult case - evaluate later...
394             result = 0;
395         }
396 
397         // CASE 3 : Comparing to a non-TimePeriod object
398         // ---------------------------------------------
399         else {
400             // consider time periods to be ordered after general objects
401             result = 1;
402         }
403 
404         return result;
405     }

417     /**
418      * Parses the string argument as a day.
419      * <P>
420      * This method is required to recognise YYYY-MM-DD as a valid format.
421      * Anything else, for now, is a bonus.
422      *
423      * @param s  the date string to parse.
424      *
425      * @return {@code null} if the string does not contain any parseable
426      *      string, the day otherwise.
427      */
428     public static Day parseDay(String s) {
429         try {
430             return new Day (Day.DATE_FORMAT.parse(s));
431         }
432         catch (ParseException e1) {
433             try {
434                 return new Day (Day.DATE_FORMAT_SHORT.parse(s));
435             }
436             catch (ParseException e2) {
437               // ignore
438             }
439         }
440         return null;
441     }
