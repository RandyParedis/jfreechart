94  /**
95   * Represents a sequence of zero or more data items in the form (x, y).  By
96   * default, items in the series will be sorted into ascending order by x-value,
97   * and duplicate x-values are permitted.  Both the sorting and duplicate
98   * defaults can be changed in the constructor.  Y-values can be
99   * {@code null} to represent missing values.
100  */

242     /**
243      * Updates the cached values for the minimum and maximum data values.
244      *
245      * @param item  the item added ({@code null} not permitted).
246      *
247      * @since 1.0.13
248      */
249     private void updateBoundsForAddedItem(XYDataItem item) {
250         double x = item.getXValue();
251         this.minX = minIgnoreNaN(this.minX, x);
252         this.maxX = maxIgnoreNaN(this.maxX, x);
253         if (item.getY() != null) {
254             double y = item.getYValue();
255             this.minY = minIgnoreNaN(this.minY, y);
256             this.maxY = maxIgnoreNaN(this.maxY, y);
257         }
258     }
259 
260     /**
261      * Updates the cached values for the minimum and maximum data values on
262      * the basis that the specified item has just been removed.
263      *
264      * @param item  the item added ({@code null} not permitted).
265      *
266      * @since 1.0.13
267      */
268     private void updateBoundsForRemovedItem(XYDataItem item) {
269         boolean itemContributesToXBounds = false;
270         boolean itemContributesToYBounds = false;
271         double x = item.getXValue();
272         if (!Double.isNaN(x)) {
273             if (x <= this.minX || x >= this.maxX) {
274                 itemContributesToXBounds = true;
275             }
276         }
277         if (item.getY() != null) {
278             double y = item.getYValue();
279             if (!Double.isNaN(y)) {
280                 if (y <= this.minY || y >= this.maxY) {
281                     itemContributesToYBounds = true;
282                 }
283             }
284         }
285         if (itemContributesToYBounds) {
286             findBoundsByIteration();
287         }
288         else if (itemContributesToXBounds) {
289             if (getAutoSort()) {
290                 this.minX = getX(0).doubleValue();
291                 this.maxX = getX(getItemCount() - 1).doubleValue();
292             }
293             else {
294                 findBoundsByIteration();
295             }
296         }
297     }

299     /**
300      * Finds the bounds of the x and y values for the series, by iterating
301      * through all the data items.
302      *
303      * @since 1.0.13
304      */
305     private void findBoundsByIteration() {
306         this.minX = Double.NaN;
307         this.maxX = Double.NaN;
308         this.minY = Double.NaN;
309         this.maxY = Double.NaN;
310         Iterator iterator = this.data.iterator();
311         while (iterator.hasNext()) {
312             XYDataItem item = (XYDataItem) iterator.next();
313             updateBoundsForAddedItem(item);
314         }
315     }

372     /**
373      * Sets the maximum number of items that will be retained in the series.
374      * If you add a new item to the series such that the number of items will
375      * exceed the maximum item count, then the first element in the series is
376      * automatically removed, ensuring that the maximum item count is not
377      * exceeded.
378      * <p>
379      * Typically this value is set before the series is populated with data,
380      * but if it is applied later, it may cause some items to be removed from
381      * the series (in which case a {@link SeriesChangeEvent} will be sent to
382      * all registered listeners).
383      *
384      * @param maximum  the maximum number of items for the series.
385      */
386     public void setMaximumItemCount(int maximum) {
387         this.maximumItemCount = maximum;
388         int remove = this.data.size() - maximum;
389         if (remove > 0) {
390             this.data.subList(0, remove).clear();
391             findBoundsByIteration();
392             fireSeriesChanged();
393         }
394     }

497     /**
498      * Adds a data item to the series and, if requested, sends a
499      * {@link SeriesChangeEvent} to all registered listeners.
500      *
501      * @param item  the (x, y) item ({@code null} not permitted).
502      * @param notify  a flag that controls whether or not a
503      *                {@link SeriesChangeEvent} is sent to all registered
504      *                listeners.
505      */
506     public void add(XYDataItem item, boolean notify) {
507         Args.nullNotPermitted(item, "item");
508         item = (XYDataItem) item.clone();
509         if (this.autoSort) {
510             int index = Collections.binarySearch(this.data, item);
511             if (index < 0) {
512                 this.data.add(-index - 1, item);
513             }
514             else {
515                 if (this.allowDuplicateXValues) {
516                     // need to make sure we are adding *after* any duplicates
517                     int size = this.data.size();
518                     while (index < size && item.compareTo(
519                             this.data.get(index)) == 0) {
520                         index++;
521                     }
522                     if (index < this.data.size()) {
523                         this.data.add(index, item);
524                     }
525                     else {
526                         this.data.add(item);
527                     }
528                 }
529                 else {
530                     throw new SeriesException("X-value already exists.");
531                 }
532             }
533         }
534         else {
535             if (!this.allowDuplicateXValues) {
536                 // can't allow duplicate values, so we need to check whether
537                 // there is an item with the given x-value already
538                 int index = indexOf(item.getX());
539                 if (index >= 0) {
540                     throw new SeriesException("X-value already exists.");
541                 }
542             }
543             this.data.add(item);
544         }
545         updateBoundsForAddedItem(item);
546         if (getItemCount() > this.maximumItemCount) {
547             XYDataItem removed = (XYDataItem) this.data.remove(0);
548             updateBoundsForRemovedItem(removed);
549         }
550         if (notify) {
551             fireSeriesChanged();
552         }
553     }

659     /**
660      * A function to find the minimum of two values, but ignoring any
661      * Double.NaN values.
662      *
663      * @param a  the first value.
664      * @param b  the second value.
665      *
666      * @return The minimum of the two values.
667      */
668     private double minIgnoreNaN(double a, double b) {
669         if (Double.isNaN(a)) {
670             return b;
671         }
672         if (Double.isNaN(b)) {
673             return a;
674         }
675         return Math.min(a, b);
676     }

678     /**
679      * A function to find the maximum of two values, but ignoring any
680      * Double.NaN values.
681      *
682      * @param a  the first value.
683      * @param b  the second value.
684      *
685      * @return The maximum of the two values.
686      */
687     private double maxIgnoreNaN(double a, double b) {
688         if (Double.isNaN(a)) {
689             return b;
690         }
691         if (Double.isNaN(b)) {
692             return a;
693         }
694         return Math.max(a, b);
695     }

697     /**
698      * Updates the value of an item in the series and sends a
699      * {@link SeriesChangeEvent} to all registered listeners.
700      *
701      * @param index  the item (zero based index).
702      * @param y  the new value ({@code null} permitted).
703      *
704      * @since 1.0.1
705      */
706     public void updateByIndex(int index, Number y) {
707         XYDataItem item = getRawDataItem(index);
708 
709         // figure out if we need to iterate through all the y-values
710         boolean iterate = false;
711         double oldY = item.getYValue();
712         if (!Double.isNaN(oldY)) {
713             iterate = oldY <= this.minY || oldY >= this.maxY;
714         }
715         item.setY(y);
716 
717         if (iterate) {

776      * Adds or updates an item in the series and sends a
777      * {@link SeriesChangeEvent} to all registered listeners.
778      *
779      * @param item  the data item ({@code null} not permitted).
780      *
781      * @return A copy of the overwritten data item, or {@code null} if no
782      *         item was overwritten.
783      *
784      * @since 1.0.14
785      */
786     public XYDataItem addOrUpdate(XYDataItem item) {
787         Args.nullNotPermitted(item, "item");
788         if (this.allowDuplicateXValues) {
789             add(item);
790             return null;
791         }
792 
793         // if we get to here, we know that duplicate X values are not permitted
794         XYDataItem overwritten = null;
795         int index = indexOf(item.getX());
796         if (index >= 0) {
797             XYDataItem existing = (XYDataItem) this.data.get(index);
798             overwritten = (XYDataItem) existing.clone();
799             // figure out if we need to iterate through all the y-values
800             boolean iterate = false;
801             double oldY = existing.getYValue();
802             if (!Double.isNaN(oldY)) {
803                 iterate = oldY <= this.minY || oldY >= this.maxY;
804             }
805             existing.setY(item.getY());
806 
807             if (iterate) {
808                 findBoundsByIteration();
809             }
810             else if (item.getY() != null) {
811                 double yy = item.getY().doubleValue();
812                 this.minY = minIgnoreNaN(this.minY, yy);
813                 this.maxY = maxIgnoreNaN(this.maxY, yy);
814             }
815         }
816         else {
817             // if the series is sorted, the negative index is a result from
818             // Collections.binarySearch() and tells us where to insert the
819             // new item...otherwise it will be just -1 and we should just
820             // append the value to the list...
821             item = (XYDataItem) item.clone();
822             if (this.autoSort) {
823                 this.data.add(-index - 1, item);
824             }
825             else {
826                 this.data.add(item);
827             }
828             updateBoundsForAddedItem(item);
829 
830             // check if this addition will exceed the maximum item count...
831             if (getItemCount() > this.maximumItemCount) {
832                 XYDataItem removed = (XYDataItem) this.data.remove(0);
833                 updateBoundsForRemovedItem(removed);
834             }
835         }
836         fireSeriesChanged();
837         return overwritten;
838     }

841      * Returns the index of the item with the specified x-value, or a negative
842      * index if the series does not contain an item with that x-value.  Be
843      * aware that for an unsorted series, the index is found by iterating
844      * through all items in the series.
845      *
846      * @param x  the x-value ({@code null} not permitted).
847      *
848      * @return The index.
849      */
850     public int indexOf(Number x) {
851         if (this.autoSort) {
852             return Collections.binarySearch(this.data, new XYDataItem(x, null));
853         }
854         else {
855             for (int i = 0; i < this.data.size(); i++) {
856                 XYDataItem item = (XYDataItem) this.data.get(i);
857                 if (item.getX().equals(x)) {
858                     return i;
859                 }
860             }
861             return -1;
862         }
863     }

974     @Override
975     public int hashCode() {
976         int result = super.hashCode();
977         // it is too slow to look at every data item, so let's just look at
978         // the first, middle and last items...
979         int count = getItemCount();
980         if (count > 0) {
981             XYDataItem item = getRawDataItem(0);
982             result = 29 * result + item.hashCode();
983         }
984         if (count > 1) {
985             XYDataItem item = getRawDataItem(count - 1);
986             result = 29 * result + item.hashCode();
987         }
988         if (count > 2) {
989             XYDataItem item = getRawDataItem(count / 2);
990             result = 29 * result + item.hashCode();
991         }
992         result = 29 * result + this.maximumItemCount;
