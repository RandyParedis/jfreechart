147     /**
148      * Some basic checks for the clone() method.
149      */
150     @Test
151     public void testCloning() throws CloneNotSupportedException {
152         MultiplePiePlot p1 = new MultiplePiePlot();
153         Rectangle2D rect = new Rectangle2D.Double(1.0, 2.0, 3.0, 4.0);
154         p1.setLegendItemShape(rect);
155         MultiplePiePlot p2 = (MultiplePiePlot) p1.clone();
156         assertTrue(p1 != p2);
157         assertTrue(p1.getClass() == p2.getClass());
158         assertTrue(p1.equals(p2));
159 
160         // check independence
161         rect.setRect(2.0, 3.0, 4.0, 5.0);
162         assertFalse(p1.equals(p2));
163     }

177     /**
178      * Fetches the legend items and checks the values.
179      */
180     @Test
181     public void testGetLegendItems() {
182         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
183         dataset.addValue(35.0, "S1", "C1");
184         dataset.addValue(45.0, "S1", "C2");
185         dataset.addValue(55.0, "S2", "C1");
186         dataset.addValue(15.0, "S2", "C2");
187         MultiplePiePlot plot = new MultiplePiePlot(dataset);
188         JFreeChart chart = new JFreeChart(plot);
189         LegendItemCollection legendItems = plot.getLegendItems();
190         assertEquals(2, legendItems.getItemCount());
191         LegendItem item1 = legendItems.get(0);
192         assertEquals("S1", item1.getLabel());
193         assertEquals("S1", item1.getSeriesKey());
194         assertEquals(0, item1.getSeriesIndex());
195         assertEquals(dataset, item1.getDataset());
196         assertEquals(0, item1.getDatasetIndex());
197 
198         LegendItem item2 = legendItems.get(1);
199         assertEquals("S2", item2.getLabel());
200         assertEquals("S2", item2.getSeriesKey());
201         assertEquals(1, item2.getSeriesIndex());
202         assertEquals(dataset, item2.getDataset());
203         assertEquals(0, item2.getDatasetIndex());
204     }
