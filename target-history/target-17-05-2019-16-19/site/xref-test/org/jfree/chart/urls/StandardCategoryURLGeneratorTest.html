60      /**
61       * Some tests for the generateURL() method.
62       */
63      @Test
64      public void testGenerateURL() {
65          StandardCategoryURLGenerator g1 = new StandardCategoryURLGenerator();
66          DefaultCategoryDataset dataset = new DefaultCategoryDataset();
67          dataset.addValue(1.0, "R1", "C1");
68          dataset.addValue(2.0, "R2", "C2");
69          dataset.addValue(3.0, "R&", "C&");
70          assertEquals("index.html?series=R1&amp;category=C1",
71                  g1.generateURL(dataset, 0, 0));
72          assertEquals("index.html?series=R1&amp;category=C2",
73                  g1.generateURL(dataset, 0, 1));
74          assertEquals("index.html?series=R2&amp;category=C2",
75                  g1.generateURL(dataset, 1, 1));
76          assertEquals("index.html?series=R%26&amp;category=C%26",
77                  g1.generateURL(dataset, 2, 2));
78      }
