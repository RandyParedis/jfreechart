112     /**
113      * Test if adjustedLog10 and adjustedPow10 are inverses of each other
114      */
115     @Test
116     public void testAdjustedLog10() {
117         checkLogPowRoundTrip(20);
118         checkLogPowRoundTrip(10);
119         checkLogPowRoundTrip(5);
120         checkLogPowRoundTrip(2);
121         checkLogPowRoundTrip(1);

124         checkLogPowRoundTrip(0.0001);

127     private void checkLogPowRoundTrip(double value) {
128         assertEquals("log(pow(x)) = x", value, this.axis.adjustedLog10(
129                 this.axis.adjustedPow10(value)), EPSILON);
130         assertEquals("pow(log(x)) = x", value, this.axis.adjustedPow10(
131                 this.axis.adjustedLog10(value)), EPSILON);
132     }

134     /**
135      * Test if switchedLog10 and switchedPow10 are inverses of each other
136      */
137     @Test
138     public void testSwitchedLog10() {
139          assertFalse("Axis should not allow negative values",
140                  this.axis.getAllowNegativesFlag());
141 
142          assertEquals(Math.log(0.5) / LogarithmicAxis.LOG10_VALUE,
143                  this.axis.switchedLog10(0.5), EPSILON);
144          checkSwitchedLogPowRoundTrip(20);
145          checkSwitchedLogPowRoundTrip(10);
146          checkSwitchedLogPowRoundTrip(5);
147          checkSwitchedLogPowRoundTrip(2);
148          checkSwitchedLogPowRoundTrip(1);
149          checkSwitchedLogPowRoundTrip(0.5);
150          checkSwitchedLogPowRoundTrip(0.2);
151          checkSwitchedLogPowRoundTrip(0.0001);
152      }

211     private void checkPointsToJava2D(RectangleEdge edge,
212               Rectangle2D plotArea) {
213         assertEquals("Left most point on the axis should be beginning of "
214                  + "range.", plotArea.getX(), this.axis.valueToJava2D(
215                  this.axis.getLowerBound(), plotArea, edge), EPSILON);
216         assertEquals("Right most point on the axis should be end of range.",
217                  plotArea.getX() + plotArea.getWidth(),
218                  this.axis.valueToJava2D(this.axis.getUpperBound(),
219                  plotArea, edge), EPSILON);
220         assertEquals("Center point on the axis should geometric mean of the bounds.",
221                  plotArea.getX() + (plotArea.getWidth() / 2),
222                  this.axis.valueToJava2D(Math.sqrt(this.axis.getLowerBound()
223                  * this.axis.getUpperBound()), plotArea, edge), EPSILON);
224     }

226     /**
227      * Check the translation java2D to value for left, right, and center point.
228      *
229      * @param edge  the edge.
230      * @param plotArea  the plot area.
231      */
232      private void checkPointsToValue(RectangleEdge edge, Rectangle2D plotArea) {
233          assertEquals("Right most point on the axis should be end of range.",
234                  this.axis.getUpperBound(), this.axis.java2DToValue(
235                  plotArea.getX() + plotArea.getWidth(), plotArea, edge),
236                  EPSILON);
237 
238          assertEquals("Left most point on the axis should be beginning of "
239                  + "range.", this.axis.getLowerBound(),
240                  this.axis.java2DToValue(plotArea.getX(), plotArea, edge),
241                  EPSILON);
242 
243          assertEquals("Center point on the axis should geometric mean of the "
244                  + "bounds.", Math.sqrt(this.axis.getUpperBound()
245                  * this.axis.getLowerBound()), this.axis.java2DToValue(
246                  plotArea.getX() + (plotArea.getWidth() / 2), plotArea, edge),
247                  EPSILON);
248     }
