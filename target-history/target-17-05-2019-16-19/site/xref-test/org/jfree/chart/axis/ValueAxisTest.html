209     /**
210      * Tests the the lower and upper margin settings produce the expected
211      * results.
212      */
213     @Test
214     public void testAxisMargins() {
215         XYSeries series = new XYSeries("S1");
216         series.add(100.0, 1.1);
217         series.add(200.0, 2.2);
218         XYSeriesCollection dataset = new XYSeriesCollection(series);
219         dataset.setIntervalWidth(0.0);
220         JFreeChart chart = ChartFactory.createScatterPlot("Title", "X", "Y", 
221                 dataset);
222         ValueAxis domainAxis = ((XYPlot) chart.getPlot()).getDomainAxis();
223         Range r = domainAxis.getRange();
224         assertEquals(110.0, r.getLength(), EPSILON);
225         domainAxis.setLowerMargin(0.10);
226         domainAxis.setUpperMargin(0.10);
227         r = domainAxis.getRange();
228         assertEquals(120.0, r.getLength(), EPSILON);
229     }

231     /**
232      * A test for bug 3555275 (where the fixed axis space is calculated 
233      * incorrectly).
234      */
235     @Test
236     public void test3555275() {
237         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
238         JFreeChart chart = ChartFactory.createLineChart("Title", "X", "Y",
239                 dataset, PlotOrientation.VERTICAL, true, false, false);
240         CategoryPlot plot = (CategoryPlot) chart.getPlot();
241         plot.setInsets(RectangleInsets.ZERO_INSETS);
242         plot.setAxisOffset(RectangleInsets.ZERO_INSETS);
243         ValueAxis yAxis = plot.getRangeAxis();
244         yAxis.setFixedDimension(100.0);
245         BufferedImage image = new BufferedImage(500, 300, 
246                 BufferedImage.TYPE_INT_RGB);
247         Graphics2D g2 = image.createGraphics();
248         ChartRenderingInfo info = new ChartRenderingInfo();
249         chart.draw(g2, new Rectangle2D.Double(0, 0, 500, 300), info);
250         g2.dispose();
251         Rectangle2D rect = info.getPlotInfo().getDataArea();
252         double x = rect.getMinX();
253         assertEquals(100.0, x, 1.0);
254     }
