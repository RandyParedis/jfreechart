121     /**
122      * Test the translation of Java2D values to data values.
123      */
124     @Test
125     public void testTranslateJava2DToValue() {
126         LogAxis axis = new LogAxis();
127         axis.setRange(50.0, 100.0);
128         Rectangle2D dataArea = new Rectangle2D.Double(10.0, 50.0, 400.0, 300.0);
129         double y1 = axis.java2DToValue(75.0, dataArea, RectangleEdge.LEFT);
130         assertEquals(94.3874312681693, y1, EPSILON);
131         double y2 = axis.java2DToValue(75.0, dataArea, RectangleEdge.RIGHT);
132         assertEquals(94.3874312681693, y2, EPSILON);
133         double x1 = axis.java2DToValue(75.0, dataArea, RectangleEdge.TOP);
134         assertEquals(55.961246381405, x1, EPSILON);
135         double x2 = axis.java2DToValue(75.0, dataArea, RectangleEdge.BOTTOM);
136         assertEquals(55.961246381405, x2, EPSILON);
137         axis.setInverted(true);
138         double y3 = axis.java2DToValue(75.0, dataArea, RectangleEdge.LEFT);
139         assertEquals(52.9731547179647, y3, EPSILON);
140         double y4 = axis.java2DToValue(75.0, dataArea, RectangleEdge.RIGHT);
141         assertEquals(52.9731547179647, y4, EPSILON);
142         double x3 = axis.java2DToValue(75.0, dataArea, RectangleEdge.TOP);
143         assertEquals(89.3475453695651, x3, EPSILON);
144         double x4 = axis.java2DToValue(75.0, dataArea, RectangleEdge.BOTTOM);
145         assertEquals(89.3475453695651, x4, EPSILON);
146     }

160      * LogAxis used as the range axis for a CategoryPlot.
161      */
162     @Test
163     public void testAutoRange1() {
164         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
165         dataset.setValue(100.0, "Row 1", "Column 1");
166         dataset.setValue(200.0, "Row 1", "Column 2");
167         JFreeChart chart = ChartFactory.createBarChart("Test", "Categories",

169         CategoryPlot plot = (CategoryPlot) chart.getPlot();
170         LogAxis axis = new LogAxis("Log(Y)");
171         plot.setRangeAxis(axis);
172         assertEquals(0.0, axis.getLowerBound(), EPSILON);
173         assertEquals(2.6066426411261268E7, axis.getUpperBound(), EPSILON);
174     }

176     /**
177      * A simple test for the auto-range calculation looking at a
178      * NumberAxis used as the range axis for a CategoryPlot.  In this
179      * case, the original dataset is replaced with a new dataset.
180      */
181     @Test
182     public void testAutoRange3() {
183         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
184         dataset.setValue(100.0, "Row 1", "Column 1");
185         dataset.setValue(200.0, "Row 1", "Column 2");
186         JFreeChart chart = ChartFactory.createLineChart("Test", "Categories",
187                 "Value", dataset, PlotOrientation.VERTICAL, false, false,
188                 false);
189         CategoryPlot plot = (CategoryPlot) chart.getPlot();
190         LogAxis axis = new LogAxis("Log(Y)");
191         plot.setRangeAxis(axis);
192         assertEquals(96.59363289248458, axis.getLowerBound(), EPSILON);
193         assertEquals(207.0529847682752, axis.getUpperBound(), EPSILON);
194 
195         // now replacing the dataset should update the axis range...
196         DefaultCategoryDataset dataset2 = new DefaultCategoryDataset();
197         dataset2.setValue(900.0, "Row 1", "Column 1");
198         dataset2.setValue(1000.0, "Row 1", "Column 2");
199         plot.setDataset(dataset2);
200         assertEquals(895.2712433374774, axis.getLowerBound(), EPSILON);
201         assertEquals(1005.2819262292991, axis.getUpperBound(), EPSILON);
202     }
