129     private static final double EPSILON = 0.0000000001;
130 
131     /**
132      * Some checks for the findDomainBounds() method (which requires special
133      * handling when the centerBarAtStartValue flag is set to true).
134      */
135     @Test
136     public void testFindDomainBounds() {
137         AbstractXYItemRenderer renderer = new ClusteredXYBarRenderer();
138         XYDataset dataset = createSampleDataset1();
139         Range r = renderer.findDomainBounds(dataset);
140         assertEquals(0.9, r.getLowerBound(), EPSILON);
141         assertEquals(13.1, r.getUpperBound(), EPSILON);
142 
143         renderer = new ClusteredXYBarRenderer(0.0, true);
144         r = renderer.findDomainBounds(dataset);
145         assertEquals(0.8, r.getLowerBound(), EPSILON);
146         assertEquals(13.0, r.getUpperBound(), EPSILON);
147 
148         // check that a null dataset returns null bounds
149         assertTrue(renderer.findDomainBounds(null) == null);
150     }
