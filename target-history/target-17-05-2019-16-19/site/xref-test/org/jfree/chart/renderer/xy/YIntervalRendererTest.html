217     /**
218      * A check for the datasetIndex and seriesIndex fields in the LegendItem
219      * returned by the getLegendItem() method.
220      */
221     @Test
222     public void testGetLegendItemSeriesIndex() {
223         YIntervalSeriesCollection d1 = new YIntervalSeriesCollection();
224         YIntervalSeries s1 = new YIntervalSeries("S1");
225         s1.add(1.0, 1.1, 1.2, 1.3);
226         YIntervalSeries s2 = new YIntervalSeries("S2");
227         s2.add(1.0, 1.1, 1.2, 1.3);
228         d1.addSeries(s1);
229         d1.addSeries(s2);
230 
231         YIntervalSeriesCollection d2 = new YIntervalSeriesCollection();
232         YIntervalSeries s3 = new YIntervalSeries("S3");
233         s3.add(1.0, 1.1, 1.2, 1.3);
234         YIntervalSeries s4 = new YIntervalSeries("S4");
235         s4.add(1.0, 1.1, 1.2, 1.3);
236         YIntervalSeries s5 = new YIntervalSeries("S5");
237         s5.add(1.0, 1.1, 1.2, 1.3);
238         d2.addSeries(s3);
239         d2.addSeries(s4);
240         d2.addSeries(s5);
241 
242         YIntervalRenderer r = new YIntervalRenderer();
243         XYPlot plot = new XYPlot(d1, new NumberAxis("x"),
244                 new NumberAxis("y"), r);
245         plot.setDataset(1, d2);
246         /*JFreeChart chart =*/ new JFreeChart(plot);
247         LegendItem li = r.getLegendItem(1, 2);
248         assertEquals("S5", li.getLabel());
249         assertEquals(1, li.getDatasetIndex());
250         assertEquals(2, li.getSeriesIndex());
251     }
