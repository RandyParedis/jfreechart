145     /**
146      * Some checks for the findRangeBounds() method.
147      */
148     @Test
149     public void testFindRangeBounds() {
150         GroupedStackedBarRenderer r = new GroupedStackedBarRenderer();
151         assertNull(r.findRangeBounds(null));
152 
153         // an empty dataset should return a null range
154         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
155         assertNull(r.findRangeBounds(dataset));
156 
157         dataset.addValue(1.0, "R1", "C1");
158         assertEquals(new Range(0.0, 1.0), r.findRangeBounds(dataset));
159 
160         dataset.addValue(-2.0, "R1", "C2");
161         assertEquals(new Range(-2.0, 1.0), r.findRangeBounds(dataset));
162 
163         dataset.addValue(null, "R1", "C3");
164         assertEquals(new Range(-2.0, 1.0), r.findRangeBounds(dataset));
165 
166         KeyToGroupMap m = new KeyToGroupMap("G1");
167         m.mapKeyToGroup("R1", "G1");
168         m.mapKeyToGroup("R2", "G1");
169         m.mapKeyToGroup("R3", "G2");
170         r.setSeriesToGroupMap(m);
171 
172         dataset.addValue(0.5, "R3", "C1");
173         assertEquals(new Range(-2.0, 1.0), r.findRangeBounds(dataset));
174 
175         dataset.addValue(5.0, "R3", "C2");
176         assertEquals(new Range(-2.0, 5.0), r.findRangeBounds(dataset));
177     }
