152     /**
153      * Confirm that cloning works.
154      */
155     @Test
156     public void testCloning() throws CloneNotSupportedException {
157         StandardXYItemRenderer r1 = new StandardXYItemRenderer();
158         Rectangle2D rect1 = new Rectangle2D.Double(1.0, 2.0, 3.0, 4.0);
159         r1.setLegendLine(rect1);
160         StandardXYItemRenderer r2 = (StandardXYItemRenderer) r1.clone();
161         assertTrue(r1 != r2);
162         assertTrue(r1.getClass() == r2.getClass());
163         assertTrue(r1.equals(r2));
164 
165         // check independence
166         rect1.setRect(4.0, 3.0, 2.0, 1.0);
167         assertFalse(r1.equals(r2));
168         r2.setLegendLine(new Rectangle2D.Double(4.0, 3.0, 2.0, 1.0));
169         assertTrue(r1.equals(r2));
170 
171         r1.setSeriesShapesFilled(1, Boolean.TRUE);
172         assertFalse(r1.equals(r2));
173         r2.setSeriesShapesFilled(1, Boolean.TRUE);
174         assertTrue(r1.equals(r2));
175     }

197     /**
198      * A check for the datasetIndex and seriesIndex fields in the LegendItem
199      * returned by the getLegendItem() method.
200      */
201     @Test
202     public void testGetLegendItemSeriesIndex() {
203         XYSeriesCollection d1 = new XYSeriesCollection();
204         XYSeries s1 = new XYSeries("S1");
205         s1.add(1.0, 1.1);
206         XYSeries s2 = new XYSeries("S2");
207         s2.add(1.0, 1.1);
208         d1.addSeries(s1);
209         d1.addSeries(s2);
210 
211         XYSeriesCollection d2 = new XYSeriesCollection();
212         XYSeries s3 = new XYSeries("S3");
213         s3.add(1.0, 1.1);
214         XYSeries s4 = new XYSeries("S4");
215         s4.add(1.0, 1.1);
216         XYSeries s5 = new XYSeries("S5");
217         s5.add(1.0, 1.1);
218         d2.addSeries(s3);
219         d2.addSeries(s4);
220         d2.addSeries(s5);
221 
222         StandardXYItemRenderer r = new StandardXYItemRenderer();
223         XYPlot plot = new XYPlot(d1, new NumberAxis("x"),
224                 new NumberAxis("y"), r);
225         plot.setDataset(1, d2);
226         /*JFreeChart chart =*/ new JFreeChart(plot);
227         LegendItem li = r.getLegendItem(1, 2);
228         assertEquals("S5", li.getLabel());
229         assertEquals(1, li.getDatasetIndex());
230         assertEquals(2, li.getSeriesIndex());
231     }

233     /**
234      * A check to ensure that an item that falls outside the plot's data area
235      * does NOT generate an item entity.
236      */
237     @Test
238     public void testNoDisplayedItem() {
239         XYSeriesCollection dataset = new XYSeriesCollection();
240         XYSeries s1 = new XYSeries("S1");
241         s1.add(10.0, 10.0);
242         dataset.addSeries(s1);
243         JFreeChart chart = ChartFactory.createXYLineChart("Title", "X", "Y",
244                 dataset, PlotOrientation.VERTICAL, false, true, false);
245         XYPlot plot = (XYPlot) chart.getPlot();
246         plot.setRenderer(new StandardXYItemRenderer());
247         NumberAxis xAxis = (NumberAxis) plot.getDomainAxis();
248         xAxis.setRange(0.0, 5.0);
249         NumberAxis yAxis = (NumberAxis) plot.getRangeAxis();
250         yAxis.setRange(0.0, 5.0);
251         BufferedImage image = new BufferedImage(200 , 100,
252                 BufferedImage.TYPE_INT_RGB);
253         Graphics2D g2 = image.createGraphics();
254         ChartRenderingInfo info = new ChartRenderingInfo();
255         chart.draw(g2, new Rectangle2D.Double(0, 0, 200, 100), null, info);
256         g2.dispose();
257         EntityCollection ec = info.getEntityCollection();
258         assertFalse(TestUtils.containsInstanceOf(ec.getEntities(),
259                 XYItemEntity.class));
260     }
