57      /**
58       * Confirm that the equals method can distinguish all the required fields.
59       */
60      @Test
61      public void testEquals() {
62          TaskSeries s1 = new TaskSeries("S");
63          s1.add(new Task("T1", new Date(1), new Date(2)));
64          s1.add(new Task("T2", new Date(11), new Date(22)));
65          TaskSeries s2 = new TaskSeries("S");
66          s2.add(new Task("T1", new Date(1), new Date(2)));
67          s2.add(new Task("T2", new Date(11), new Date(22)));
68          assertTrue(s1.equals(s2));
69          assertTrue(s2.equals(s1));
70  
71          s1.add(new Task("T3", new Date(22), new Date(33)));
72          assertFalse(s1.equals(s2));
73          s2.add(new Task("T3", new Date(22), new Date(33)));
74          assertTrue(s1.equals(s2));
75      }

77      /**
78       * Confirm that cloning works.
79       */
80      @Test
81      public void testCloning() throws CloneNotSupportedException {
82          TaskSeries s1 = new TaskSeries("S");
83          s1.add(new Task("T1", new Date(1), new Date(2)));
84          s1.add(new Task("T2", new Date(11), new Date(22)));
85          TaskSeries s2 = (TaskSeries) s1.clone();
86          assertTrue(s1 != s2);
87          assertTrue(s1.getClass() == s2.getClass());
88          assertTrue(s1.equals(s2));
89  
90          // basic check for independence
91          s1.add(new Task("T3", new Date(22), new Date(33)));
92          assertFalse(s1.equals(s2));
93          s2.add(new Task("T3", new Date(22), new Date(33)));
94          assertTrue(s1.equals(s2));
95      }
