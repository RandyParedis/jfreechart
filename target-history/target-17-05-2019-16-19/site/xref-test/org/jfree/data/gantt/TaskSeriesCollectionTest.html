457      */
458     @Test
459     public void testCloning() throws CloneNotSupportedException {
460         TaskSeries s1 = new TaskSeries("S1");
461         s1.add(new Task("T1", new Date(1), new Date(2)));
462         s1.add(new Task("T2", new Date(11), new Date(22)));
463         TaskSeries s2 = new TaskSeries("S2");
464         s2.add(new Task("T1", new Date(33), new Date(44)));
465         s2.add(new Task("T2", new Date(55), new Date(66)));
466         TaskSeriesCollection c1 = new TaskSeriesCollection();
467         c1.add(s1);
468         c1.add(s2);
469 
470         TaskSeriesCollection c2 = (TaskSeriesCollection) c1.clone();
471         assertTrue(c1 != c2);
472         assertTrue(c1.getClass() == c2.getClass());
473         assertTrue(c1.equals(c2));
474 
475         // basic check for independence
476         s1.add(new Task("T3", new Date(21), new Date(33)));
477         assertFalse(c1.equals(c2));
478         TaskSeries series = c2.getSeries("S1");
479         series.add(new Task("T3", new Date(21), new Date(33)));
480         assertTrue(c1.equals(c2));
481 
482     }
