72  /**
73   * Defines a quarter (in a given year).  The range supported is Q1 1900 to
74   * Q4 9999.  This class is immutable, which is a requirement for all
75   * {@link RegularTimePeriod} subclasses.
76   */
77  public class Quarter extends RegularTimePeriod implements Serializable {
78  
79      /** For serialization. */
80      private static final long serialVersionUID = 3810061714380888671L;

82      /** Constant for quarter 1. */
83      public static final int FIRST_QUARTER = 1;
84  
85      /** Constant for quarter 4. */
86      public static final int LAST_QUARTER = 4;
87  
88      /** The first month in each quarter. */
89      public static final int[] FIRST_MONTH_IN_QUARTER = {
90          0, MonthConstants.JANUARY, MonthConstants.APRIL, MonthConstants.JULY,
91          MonthConstants.OCTOBER
92      };

94      /** The last month in each quarter. */
95      public static final int[] LAST_MONTH_IN_QUARTER = {
96          0, MonthConstants.MARCH, MonthConstants.JUNE, MonthConstants.SEPTEMBER,
97          MonthConstants.DECEMBER
98      };

119     /**
120      * Constructs a new quarter.
121      *
122      * @param year  the year (1900 to 9999).
123      * @param quarter  the quarter (1 to 4).
124      */
125     public Quarter(int quarter, int year) {
126         if ((quarter < FIRST_QUARTER) || (quarter > LAST_QUARTER)) {
127             throw new IllegalArgumentException("Quarter outside valid range.");
128         }
129         this.year = (short) year;
130         this.quarter = (byte) quarter;
131         peg(Calendar.getInstance());
132     }

161     /**
162      * Creates a new {@code Quarter} instance, using the specified
163      * zone and locale.
164      *
165      * @param time  the current time.
166      * @param zone  the time zone.
167      * @param locale  the locale.
168      *
169      * @since 1.0.12
170      */
171     public Quarter(Date time, TimeZone zone, Locale locale) {
172         Calendar calendar = Calendar.getInstance(zone, locale);
173         calendar.setTime(time);
174         int month = calendar.get(Calendar.MONTH) + 1;
175         this.quarter = (byte) SerialDate.monthCodeToQuarter(month);
176         this.year = (short) calendar.get(Calendar.YEAR);
177         peg(calendar);
178     }

253     /**
254      * Returns the quarter preceding this one.
255      *
256      * @return The quarter preceding this one (or {@code null} if this is
257      *     Q1 1900).
258      */
259     @Override
260     public RegularTimePeriod previous() {
261         Quarter result;
262         if (this.quarter > FIRST_QUARTER) {
263             result = new Quarter(this.quarter - 1, this.year);
264         }
265         else {
266             if (this.year > 1900) {
267                 result = new Quarter(LAST_QUARTER, this.year - 1);
268             }
269             else {
270                 result = null;
271             }
272         }
273         return result;
274     }

285             result = new Quarter(this.quarter + 1, this.year);
286         }
287         else {
288             if (this.year < 9999) {
289                 result = new Quarter(FIRST_QUARTER, this.year + 1);
290             }
291             else {
292                 result = null;
293             }
294         }

323             if (obj instanceof Quarter) {
324                 Quarter target = (Quarter) obj;
325                 return (this.quarter == target.getQuarter()
326                         && (this.year == target.getYearValue()));
327             }
328             else {

330             }
331         }

355     /**
356      * Returns an integer indicating the order of this Quarter object relative
357      * to the specified object:
358      *
359      * negative == before, zero == same, positive == after.
360      *
361      * @param o1  the object to compare
362      *
363      * @return negative == before, zero == same, positive == after.
364      */
365     @Override
366     public int compareTo(Object o1) {
367 
368         int result;
369 
370         // CASE 1 : Comparing to another Quarter object
371         // --------------------------------------------
372         if (o1 instanceof Quarter) {
373             Quarter q = (Quarter) o1;
374             result = this.year - q.getYearValue();
375             if (result == 0) {
376                 result = this.quarter - q.getQuarter();
377             }
378         }
379 
380         // CASE 2 : Comparing to another TimePeriod object
381         // -----------------------------------------------
382         else if (o1 instanceof RegularTimePeriod) {
383             // more difficult case - evaluate later...
384             result = 0;
385         }
386 
387         // CASE 3 : Comparing to a non-TimePeriod object
388         // ---------------------------------------------
389         else {
390             // consider time periods to be ordered after general objects
391             result = 1;
392         }
393 
394         return result;
395 
396     }

408     /**
409      * Returns the first millisecond in the Quarter, evaluated using the
410      * supplied calendar (which determines the time zone).
411      *
412      * @param calendar  the calendar ({@code null} not permitted).
413      *
414      * @return The first millisecond in the Quarter.
415      *
416      * @throws NullPointerException if {@code calendar} is
417      *     {@code null}.
418      */
419     @Override
420     public long getFirstMillisecond(Calendar calendar) {
421         int month = Quarter.FIRST_MONTH_IN_QUARTER[this.quarter];
422         calendar.set(this.year, month - 1, 1, 0, 0, 0);
423         calendar.set(Calendar.MILLISECOND, 0);
424         return calendar.getTimeInMillis();
425     }

427     /**
428      * Returns the last millisecond of the Quarter, evaluated using the
429      * supplied calendar (which determines the time zone).
430      *
431      * @param calendar  the calendar ({@code null} not permitted).
432      *
433      * @return The last millisecond of the Quarter.
434      *
435      * @throws NullPointerException if {@code calendar} is
436      *     {@code null}.
437      */
438     @Override
439     public long getLastMillisecond(Calendar calendar) {
440         int month = Quarter.LAST_MONTH_IN_QUARTER[this.quarter];
441         int eom = SerialDate.lastDayOfMonth(month, this.year);
442         calendar.set(this.year, month - 1, eom, 23, 59, 59);
443         calendar.set(Calendar.MILLISECOND, 999);
444         return calendar.getTimeInMillis();
445     }

447     /**
448      * Parses the string argument as a quarter.
449      * <P>
450      * This method should accept the following formats: "YYYY-QN" and "QN-YYYY",
451      * where the "-" can be a space, a forward-slash (/), comma or a dash (-).
452      * @param s A string representing the quarter.
453      *
454      * @return The quarter.
455      */
456     public static Quarter parseQuarter(String s) {
457 
458         // find the Q and the integer following it (remove both from the
459         // string)...
460         int i = s.indexOf("Q");
461         if (i == -1) {
462             throw new TimePeriodFormatException("Missing Q.");
463         }
464 
465         if (i == s.length() - 1) {
466             throw new TimePeriodFormatException("Q found at end of string.");
467         }
468 
469         String qstr = s.substring(i + 1, i + 2);
470         int quarter = Integer.parseInt(qstr);
471         String remaining = s.substring(0, i) + s.substring(i + 2, s.length());
472 
473         // replace any / , or - with a space
474         remaining = remaining.replace('/', ' ');
475         remaining = remaining.replace(',', ' ');
476         remaining = remaining.replace('-', ' ');
477 
478         // parse the string...
479         Year year = Year.parseYear(remaining.trim());
480         Quarter result = new Quarter(quarter, year);
481         return result;
482 
483     }
