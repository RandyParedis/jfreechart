64  /**
65   * An abstract class representing a unit of time.  Convenient methods are
66   * provided for calculating the next and previous time periods.  Conversion
67   * methods are defined that return the first and last milliseconds of the time
68   * period.  The results from these methods are timezone dependent.
69   * <P>
70   * This class is immutable, and all subclasses should be immutable also.
71   */
72  public abstract class RegularTimePeriod implements TimePeriod, Comparable,
73          MonthConstants {
74  
75      /**
76       * Creates a time period that includes the specified millisecond, assuming
77       * the given time zone.
78       *
79       * @param c  the time period class.
80       * @param millisecond  the time.
81       * @param zone  the time zone.
82       * @param locale  the locale.
83       *
84       * @return The time period.
85       */
86      public static RegularTimePeriod createInstance(Class c, Date millisecond,
87              TimeZone zone, Locale locale) {
88          RegularTimePeriod result = null;
89          try {
90              Constructor constructor = c.getDeclaredConstructor(
91                      new Class[] {Date.class, TimeZone.class, Locale.class});
92              result = (RegularTimePeriod) constructor.newInstance(
93                      new Object[] {millisecond, zone, locale});
94          }
95          catch (Exception e) {
96              // do nothing, so null is returned
97          }
98          return result;
99      }
100 
101     /**
102      * Returns a subclass of {@link RegularTimePeriod} that is smaller than
103      * the specified class.
104      *
105      * @param c  a subclass of {@link RegularTimePeriod}.
106      *
107      * @return A class.
108      */
109     public static Class downsize(Class c) {
110         if (c.equals(Year.class)) {
111             return Quarter.class;
112         }
113         else if (c.equals(Quarter.class)) {
114             return Month.class;
115         }
116         else if (c.equals(Month.class)) {
117             return Day.class;
118         }
119         else if (c.equals(Day.class)) {
120             return Hour.class;
121         }
122         else if (c.equals(Hour.class)) {
123             return Minute.class;
124         }
125         else if (c.equals(Minute.class)) {
126             return Second.class;
127         }
128         else if (c.equals(Second.class)) {
129             return Millisecond.class;
130         }
131         else {
132             return Millisecond.class;
133         }
134     }

247     /**
248      * Returns the millisecond closest to the middle of the time period.
249      *
250      * @return The middle millisecond.
251      */
252     public long getMiddleMillisecond() {
253         long m1 = getFirstMillisecond();
254         long m2 = getLastMillisecond();
255         return m1 + (m2 - m1) / 2;
256     }
257 
258     /**
259      * Returns the millisecond closest to the middle of the time period,
260      * evaluated using the supplied calendar (which incorporates a timezone).
261      *
262      * @param calendar  the calendar.
263      *
264      * @return The middle millisecond.
265      */
266     public long getMiddleMillisecond(Calendar calendar) {
267         long m1 = getFirstMillisecond(calendar);
268         long m2 = getLastMillisecond(calendar);
269         return m1 + (m2 - m1) / 2;
270     }
271 
272     /**
273      * Returns the millisecond (relative to the epoch) corresponding to the 
274      * specified {@code anchor} using the supplied {@code calendar} 
275      * (which incorporates a time zone).
276      * 
277      * @param anchor  the anchor ({@code null} not permitted).
278      * @param calendar  the calendar ({@code null} not permitted).
279      * 
280      * @return Milliseconds since the epoch.
281      * 
282      * @since 1.0.18
283      */
284     public long getMillisecond(TimePeriodAnchor anchor, Calendar calendar) {
285         if (anchor.equals(TimePeriodAnchor.START)) {
286             return getFirstMillisecond(calendar);
287         } else if (anchor.equals(TimePeriodAnchor.MIDDLE)) {
288             return getMiddleMillisecond(calendar);
289         } else if (anchor.equals(TimePeriodAnchor.END)) {
290             return getLastMillisecond(calendar);
291         } else {
292             throw new IllegalStateException("Unrecognised anchor: " + anchor);
293         }
294     }
