166         super();
167         setDefaultToolTipGenerator(toolTipGenerator);
168         setURLGenerator(urlGenerator);
169 
170         if (type == AREA) {
171             this.plotArea = true;
172         }
173         else if (type == SHAPES) {
174             this.shapesVisible = true;
175         }
176         else if (type == AREA_AND_SHAPES) {
177             this.plotArea = true;
178             this.shapesVisible = true;
179         }
180         this.showOutline = false;
181         this.stepPoint = 1.0;
182     }

375     /**
376      * Draws the visual representation of a single data item.
377      *
378      * @param g2  the graphics device.
379      * @param state  the renderer state.
380      * @param dataArea  the area within which the data is being drawn.
381      * @param info  collects information about the drawing.
382      * @param plot  the plot (can be used to obtain standard color information
383      *              etc).
384      * @param domainAxis  the domain axis.
385      * @param rangeAxis  the range axis.
386      * @param dataset  the dataset.
387      * @param series  the series index (zero-based).
388      * @param item  the item index (zero-based).
389      * @param crosshairState  crosshair information for the plot
390      *                        ({@code null} permitted).
391      * @param pass  the pass index.
392      */
393     @Override
394     public void drawItem(Graphics2D g2, XYItemRendererState state, 
395             Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
396             ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
397             int series, int item, CrosshairState crosshairState, int pass) {
398 
399         PlotOrientation orientation = plot.getOrientation();
400 
401         // Get the item count for the series, so that we can know which is the
402         // end of the series.
403         int itemCount = dataset.getItemCount(series);
404 
405         Paint paint = getPaint().getItemPaint(series, item);
406         Stroke seriesStroke = getItemStroke(series, item);
407         g2.setPaint(paint);
408         g2.setStroke(seriesStroke);
409 
410         // get the data point...
411         double x1 = dataset.getXValue(series, item);
412         double y1 = dataset.getYValue(series, item);
413         double x = x1;
414         double y = Double.isNaN(y1) ? getRangeBase() : y1;
415         double transX1 = domainAxis.valueToJava2D(x, dataArea,
416                 plot.getDomainAxisEdge());
417         double transY1 = rangeAxis.valueToJava2D(y, dataArea,
418                 plot.getRangeAxisEdge());
419 
420         // avoid possible sun.dc.pr.PRException: endPath: bad path
421         transY1 = restrictValueToDataArea(transY1, plot, dataArea);
422 
423         if (this.pArea == null && !Double.isNaN(y1)) {
424 
425             // Create a new Area for the series
426             this.pArea = new Polygon();
427 
428             // start from Y = rangeBase
429             double transY2 = rangeAxis.valueToJava2D(getRangeBase(), dataArea,
430                     plot.getRangeAxisEdge());
431 
432             // avoid possible sun.dc.pr.PRException: endPath: bad path
433             transY2 = restrictValueToDataArea(transY2, plot, dataArea);
434 
435             // The first point is (x, this.baseYValue)
436             if (orientation == PlotOrientation.VERTICAL) {
437                 this.pArea.addPoint((int) transX1, (int) transY2);
438             }
439             else if (orientation == PlotOrientation.HORIZONTAL) {
440                 this.pArea.addPoint((int) transY2, (int) transX1);
441             }
442         }
443 
444         double transX0;
445         double transY0;
446 
447         double x0;
448         double y0;
449         if (item > 0) {
450             // get the previous data point...
451             x0 = dataset.getXValue(series, item - 1);
452             y0 = Double.isNaN(y1) ? y1 : dataset.getYValue(series, item - 1);
453 
454             x = x0;
455             y = Double.isNaN(y0) ? getRangeBase() : y0;
456             transX0 = domainAxis.valueToJava2D(x, dataArea,
457                     plot.getDomainAxisEdge());
458             transY0 = rangeAxis.valueToJava2D(y, dataArea,
459                     plot.getRangeAxisEdge());
460 
461             // avoid possible sun.dc.pr.PRException: endPath: bad path
462             transY0 = restrictValueToDataArea(transY0, plot, dataArea);
463 
464             if (Double.isNaN(y1)) {
465                 // NULL value -> insert point on base line
466                 // instead of 'step point'
467                 transX1 = transX0;
468                 transY0 = transY1;
469             }
470             if (transY0 != transY1) {
471                 // not just a horizontal bar but need to perform a 'step'.
472                 double transXs = transX0 + (getStepPoint()
473                         * (transX1 - transX0));
474                 if (orientation == PlotOrientation.VERTICAL) {
475                     this.pArea.addPoint((int) transXs, (int) transY0);
476                     this.pArea.addPoint((int) transXs, (int) transY1);
477                 }
478                 else if (orientation == PlotOrientation.HORIZONTAL) {
479                     this.pArea.addPoint((int) transY0, (int) transXs);
480                     this.pArea.addPoint((int) transY1, (int) transXs);
481                 }
482             }
483         }
484 
485         Shape shape = null;
486         if (!Double.isNaN(y1)) {
487             // Add each point to Area (x, y)
488             if (orientation == PlotOrientation.VERTICAL) {
489                 this.pArea.addPoint((int) transX1, (int) transY1);
490             }
491             else if (orientation == PlotOrientation.HORIZONTAL) {
492                 this.pArea.addPoint((int) transY1, (int) transX1);
493             }
494 
495             if (getShapesVisible()) {
496                 shape = getItemShape(series, item);
497                 if (orientation == PlotOrientation.VERTICAL) {
498                     shape = ShapeUtils.createTranslatedShape(shape,
499                             transX1, transY1);
500                 }
501                 else if (orientation == PlotOrientation.HORIZONTAL) {
502                     shape = ShapeUtils.createTranslatedShape(shape,
503                             transY1, transX1);
504                 }
505                 if (isShapesFilled()) {
506                     g2.fill(shape);
507                 }
508                 else {
509                     g2.draw(shape);
510                 }
511             }
512             else {
513                 if (orientation == PlotOrientation.VERTICAL) {
514                     shape = new Rectangle2D.Double(transX1 - 2, transY1 - 2,
515                             4.0, 4.0);
516                 } else if (orientation == PlotOrientation.HORIZONTAL) {
517                     shape = new Rectangle2D.Double(transY1 - 2, transX1 - 2,
518                             4.0, 4.0);
519                 }
520             }
521         }
522 
523         // Check if the item is the last item for the series or if it
524         // is a NULL value and number of items > 0.  We can't draw an area for
525         // a single point.
526         if (getPlotArea() && item > 0 && this.pArea != null
527                           && (item == (itemCount - 1) || Double.isNaN(y1))) {
528 
529             double transY2 = rangeAxis.valueToJava2D(getRangeBase(), dataArea,
530                     plot.getRangeAxisEdge());
531 
532             // avoid possible sun.dc.pr.PRException: endPath: bad path
533             transY2 = restrictValueToDataArea(transY2, plot, dataArea);
534 
535             if (orientation == PlotOrientation.VERTICAL) {
536                 // Add the last point (x,0)
537                 this.pArea.addPoint((int) transX1, (int) transY2);
538             }
539             else if (orientation == PlotOrientation.HORIZONTAL) {
540                 // Add the last point (x,0)
541                 this.pArea.addPoint((int) transY2, (int) transX1);
542             }
543 
544             // fill the polygon
545             g2.fill(this.pArea);
546 
547             // draw an outline around the Area.
548             if (isOutline()) {
549                 g2.setStroke(plot.getOutlineStroke());
550                 g2.setPaint(plot.getOutlinePaint());
551                 g2.draw(this.pArea);
552             }
553 
554             // start new area when needed (see above)
555             this.pArea = null;
556         }
557 
558         // do we need to update the crosshair values?
559         if (!Double.isNaN(y1)) {
560             int datasetIndex = plot.indexOf(dataset);
561             updateCrosshairValues(crosshairState, x1, y1, datasetIndex,
562                     transX1, transY1, orientation);
563         }
564 
565         // collect entity and tool tip information...
566         EntityCollection entities = state.getEntityCollection();
567         if (entities != null && shape != null) {
568             addEntity(entities, shape, dataset, series, item, 0.0, 0.0);
569         }
570     }

621     /**
622      * Helper method which returns a value if it lies
623      * inside the visible dataArea and otherwise the corresponding
624      * coordinate on the border of the dataArea. The PlotOrientation
625      * is taken into account.
626      * Useful to avoid possible sun.dc.pr.PRException: endPath: bad path
627      * which occurs when trying to draw lines/shapes which in large part
628      * lie outside of the visible dataArea.
629      *
630      * @param value the value which shall be
631      * @param dataArea  the area within which the data is being drawn.
632      * @param plot  the plot (can be used to obtain standard color
633      *              information etc).
634      * @return {@code double} value inside the data area.
635      */
636     protected static double restrictValueToDataArea(double value,
637                                                     XYPlot plot,
638                                                     Rectangle2D dataArea) {
639         double min = 0;
640         double max = 0;
641         if (plot.getOrientation() == PlotOrientation.VERTICAL) {
642             min = dataArea.getMinY();
643             max = dataArea.getMaxY();
644         }
645         else if (plot.getOrientation() ==  PlotOrientation.HORIZONTAL) {
646             min = dataArea.getMinX();
647             max = dataArea.getMaxX();
648         }
649         if (value < min) {
650             value = min;
651         }
652         else if (value > max) {
653             value = max;
654         }
655         return value;
656     }
