125 /**
126  * Area item renderer for an {@link XYPlot}.  This class can draw (a) shapes at
127  * each point, or (b) lines between points, or (c) both shapes and lines,
128  * or (d) filled areas, or (e) filled areas and shapes. The example shown here
129  * is generated by the {@code XYAreaRendererDemo1.java} program included
130  * in the JFreeChart demo collection:
131  * <br><br>
132  * <img src="../../../../../images/XYAreaRendererSample.png"
133  * alt="XYAreaRendererSample.png">
134  */
135 public class XYAreaRenderer extends AbstractXYItemRenderer
136         implements XYItemRenderer, PublicCloneable {

234     /**
235      * Constructs a new renderer.  To specify the type of renderer, use one of
236      * the constants: {@code SHAPES}, {@code LINES}, {@code SHAPES_AND_LINES}, 
237      * {@code AREA} or {@code AREA_AND_SHAPES}.
238      *
239      * @param type  the type of renderer.
240      * @param toolTipGenerator  the tool tip generator ({@code null} permitted).
241      * @param urlGenerator  the URL generator ({@code null} permitted).
242      */
243     public XYAreaRenderer(int type, XYToolTipGenerator toolTipGenerator,
244                           XYURLGenerator urlGenerator) {
245 
246         super();
247         setDefaultToolTipGenerator(toolTipGenerator);
248         setURLGenerator(urlGenerator);
249 
250         if (type == SHAPES) {
251             this.plotShapes = true;
252         }
253         if (type == LINES) {
254             this.plotLines = true;
255         }
256         if (type == SHAPES_AND_LINES) {
257             this.plotShapes = true;
258             this.plotLines = true;
259         }
260         if (type == AREA) {
261             this.plotArea = true;
262         }
263         if (type == AREA_AND_SHAPES) {
264             this.plotArea = true;
265             this.plotShapes = true;
266         }
267         this.showOutline = false;
268         GeneralPath area = new GeneralPath();
269         area.moveTo(0.0f, -4.0f);
270         area.lineTo(3.0f, -2.0f);
271         area.lineTo(4.0f, 4.0f);
272         area.lineTo(-4.0f, 4.0f);
273         area.lineTo(-3.0f, -2.0f);
274         area.closePath();
275         this.legendArea = area;
276         this.useFillPaint = false;
277         this.gradientTransformer = new StandardGradientPaintTransformer();
278     }

417     @Override
418     public XYItemRendererState initialise(Graphics2D g2, Rectangle2D dataArea,
419             XYPlot plot, XYDataset data, PlotRenderingInfo info) {
420         XYAreaRendererState state = new XYAreaRendererState(info);

428     /**
429      * Returns a default legend item for the specified series.  Subclasses
430      * should override this method to generate customised items.
431      *
432      * @param datasetIndex  the dataset index (zero-based).
433      * @param series  the series index (zero-based).
434      *
435      * @return A legend item for the series.
436      */
437     @Override
438     public LegendItem getLegendItem(int datasetIndex, int series) {
439         LegendItem result = null;
440         XYPlot xyplot = getPlot();
441         if (xyplot != null) {
442             XYDataset dataset = xyplot.getDataset(datasetIndex);
443             if (dataset != null) {
444                 XYSeriesLabelGenerator lg = getLegendItemLabelGenerator();
445                 String label = lg.generateLabel(dataset, series);
446                 String description = label;
447                 String toolTipText = null;
448                 if (getLegendItemToolTipGenerator() != null) {
449                     toolTipText = getLegendItemToolTipGenerator().generateLabel(
450                             dataset, series);
451                 }
452                 String urlText = null;
453                 if (getLegendItemURLGenerator() != null) {
454                     urlText = getLegendItemURLGenerator().generateLabel(
455                             dataset, series);
456                 }
457                 Paint paint = getPaint().lookupSeriesPaint(series);
458                 result = new LegendItem(label, description, toolTipText,
459                         urlText, this.legendArea, paint);
460                 result.setLabelFont(lookupLegendTextFont(series));
461                 Paint labelPaint = lookupLegendTextPaint(series);
462                 if (labelPaint != null) {
463                     result.setLabelPaint(labelPaint);
464                 }
465                 result.setDataset(dataset);
466                 result.setDatasetIndex(datasetIndex);
467                 result.setSeriesKey(dataset.getSeriesKey(series));
468                 result.setSeriesIndex(series);
469             }
470         }
471         return result;
472     }

474     /**
475      * Draws the visual representation of a single data item.
476      *
477      * @param g2  the graphics device.
478      * @param state  the renderer state.
479      * @param dataArea  the area within which the data is being drawn.
480      * @param info  collects information about the drawing.
481      * @param plot  the plot (can be used to obtain standard color information
482      *              etc).
483      * @param domainAxis  the domain axis.
484      * @param rangeAxis  the range axis.
485      * @param dataset  the dataset.
486      * @param series  the series index (zero-based).
487      * @param item  the item index (zero-based).
488      * @param crosshairState  crosshair information for the plot
489      *                        ({@code null} permitted).
490      * @param pass  the pass index.
491      */
492     @Override
493     public void drawItem(Graphics2D g2, XYItemRendererState state,
494             Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
495             ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
496             int series, int item, CrosshairState crosshairState, int pass) {
497 
498         if (!getVisibility().getItemVisible(series, item)) {
499             return;
500         }
501         XYAreaRendererState areaState = (XYAreaRendererState) state;
502 
503         // get the data point...
504         double x1 = dataset.getXValue(series, item);
505         double y1 = dataset.getYValue(series, item);
506         if (Double.isNaN(y1)) {
507             y1 = 0.0;
508         }
509         double transX1 = domainAxis.valueToJava2D(x1, dataArea,
510                 plot.getDomainAxisEdge());
511         double transY1 = rangeAxis.valueToJava2D(y1, dataArea,
512                 plot.getRangeAxisEdge());
513 
514         // get the previous point and the next point so we can calculate a
515         // "hot spot" for the area (used by the chart entity)...
516         int itemCount = dataset.getItemCount(series);
517         double x0 = dataset.getXValue(series, Math.max(item - 1, 0));
518         double y0 = dataset.getYValue(series, Math.max(item - 1, 0));
519         if (Double.isNaN(y0)) {
520             y0 = 0.0;
521         }
522         double transX0 = domainAxis.valueToJava2D(x0, dataArea,
523                 plot.getDomainAxisEdge());
524         double transY0 = rangeAxis.valueToJava2D(y0, dataArea,
525                 plot.getRangeAxisEdge());
526 
527         double x2 = dataset.getXValue(series, Math.min(item + 1,
528                 itemCount - 1));
529         double y2 = dataset.getYValue(series, Math.min(item + 1,
530                 itemCount - 1));
531         if (Double.isNaN(y2)) {
532             y2 = 0.0;
533         }
534         double transX2 = domainAxis.valueToJava2D(x2, dataArea,
535                 plot.getDomainAxisEdge());
536         double transY2 = rangeAxis.valueToJava2D(y2, dataArea,
537                 plot.getRangeAxisEdge());
538 
539         double transZero = rangeAxis.valueToJava2D(0.0, dataArea,
540                 plot.getRangeAxisEdge());
541 
542         if (item == 0) {  // create a new area polygon for the series
543             areaState.area = new GeneralPath();
544             // the first point is (x, 0)
545             double zero = rangeAxis.valueToJava2D(0.0, dataArea,
546                     plot.getRangeAxisEdge());
547             if (plot.getOrientation().isVertical()) {
548                 moveTo(areaState.area, transX1, zero);
549             } else if (plot.getOrientation().isHorizontal()) {
550                 moveTo(areaState.area, zero, transX1);
551             }
552         }
553 
554         // Add each point to Area (x, y)
555         if (plot.getOrientation().isVertical()) {
556             lineTo(areaState.area, transX1, transY1);
557         } else if (plot.getOrientation().isHorizontal()) {
558             lineTo(areaState.area, transY1, transX1);
559         }
560 
561         PlotOrientation orientation = plot.getOrientation();
562         Paint paint = getPaint().getItemPaint(series, item);
563         Stroke stroke = getItemStroke(series, item);
564         g2.setPaint(paint);
565         g2.setStroke(stroke);
566 
567         Shape shape;
568         if (getPlotShapes()) {
569             shape = getItemShape(series, item);
570             if (orientation == PlotOrientation.VERTICAL) {
571                 shape = ShapeUtils.createTranslatedShape(shape, transX1,
572                         transY1);
573             } else if (orientation == PlotOrientation.HORIZONTAL) {
574                 shape = ShapeUtils.createTranslatedShape(shape, transY1,
575                         transX1);
576             }
577             g2.draw(shape);
578         }
579 
580         if (getPlotLines()) {
581             if (item > 0) {
582                 if (plot.getOrientation() == PlotOrientation.VERTICAL) {
583                     areaState.line.setLine(transX0, transY0, transX1, transY1);
584                 } else if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
585                     areaState.line.setLine(transY0, transX0, transY1, transX1);
586                 }
587                 g2.draw(areaState.line);
588             }
589         }
590 
591         // Check if the item is the last item for the series.
592         // and number of items > 0.  We can't draw an area for a single point.
593         if (getPlotArea() && item > 0 && item == (itemCount - 1)) {
594 
595             if (orientation == PlotOrientation.VERTICAL) {
596                 // Add the last point (x,0)
597                 lineTo(areaState.area, transX1, transZero);
598                 areaState.area.closePath();
599             } else if (orientation == PlotOrientation.HORIZONTAL) {
600                 // Add the last point (x,0)
601                 lineTo(areaState.area, transZero, transX1);
602                 areaState.area.closePath();
603             }
604 
605             if (this.useFillPaint) {
606                 paint = getPaint().lookupSeriesFillPaint(series);
607             }
608             if (paint instanceof GradientPaint) {
609                 GradientPaint gp = (GradientPaint) paint;
610                 GradientPaint adjGP = this.gradientTransformer.transform(gp,
611                         dataArea);
612                 g2.setPaint(adjGP);
613             }
614             g2.fill(areaState.area);
615 
616             // draw an outline around the Area.
617             if (isOutline()) {
618                 Shape area = areaState.area;
619 
620                 // Java2D has some issues drawing dashed lines around "large"
621                 // geometrical shapes - for example, see bug 6620013 in the
622                 // Java bug database.  So, we'll check if the outline is
623                 // dashed and, if it is, do our own clipping before drawing
624                 // the outline...
625                 Stroke outlineStroke = lookupSeriesOutlineStroke(series);
626                 if (outlineStroke instanceof BasicStroke) {
627                     BasicStroke bs = (BasicStroke) outlineStroke;
628                     if (bs.getDashArray() != null) {
629                         Area poly = new Area(areaState.area);
630                         // we make the clip region slightly larger than the
631                         // dataArea so that the clipped edges don't show lines
632                         // on the chart
633                         Area clip = new Area(new Rectangle2D.Double(
634                                 dataArea.getX() - 5.0, dataArea.getY() - 5.0,
635                                 dataArea.getWidth() + 10.0,
636                                 dataArea.getHeight() + 10.0));
637                         poly.intersect(clip);
638                         area = poly;
639                     }
640                 } // end of workaround
641 
642                 g2.setStroke(outlineStroke);
643                 g2.setPaint(getPaint().lookupSeriesOutlinePaint(series));
644                 g2.draw(area);
645             }
646         }
647 
648         int datasetIndex = plot.indexOf(dataset);
649         updateCrosshairValues(crosshairState, x1, y1, datasetIndex,
650                 transX1, transY1, orientation);
651 
652         // collect entity and tool tip information...
653         EntityCollection entities = state.getEntityCollection();
654         if (entities != null) {
655             GeneralPath hotspot = new GeneralPath();
656             if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
657                 moveTo(hotspot, transZero, ((transX0 + transX1) / 2.0));
658                 lineTo(hotspot, ((transY0 + transY1) / 2.0), ((transX0 + transX1) / 2.0));
