65       * Confirm that the equals method can distinguish all the required fields.

67      @Test
68      public void testEquals() {
69          OHLCSeriesCollection c1 = new OHLCSeriesCollection();
70          OHLCSeriesCollection c2 = new OHLCSeriesCollection();
71          assertEquals(c1, c2);
72  
73          // add a series
74          OHLCSeries s1 = new OHLCSeries("Series");
75          s1.add(new Year(2006), 1.0, 1.1, 1.2, 1.3);
76          c1.addSeries(s1);
77          assertFalse(c1.equals(c2));
78          OHLCSeries s2 = new OHLCSeries("Series");
79          s2.add(new Year(2006), 1.0, 1.1, 1.2, 1.3);
80          c2.addSeries(s2);
81          assertTrue(c1.equals(c2));
82  
83          // add an empty series
84          c1.addSeries(new OHLCSeries("Empty Series"));
85          assertFalse(c1.equals(c2));
86          c2.addSeries(new OHLCSeries("Empty Series"));
87          assertTrue(c1.equals(c2));
88  
89          c1.setXPosition(TimePeriodAnchor.END);
90          assertFalse(c1.equals(c2));
91          c2.setXPosition(TimePeriodAnchor.END);
92          assertTrue(c1.equals(c2));
93  
94      }

102         OHLCSeries s1 = new OHLCSeries("Series");
103         s1.add(new Year(2006), 1.0, 1.1, 1.2, 1.3);
104         c1.addSeries(s1);
105         OHLCSeriesCollection c2 = (OHLCSeriesCollection) c1.clone();
106         assertTrue(c1 != c2);
107         assertTrue(c1.getClass() == c2.getClass());
108         assertTrue(c1.equals(c2));
109 
110         // check independence
111         s1.setDescription("XYZ");
112         assertFalse(c1.equals(c2));

213      * A simple check for the removeAllSeries() method.

215     @Test
216     public void testRemoveAllSeries() {
217         OHLCSeriesCollection c1 = new OHLCSeriesCollection();
218         c1.addChangeListener(this);
219 
220         // there should be no change event when clearing an empty series
221         this.lastEvent = null;
222         c1.removeAllSeries();
223         assertNull(this.lastEvent);
224 
225         OHLCSeries s1 = new OHLCSeries("Series 1");
226         OHLCSeries s2 = new OHLCSeries("Series 2");
227         c1.addSeries(s1);
228         c1.addSeries(s2);
229         c1.removeAllSeries();
230         assertEquals(0, c1.getSeriesCount());
231         assertNotNull(this.lastEvent);
232         this.lastEvent = null;  // clean up
233     }
