148     /**
149      * Constructs a new renderer.
150      *
151      * @param labelGenerator  the tool tip generator to use ({@code null} 
152      *     permitted).
153      * @param urlGenerator  the URL generator ({@code null} permitted).
154      */
155     public XYAreaRenderer2(XYToolTipGenerator labelGenerator,
156                            XYURLGenerator urlGenerator) {
157         super();
158         this.showOutline = false;
159         setDefaultToolTipGenerator(labelGenerator);
160         setURLGenerator(urlGenerator);
161         GeneralPath area = new GeneralPath();
162         area.moveTo(0.0f, -4.0f);
163         area.lineTo(3.0f, -2.0f);
164         area.lineTo(4.0f, 4.0f);
165         area.lineTo(-4.0f, 4.0f);
166         area.lineTo(-3.0f, -2.0f);
167         area.closePath();
168         this.legendArea = area;
169     }

222     /**
223      * Returns a default legend item for the specified series.  Subclasses
224      * should override this method to generate customised items.
225      *
226      * @param datasetIndex  the dataset index (zero-based).
227      * @param series  the series index (zero-based).
228      *
229      * @return A legend item for the series.
230      */
231     @Override
232     public LegendItem getLegendItem(int datasetIndex, int series) {
233         LegendItem result = null;
234         XYPlot xyplot = getPlot();
235         if (xyplot != null) {
236             XYDataset dataset = xyplot.getDataset(datasetIndex);
237             if (dataset != null) {
238                 XYSeriesLabelGenerator lg = getLegendItemLabelGenerator();
239                 String label = lg.generateLabel(dataset, series);
240                 String description = label;
241                 String toolTipText = null;
242                 if (getLegendItemToolTipGenerator() != null) {
243                     toolTipText = getLegendItemToolTipGenerator().generateLabel(
244                             dataset, series);
245                 }
246                 String urlText = null;
247                 if (getLegendItemURLGenerator() != null) {
248                     urlText = getLegendItemURLGenerator().generateLabel(
249                             dataset, series);
250                 }
251                 Paint paint = getPaint().lookupSeriesPaint(series);
252                 result = new LegendItem(label, description, toolTipText,
253                         urlText, this.legendArea, paint);
254                 result.setLabelFont(lookupLegendTextFont(series));
255                 Paint labelPaint = lookupLegendTextPaint(series);
256                 if (labelPaint != null) {
257                     result.setLabelPaint(labelPaint);
258                 }
259                 result.setDataset(dataset);
260                 result.setDatasetIndex(datasetIndex);
261                 result.setSeriesKey(dataset.getSeriesKey(series));
262                 result.setSeriesIndex(series);
263             }
264         }
265         return result;
266     }

268     /**
269      * Draws the visual representation of a single data item.
270      *
271      * @param g2  the graphics device.
272      * @param state  the renderer state.
273      * @param dataArea  the area within which the data is being drawn.
274      * @param info  collects information about the drawing.
275      * @param plot  the plot (can be used to obtain standard color
276      *              information etc).
277      * @param domainAxis  the domain axis.
278      * @param rangeAxis  the range axis.
279      * @param dataset  the dataset.
280      * @param series  the series index (zero-based).
281      * @param item  the item index (zero-based).
282      * @param crosshairState  crosshair information for the plot
283      *                        ({@code null} permitted).
284      * @param pass  the pass index.
285      */
286     @Override
287     public void drawItem(Graphics2D g2, XYItemRendererState state,
288          Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
289          ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
290          int series, int item, CrosshairState crosshairState, int pass) {
291 
292         if (!getVisibility().getItemVisible(series, item)) {
293             return;
294         }
295         // get the data point...
296         double x1 = dataset.getXValue(series, item);
297         double y1 = dataset.getYValue(series, item);
298         if (Double.isNaN(y1)) {
299             y1 = 0.0;
300         }
301 
302         double transX1 = domainAxis.valueToJava2D(x1, dataArea,
303                 plot.getDomainAxisEdge());
304         double transY1 = rangeAxis.valueToJava2D(y1, dataArea,
305                 plot.getRangeAxisEdge());
306 
307         // get the previous point and the next point so we can calculate a
308         // "hot spot" for the area (used by the chart entity)...
309         double x0 = dataset.getXValue(series, Math.max(item - 1, 0));
310         double y0 = dataset.getYValue(series, Math.max(item - 1, 0));
311         if (Double.isNaN(y0)) {
312             y0 = 0.0;
313         }
314         double transX0 = domainAxis.valueToJava2D(x0, dataArea,
315                 plot.getDomainAxisEdge());
316         double transY0 = rangeAxis.valueToJava2D(y0, dataArea,
317                 plot.getRangeAxisEdge());
318 
319         int itemCount = dataset.getItemCount(series);
320         double x2 = dataset.getXValue(series, Math.min(item + 1,
321                 itemCount - 1));
322         double y2 = dataset.getYValue(series, Math.min(item + 1,
323                 itemCount - 1));
324         if (Double.isNaN(y2)) {
325             y2 = 0.0;
326         }
327         double transX2 = domainAxis.valueToJava2D(x2, dataArea,
328                 plot.getDomainAxisEdge());
329         double transY2 = rangeAxis.valueToJava2D(y2, dataArea,
330                 plot.getRangeAxisEdge());
331 
332         double transZero = rangeAxis.valueToJava2D(0.0, dataArea,
333                 plot.getRangeAxisEdge());
334         GeneralPath hotspot = new GeneralPath();
335         if (plot.getOrientation() == PlotOrientation.HORIZONTAL) {
336             moveTo(hotspot, transZero, ((transX0 + transX1) / 2.0));
337             lineTo(hotspot, ((transY0 + transY1) / 2.0),
338                             ((transX0 + transX1) / 2.0));
339             lineTo(hotspot, transY1, transX1);
340             lineTo(hotspot, ((transY1 + transY2) / 2.0),
341                             ((transX1 + transX2) / 2.0));
342             lineTo(hotspot, transZero, ((transX1 + transX2) / 2.0));
343         }
344         else {  // vertical orientation
345             moveTo(hotspot, ((transX0 + transX1) / 2.0), transZero);
346             lineTo(hotspot, ((transX0 + transX1) / 2.0),
347                             ((transY0 + transY1) / 2.0));
348             lineTo(hotspot, transX1, transY1);
349             lineTo(hotspot, ((transX1 + transX2) / 2.0),
350                             ((transY1 + transY2) / 2.0));
351             lineTo(hotspot, ((transX1 + transX2) / 2.0), transZero);
352         }
353         hotspot.closePath();
354 
355         PlotOrientation orientation = plot.getOrientation();
356         Paint paint = getPaint().getItemPaint(series, item);
357         Stroke stroke = getStroke().getItemStroke(series, item);
358         g2.setPaint(paint);
359         g2.setStroke(stroke);
360 
361         // Check if the item is the last item for the series.
362         // and number of items > 0.  We can't draw an area for a single point.
363         g2.fill(hotspot);
364 
365         // draw an outline around the Area.
366         if (isOutline()) {
367             g2.setStroke(getStroke().lookupSeriesOutlineStroke(series));
368             g2.setPaint(getPaint().lookupSeriesOutlinePaint(series));
369             g2.draw(hotspot);
370         }
371         int datasetIndex = plot.indexOf(dataset);
372         updateCrosshairValues(crosshairState, x1, y1, datasetIndex,
373                 transX1, transY1, orientation);
374 
375         // collect entity and tool tip information...
376         if (state.getInfo() != null) {
377             EntityCollection entities = state.getEntityCollection();
378             if (entities != null) {
379                 // limit the entity hotspot area to the data area
380                 Area dataAreaHotspot = new Area(hotspot);
381                 dataAreaHotspot.intersect(new Area(dataArea));
382                 if (!dataAreaHotspot.isEmpty()) {
383                     String tip = null;
384                     XYToolTipGenerator generator = getToolTipGenerator(series,
385                          item);
386                     if (generator != null) {
387                         tip = generator.generateToolTip(dataset, series, item);
388                     }
389                     String url = null;
390                     if (getURLGenerator() != null) {
391                         url = getURLGenerator().generateURL(dataset, series, 
392                                 item);
393                     }
394                     XYItemEntity entity = new XYItemEntity(dataAreaHotspot, 
395                             dataset, series, item, tip, url);
396                     entities.add(entity);
397                 }
398             }
399         }
400

403     /**
404      * Tests this renderer for equality with an arbitrary object.
405      *
406      * @param obj  the object ({@code null} not permitted).
407      *
408      * @return A boolean.
409      */
410     @Override
411     public boolean equals(Object obj) {
412         if (obj == this) {
413             return true;
414         }
415         if (!(obj instanceof XYAreaRenderer2)) {
416             return false;
417         }
418         XYAreaRenderer2 that = (XYAreaRenderer2) obj;
419         if (this.showOutline != that.showOutline) {
420             return false;
421         }
422         if (!ShapeUtils.equal(this.legendArea, that.legendArea)) {
423             return false;
424         }
425         return super.equals(obj);
426     }
