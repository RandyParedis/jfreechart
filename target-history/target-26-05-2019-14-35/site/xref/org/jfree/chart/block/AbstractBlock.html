387     protected double trimToContentWidth(double fixedWidth) {
388         double result = this.margin.trimWidth(fixedWidth);

412     /**
413      * Returns a constraint for the content of this block that will result in
414      * the bounds of the block matching the specified constraint.
415      *
416      * @param c  the outer constraint ({@code null} not permitted).
417      *
418      * @return The content constraint.
419      */
420     protected RectangleConstraint toContentConstraint(RectangleConstraint c) {
421         Args.nullNotPermitted(c, "c");
422         if (c.equals(RectangleConstraint.NONE)) {
423             return c;
424         }
425         double w = c.getWidth();
426         Range wr = c.getWidthRange();
427         double h = c.getHeight();
428         Range hr = c.getHeightRange();
429         double ww = trimToContentWidth(w);
430         double hh = trimToContentHeight(h);
431         Range wwr = trimToContentWidth(wr);
432         Range hhr = trimToContentHeight(hr);
433         return new RectangleConstraint(ww, wwr, c.getWidthConstraintType(),
434             hh, hhr, c.getHeightConstraintType());
435     }

437     private Range trimToContentWidth(Range r) {
438         if (r == null) {
439             return null;
440         }
441         double lowerBound = 0.0;
442         double upperBound = Double.POSITIVE_INFINITY;
443         if (r.getLowerBound() > 0.0) {
444             lowerBound = trimToContentWidth(r.getLowerBound());
445         }
446         if (r.getUpperBound() < Double.POSITIVE_INFINITY) {
447             upperBound = trimToContentWidth(r.getUpperBound());
448         }
449         return new Range(lowerBound, upperBound);
450     }

452     private Range trimToContentHeight(Range r) {
453         if (r == null) {
454             return null;
455         }
456         double lowerBound = 0.0;
457         double upperBound = Double.POSITIVE_INFINITY;
458         if (r.getLowerBound() > 0.0) {
459             lowerBound = trimToContentHeight(r.getLowerBound());
460         }
461         if (r.getUpperBound() < Double.POSITIVE_INFINITY) {
462             upperBound = trimToContentHeight(r.getUpperBound());
463         }
464         return new Range(lowerBound, upperBound);
465     }

549     /**
550      * Tests this block for equality with an arbitrary object.
551      *
552      * @param obj  the object ({@code null} permitted).
553      *
554      * @return A boolean.
555      */
556     @Override
557     public boolean equals(Object obj) {
558         if (obj == this) {
559             return true;
560         }
561         if (!(obj instanceof AbstractBlock)) {
562             return false;
563         }
564         AbstractBlock that = (AbstractBlock) obj;
565         if (!ObjectUtils.equal(this.id, that.id)) {
566             return false;
567         }
568         if (!this.frame.equals(that.frame)) {
569             return false;
570         }
571         if (!this.bounds.equals(that.bounds)) {
572             return false;
573         }
574         if (!this.margin.equals(that.margin)) {
575             return false;
576         }
577         if (!this.padding.equals(that.padding)) {
578             return false;
579         }
580         if (this.height != that.height) {
581             return false;
582         }
583         if (this.width != that.width) {
584             return false;
585         }
586         return true;
587     }

589     /**
590      * Returns a clone of this block.
591      *
592      * @return A clone.
593      *
594      * @throws CloneNotSupportedException if there is a problem creating the
595      *         clone.
596      */
597     @Override
598     public Object clone() throws CloneNotSupportedException {
599         AbstractBlock clone = (AbstractBlock) super.clone();
600         clone.bounds = (Rectangle2D) ShapeUtils.clone(this.bounds);
601         if (this.frame instanceof PublicCloneable) {
602             PublicCloneable pc = (PublicCloneable) this.frame;
603             clone.frame = (BlockFrame) pc.clone();
604         }
605         return clone;
606     }
