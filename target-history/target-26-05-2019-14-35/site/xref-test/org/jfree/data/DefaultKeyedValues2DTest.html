91      @Test
92      public void testCloning() throws CloneNotSupportedException {
93          DefaultKeyedValues2D v1 = new DefaultKeyedValues2D();
94          v1.setValue(new Integer(1), "V1", "C1");
95          v1.setValue(null, "V2", "C1");
96          v1.setValue(new Integer(3), "V3", "C2");
97          DefaultKeyedValues2D v2 = (DefaultKeyedValues2D) v1.clone();
98          assertTrue(v1 != v2);
99          assertTrue(v1.getClass() == v2.getClass());
100         assertTrue(v1.equals(v2));
101 
102         // check that clone is independent of the original
103         v2.setValue(new Integer(2), "V2", "C1");
104         assertFalse(v1.equals(v2));
105     }
