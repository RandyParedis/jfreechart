43  package org.jfree.chart.block;
44  
45  import java.awt.Graphics2D;
46  import org.jfree.chart.ui.Size2D;
47  
48  /**
49   * An object that is responsible for arranging a collection of {@link Block}s
50   * within a {@link BlockContainer}.
51   */
52  public interface Arrangement {
53  
54      /**
55       * Adds a block and a key which can be used to determine the position of
56       * the block in the arrangement.  This method is called by the container
57       * (you don't need to call this method directly) and gives the arrangement
58       * an opportunity to record the details if they are required.
59       *
60       * @param block  the block.
61       * @param key  the key ({@code null} permitted).
62       */
63      public void add(Block block, Object key);
64  
65      /**
66       * Arranges the blocks within the specified container, subject to the given
67       * constraint.
68       *
69       * @param container  the container ({@code null} not permitted).
70       * @param g2  the graphics device.
71       * @param constraint  the constraint.
72       *
73       * @return The container size after the arrangement.
74       */
75      public Size2D arrange(BlockContainer container,
76                            Graphics2D g2,
77                            RectangleConstraint constraint);
78  
79      /**
80       * Clears any cached layout information retained by the arrangement.
81       */
82      public void clear();
83  
84  }
