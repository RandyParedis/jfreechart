191     /**
192      * Creates a new instance.
193      *
194      * @param lowerBound  the lower bound of the scale.
195      * @param upperBound  the upper bound of the scale.
196      * @param startAngle  the start angle (in degrees, using the same
197      *     orientation as Java's {@code Arc2D} class).
198      * @param extent  the extent (in degrees, counter-clockwise).
199      * @param majorTickIncrement  the interval between major tick marks (must
200      *     be &gt; 0).
201      * @param minorTickCount  the number of minor ticks between major tick
202      *          marks.
203      */
204     public StandardDialScale(double lowerBound, double upperBound,
205             double startAngle, double extent, double majorTickIncrement,
206             int minorTickCount) {
207         if (majorTickIncrement <= 0.0) {
208             throw new IllegalArgumentException(
209                     "Requires 'majorTickIncrement' > 0.");
210         }
211         this.startAngle = startAngle;
212         this.extent = extent;
213         this.lowerBound = lowerBound;
214         this.upperBound = upperBound;
215         this.tickRadius = 0.70;
216         this.tickLabelsVisible = true;
217         this.tickLabelFormatter = new DecimalFormat("0.0");
218         this.firstTickLabelVisible = true;
219         this.tickLabelFont = new Font("Dialog", Font.BOLD, 16);
220         this.tickLabelPaint = Color.BLUE;
221         this.tickLabelOffset = 0.10;
222         this.majorTickIncrement = majorTickIncrement;
223         this.majorTickLength = 0.04;
224         this.majorTickPaint = Color.BLACK;
225         this.majorTickStroke = new BasicStroke(3.0f);
226         this.minorTickCount = minorTickCount;
227         this.minorTickLength = 0.02;
228         this.minorTickPaint = Color.BLACK;
229         this.minorTickStroke = new BasicStroke(1.0f);
230     }

746     /**
747      * Draws the scale on the dial plot.
748      *
749      * @param g2  the graphics target ({@code null} not permitted).
750      * @param plot  the dial plot ({@code null} not permitted).
751      * @param frame  the reference frame that is used to construct the
752      *     geometry of the plot ({@code null} not permitted).
753      * @param view  the visible part of the plot ({@code null} not
754      *     permitted).
755      */
756     @Override
757     public void draw(Graphics2D g2, DialPlot plot, Rectangle2D frame,
758             Rectangle2D view) {
759 
760         Rectangle2D arcRect = DialPlot.rectangleByRadius(frame,
761                 this.tickRadius, this.tickRadius);
762         Rectangle2D arcRectMajor = DialPlot.rectangleByRadius(frame,
763                 this.tickRadius - this.majorTickLength,
764                 this.tickRadius - this.majorTickLength);
765         Rectangle2D arcRectMinor = arcRect;
766         if (this.minorTickCount > 0 && this.minorTickLength > 0.0) {
767             arcRectMinor = DialPlot.rectangleByRadius(frame,
768                     this.tickRadius - this.minorTickLength,
769                     this.tickRadius - this.minorTickLength);
770         }
771         Rectangle2D arcRectForLabels = DialPlot.rectangleByRadius(frame,
772                 this.tickRadius - this.tickLabelOffset,
773                 this.tickRadius - this.tickLabelOffset);
774 
775         boolean firstLabel = true;
776 
777         Arc2D arc = new Arc2D.Double();
778         Line2D workingLine = new Line2D.Double();
779         for (double v = this.lowerBound; v <= this.upperBound;
780                 v += this.majorTickIncrement) {
781             arc.setArc(arcRect, this.startAngle, valueToAngle(v)
782                     - this.startAngle, Arc2D.OPEN);
783             Point2D pt0 = arc.getEndPoint();
784             arc.setArc(arcRectMajor, this.startAngle, valueToAngle(v)
