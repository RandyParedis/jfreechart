439     /**
440      * Some checks for title changes and event notification.
441      */
442     @Test
443     public void testTitleChangeEvent() {
444         DefaultPieDataset dataset = new DefaultPieDataset();
445         JFreeChart chart = ChartFactory.createPieChart("title", dataset);
446         chart.addChangeListener(this);
447         this.lastChartChangeEvent = null;
448         TextTitle t = chart.getTitle();
449         t.setFont(new Font("Dialog", Font.BOLD, 9));
450         assertNotNull(this.lastChartChangeEvent);
451         this.lastChartChangeEvent = null;
452 
453         // now create a new title and replace the existing title, several
454         // things should happen:
455         // (1) Adding the new title should trigger an immediate
456         //     ChartChangeEvent;
457         // (2) Modifying the new title should trigger a ChartChangeEvent;
458         // (3) Modifying the old title should NOT trigger a ChartChangeEvent
459         TextTitle t2 = new TextTitle("T2");
460         chart.setTitle(t2);
461         assertNotNull(this.lastChartChangeEvent);
462         this.lastChartChangeEvent = null;
463 
464         t2.setFont(new Font("Dialog", Font.BOLD, 9));
465         assertNotNull(this.lastChartChangeEvent);
466         this.lastChartChangeEvent = null;
467 
468         t.setFont(new Font("Dialog", Font.BOLD, 9));
469         assertNull(this.lastChartChangeEvent);
470         this.lastChartChangeEvent = null;
471     }
