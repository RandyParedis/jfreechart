304     /**
305      * Creates a new bar renderer with default settings.
306      */
307     public BarRenderer() {
308         super();
309         this.base = 0.0;
310         this.includeBaseInRange = true;
311         this.itemMargin = DEFAULT_ITEM_MARGIN;
312         this.drawBarOutline = false;
313         this.maximumBarWidth = 1.0;
314             // 100 percent, so it will not apply unless changed
315         this.positiveItemLabelPositionFallback = null;
316         this.negativeItemLabelPositionFallback = null;
317         this.gradientPaintTransformer = new StandardGradientPaintTransformer();
318         this.minimumBarLength = 0.0;
319         setDefaultLegendShape(new Rectangle2D.Double(-4.0, -4.0, 8.0, 8.0));
320         this.barPainter = getDefaultBarPainter();
321         this.shadowsVisible = getDefaultShadowsVisible();
322         this.shadowPaint = Color.GRAY;
323         this.shadowXOffset = 4.0;
324         this.shadowYOffset = 4.0;
325     }

441     /**
442      * Sets the minimum bar length and sends a {@link RendererChangeEvent} to
443      * all registered listeners.  The minimum bar length is specified in Java2D
444      * units, and can be used to prevent bars that represent very small data
445      * values from disappearing when drawn on the screen.  Typically you would
446      * set this to (say) 0.5 or 1.0 Java 2D units.  Use this attribute with
447      * caution, however, because setting it to a non-zero value will
448      * artificially increase the length of bars representing small values,
449      * which may misrepresent your data.
450      *
451      * @param min  the minimum bar length (in Java2D units, must be &gt;= 0.0).
452      *
453      * @see #getMinimumBarLength()
454      */
455     public void setMinimumBarLength(double min) {
456         if (min < 0.0) {
457             throw new IllegalArgumentException("Requires 'min' >= 0.0");
458         }
459         this.minimumBarLength = min;
460         fireChangeEvent();
461     }

764     /**
765      * Calculates the bar width and stores it in the renderer state.
766      *
767      * @param plot  the plot.
768      * @param dataArea  the data area.
769      * @param rendererIndex  the renderer index.
770      * @param state  the renderer state.
771      */
772     protected void calculateBarWidth(CategoryPlot plot,
773                                      Rectangle2D dataArea,
774                                      int rendererIndex,
775                                      CategoryItemRendererState state) {
776 
777         CategoryAxis domainAxis = getDomainAxis(plot, rendererIndex);
778         CategoryDataset dataset = plot.getDataset(rendererIndex);
779         if (dataset != null) {
780             int columns = dataset.getColumnCount();
781             int rows = state.getVisibleSeriesCount() >= 0
782                     ? state.getVisibleSeriesCount() : dataset.getRowCount();
783             double space = 0.0;
784             PlotOrientation orientation = plot.getOrientation();
785             if (orientation == PlotOrientation.HORIZONTAL) {
786                 space = dataArea.getHeight();
787             }
788             else if (orientation == PlotOrientation.VERTICAL) {
789                 space = dataArea.getWidth();
790             }
791             double maxWidth = space * getMaximumBarWidth();
792             double categoryMargin = 0.0;
793             double currentItemMargin = 0.0;
794             if (columns > 1) {
795                 categoryMargin = domainAxis.getCategoryMargin();
796             }
797             if (rows > 1) {
798                 currentItemMargin = getItemMargin();
799             }
800             double used = space * (1 - domainAxis.getLowerMargin()
801                                      - domainAxis.getUpperMargin()
802                                      - categoryMargin - currentItemMargin);
803             if ((rows * columns) > 0) {
804                 state.setBarWidth(Math.min(used / (rows * columns), maxWidth));
805             }
806             else {
807                 state.setBarWidth(Math.min(used, maxWidth));
808             }
809         }
810     }

812     /**
813      * Calculates the coordinate of the first "side" of a bar.  This will be
814      * the minimum x-coordinate for a vertical bar, and the minimum
815      * y-coordinate for a horizontal bar.
816      *
817      * @param plot  the plot.
818      * @param orientation  the plot orientation.
819      * @param dataArea  the data area.
820      * @param domainAxis  the domain axis.
821      * @param state  the renderer state (has the bar width precalculated).
822      * @param row  the row index.
823      * @param column  the column index.
824      *
825      * @return The coordinate.
826      */
827     protected double calculateBarW0(CategoryPlot plot, 
828             PlotOrientation orientation, Rectangle2D dataArea, 
829             CategoryAxis domainAxis, CategoryItemRendererState state,
830             int row, int column) {
831         // calculate bar width...
832         double space;
833         if (orientation == PlotOrientation.HORIZONTAL) {
834             space = dataArea.getHeight();
835         }
836         else {
837             space = dataArea.getWidth();
838         }
839         double barW0 = domainAxis.getCategoryStart(column, getColumnCount(),
840                 dataArea, plot.getDomainAxisEdge());
841         int seriesCount = state.getVisibleSeriesCount() >= 0
842                 ? state.getVisibleSeriesCount() : getRowCount();
843         int categoryCount = getColumnCount();
844         if (seriesCount > 1) {
845             double seriesGap = space * getItemMargin()
846                                / (categoryCount * (seriesCount - 1));
847             double seriesW = calculateSeriesWidth(space, domainAxis,
848                     categoryCount, seriesCount);
849             barW0 = barW0 + row * (seriesW + seriesGap)
850                           + (seriesW / 2.0) - (state.getBarWidth() / 2.0);
851         }
852         else {
853             barW0 = domainAxis.getCategoryMiddle(column, getColumnCount(),
854                     dataArea, plot.getDomainAxisEdge()) - state.getBarWidth()
855                     / 2.0;
856         }
857         return barW0;
858     }

860     /**
861      * Calculates the coordinates for the length of a single bar.
862      *
863      * @param value  the value represented by the bar.
864      *
865      * @return The coordinates for each end of the bar (or {@code null} if
866      *         the bar is not visible for the current axis range).
867      */
868     protected double[] calculateBarL0L1(double value) {
869         double lclip = getLowerClip();
870         double uclip = getUpperClip();
871         double barLow = Math.min(this.base, value);
872         double barHigh = Math.max(this.base, value);
873         if (barHigh < lclip) {  // bar is not visible
874             return null;
875         }
876         if (barLow > uclip) {   // bar is not visible
877             return null;
878         }
879         barLow = Math.max(barLow, lclip);
880         barHigh = Math.min(barHigh, uclip);
881         return new double[] {barLow, barHigh};
882     }

970     /**
971      * Draws the bar for a single (series, category) data item.
972      *
973      * @param g2  the graphics device.
974      * @param state  the renderer state.
975      * @param dataArea  the data area.
976      * @param plot  the plot.
977      * @param domainAxis  the domain axis.
978      * @param rangeAxis  the range axis.
979      * @param dataset  the dataset.
980      * @param row  the row index (zero-based).
981      * @param column  the column index (zero-based).
982      * @param pass  the pass index.
983      */
984     @Override
985     public void drawItem(Graphics2D g2, CategoryItemRendererState state,
986             Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
987             ValueAxis rangeAxis, CategoryDataset dataset, int row,
988             int column, int pass) {
989 
990         // nothing is drawn if the row index is not included in the list with
991         // the indices of the visible rows...
992         int visibleRow = state.getVisibleSeriesIndex(row);
993         if (visibleRow < 0) {
994             return;
995         }
996         // nothing is drawn for null values...
997         Number dataValue = dataset.getValue(row, column);
998         if (dataValue == null) {
999             return;
1000         }
1001 
1002         final double value = dataValue.doubleValue();
1003         PlotOrientation orientation = plot.getOrientation();
1004         double barW0 = calculateBarW0(plot, orientation, dataArea, domainAxis,
1005                 state, visibleRow, column);
1006         double[] barL0L1 = calculateBarL0L1(value);
1007         if (barL0L1 == null) {
1008             return;  // the bar is not visible
1009         }
1010 
1011         RectangleEdge edge = plot.getRangeAxisEdge();
1012         double transL0 = rangeAxis.valueToJava2D(barL0L1[0], dataArea, edge);
1013         double transL1 = rangeAxis.valueToJava2D(barL0L1[1], dataArea, edge);
1014 
1015         // in the following code, barL0 is (in Java2D coordinates) the LEFT
1016         // end of the bar for a horizontal bar chart, and the TOP end of the
1017         // bar for a vertical bar chart.  Whether this is the BASE of the bar
1018         // or not depends also on (a) whether the data value is 'negative'
1019         // relative to the base value and (b) whether or not the range axis is
1020         // inverted.  This only matters if/when we apply the minimumBarLength
1021         // attribute, because we should extend the non-base end of the bar
1022         boolean positive = (value >= this.base);
1023         boolean inverted = rangeAxis.isInverted();
1024         double barL0 = Math.min(transL0, transL1);
1025         double barLength = Math.abs(transL1 - transL0);
1026         double barLengthAdj = 0.0;
1027         if (barLength > 0.0 && barLength < getMinimumBarLength()) {
1028             barLengthAdj = getMinimumBarLength() - barLength;
1029         }
1030         double barL0Adj = 0.0;
1031         RectangleEdge barBase;
1032         if (orientation == PlotOrientation.HORIZONTAL) {
1033             if (positive && inverted || !positive && !inverted) {
1034                 barL0Adj = barLengthAdj;
1035                 barBase = RectangleEdge.RIGHT;
1036             }
1037             else {
1038                 barBase = RectangleEdge.LEFT;
1039             }
1040         }
1041         else {
1042             if (positive && !inverted || !positive && inverted) {
1043                 barL0Adj = barLengthAdj;
1044                 barBase = RectangleEdge.BOTTOM;
1045             }
1046             else {
1047                 barBase = RectangleEdge.TOP;
1048             }
1049         }
1050 
1051         // draw the bar...
1052         Rectangle2D bar;
1053         if (orientation == PlotOrientation.HORIZONTAL) {
1054             bar = new Rectangle2D.Double(barL0 - barL0Adj, barW0,
1055                     barLength + barLengthAdj, state.getBarWidth());
1056         }
1057         else {
1058             bar = new Rectangle2D.Double(barW0, barL0 - barL0Adj,
1059                     state.getBarWidth(), barLength + barLengthAdj);
1060         }
1061         if (state.getElementHinting()) {
1062             KeyedValues2DItemKey key = new KeyedValues2DItemKey(
1063                     dataset.getRowKey(row), dataset.getColumnKey(column));
1064             beginElementGroup(g2, key);
1065         }
1066         if (getShadowsVisible()) {
1067             this.barPainter.paintBarShadow(g2, this, row, column, bar, barBase,
1068                 true);
1069         }
1070         this.barPainter.paintBar(g2, this, row, column, bar, barBase);
1071         if (state.getElementHinting()) {
1072             endElementGroup(g2);
1073         }
1074         
1075         CategoryItemLabelGenerator generator = getItemLabelGenerator(row,
1076                 column);
1077         if (generator != null && isItemLabelVisible(row, column)) {
1078             drawItemLabel(g2, dataset, row, column, plot, generator, bar,
1079                     (value < 0.0));
1080         }
1081 
1082         // submit the current data point as a crosshair candidate
1083         int datasetIndex = plot.indexOf(dataset);
1084         updateCrosshairValues(state.getCrosshairState(),
1085                 dataset.getRowKey(row), dataset.getColumnKey(column), value,
1086                 datasetIndex, barW0, barL0, orientation);
1087 
1088         // add an item entity, if this information is being collected
1089         EntityCollection entities = state.getEntityCollection();
1090         if (entities != null) {
1091             addItemEntity(entities, dataset, row, column, bar);
1092         }
1093 
1094     }

1096     /**
1097      * Calculates the available space for each series.
1098      *
1099      * @param space  the space along the entire axis (in Java2D units).
1100      * @param axis  the category axis.
1101      * @param categories  the number of categories.
1102      * @param series  the number of series.
1103      *
1104      * @return The width of one series.
1105      */
1106     protected double calculateSeriesWidth(double space, CategoryAxis axis,
1107                                           int categories, int series) {
1108         double factor = 1.0 - getItemMargin() - axis.getLowerMargin()
1109                             - axis.getUpperMargin();
1110         if (categories > 1) {
1111             factor = factor - axis.getCategoryMargin();
1112         }
1113         return (space * factor) / (categories * series);
1114     }
