44  import java.awt.Graphics2D;
45  import java.awt.Paint;
46  import java.awt.image.BufferedImage;
47  import org.jfree.chart.renderer.PaintScale;

53  /**
54   * A utility class for the {@link HeatMapDataset}.
55   *
56   * @since 1.0.13
57   */
58  public abstract class HeatMapUtils {
59  
60      /**
61       * Returns a dataset containing one series that holds a copy of the (x, z)
62       * data from one row (y-index) of the specified dataset.
63       *
64       * @param dataset  the dataset ({@code null} not permitted).
65       * @param row  the row (y) index.
66       * @param seriesName  the series name/key ({@code null} not permitted).
67       *
68       * @return The dataset.
69       */
70      public static XYDataset extractRowFromHeatMapDataset(HeatMapDataset dataset,
71              int row, Comparable seriesName) {
72          XYSeries series = new XYSeries(seriesName);
73          int cols = dataset.getXSampleCount();
74          for (int c = 0; c < cols; c++) {
75              series.add(dataset.getXValue(c), dataset.getZValue(c, row));
76          }
77          XYSeriesCollection result = new XYSeriesCollection(series);
78          return result;
79      }
80  
81      /**
82       * Returns a dataset containing one series that holds a copy of the (y, z)
83       * data from one column (x-index) of the specified dataset.
84       *
85       * @param dataset  the dataset ({@code null} not permitted).
86       * @param column  the column (x) index.
87       * @param seriesName  the series name ({@code null} not permitted).
88       *
89       * @return The dataset.
90       */
91      public static XYDataset extractColumnFromHeatMapDataset(
92              HeatMapDataset dataset, int column, Comparable seriesName) {
93          XYSeries series = new XYSeries(seriesName);
94          int rows = dataset.getYSampleCount();
95          for (int r = 0; r < rows; r++) {
96              series.add(dataset.getYValue(r), dataset.getZValue(column, r));
97          }
98          XYSeriesCollection result = new XYSeriesCollection(series);
99          return result;
100     }
101 
102     /**
103      * Creates an image that displays the values from the specified dataset.
104      *
105      * @param dataset  the dataset ({@code null} not permitted).
106      * @param paintScale  the paint scale for the z-values ({@code null}
107      *         not permitted).
108      *
109      * @return A buffered image.
110      */
111     public static BufferedImage createHeatMapImage(HeatMapDataset dataset,
112             PaintScale paintScale) {
113 
114         Args.nullNotPermitted(dataset, "dataset");
115         Args.nullNotPermitted(paintScale, "paintScale");
116         int xCount = dataset.getXSampleCount();
117         int yCount = dataset.getYSampleCount();
118         BufferedImage image = new BufferedImage(xCount, yCount,
119                 BufferedImage.TYPE_INT_ARGB);
120         Graphics2D g2 = image.createGraphics();
121         for (int xIndex = 0; xIndex < xCount; xIndex++) {
122             for (int yIndex = 0; yIndex < yCount; yIndex++) {
123                 double z = dataset.getZValue(xIndex, yIndex);
124                 Paint p = paintScale.getPaint(z);
125                 g2.setPaint(p);
126                 g2.fillRect(xIndex, yCount - yIndex - 1, 1, 1);
127             }
128         }
129         return image;
130     }
