256     /**
257      * Confirm that cloning works.
258      */
259     @Test
260     public void testCloning() throws CloneNotSupportedException {
261         SpiderWebPlot p1 = new SpiderWebPlot(new DefaultCategoryDataset());
262         Rectangle2D legendShape = new Rectangle2D.Double(1.0, 2.0, 3.0, 4.0);
263         p1.setLegendItemShape(legendShape);
264         SpiderWebPlot p2 = (SpiderWebPlot) p1.clone();
265         assertTrue(p1 != p2);
266         assertTrue(p1.getClass() == p2.getClass());
267         assertTrue(p1.equals(p2));
268 
269         // change the legendItemShape
270         legendShape.setRect(4.0, 3.0, 2.0, 1.0);
271         assertFalse(p1.equals(p2));
272         p2.setLegendItemShape(legendShape);
273         assertTrue(p1.equals(p2));
274 
275         // change a series paint
276         p1.setSeriesPaint(1, Color.BLACK);
277         assertFalse(p1.equals(p2));
278         p2.setSeriesPaint(1, Color.BLACK);
279         assertTrue(p1.equals(p2));
280 
281         // change a series outline paint
282         p1.setSeriesOutlinePaint(0, Color.RED);
283         assertFalse(p1.equals(p2));
284         p2.setSeriesOutlinePaint(0, Color.RED);
285         assertTrue(p1.equals(p2));
286 
287         // change a series outline stroke
288         p1.setSeriesOutlineStroke(0, new BasicStroke(1.1f));
289         assertFalse(p1.equals(p2));
290         p2.setSeriesOutlineStroke(0, new BasicStroke(1.1f));
291         assertTrue(p1.equals(p2));
292 
293     }

309     @Test
310     public void testDrawWithNullInfo() {
311         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
312         dataset.addValue(35.0, "S1", "C1");
313         dataset.addValue(45.0, "S1", "C2");
314         dataset.addValue(55.0, "S1", "C3");
315         dataset.addValue(15.0, "S1", "C4");
316         dataset.addValue(25.0, "S1", "C5");
317         SpiderWebPlot plot = new SpiderWebPlot(dataset);
318         JFreeChart chart = new JFreeChart(plot);
319         try {
320             BufferedImage image = new BufferedImage(200 , 100,
321                     BufferedImage.TYPE_INT_RGB);
322             Graphics2D g2 = image.createGraphics();
323             chart.draw(g2, new Rectangle2D.Double(0, 0, 200, 100), null, null);
324             g2.dispose();
325         }
326         catch (Exception e) {
327             fail("There should be no exception.");
328         }

331     /**
332      * Fetches the legend items and checks the values.
333      */
334     @Test
335     public void testGetLegendItems() {
336         DefaultCategoryDataset dataset = new DefaultCategoryDataset();
337         dataset.addValue(35.0, "S1", "C1");
338         dataset.addValue(45.0, "S1", "C2");
339         dataset.addValue(55.0, "S2", "C1");
340         dataset.addValue(15.0, "S2", "C2");
341         SpiderWebPlot plot = new SpiderWebPlot(dataset);
342         JFreeChart chart = new JFreeChart(plot);
343         LegendItemCollection legendItems = plot.getLegendItems();
344         assertEquals(2, legendItems.getItemCount());
345         LegendItem item1 = legendItems.get(0);
346         assertEquals("S1", item1.getLabel());
347         assertEquals("S1", item1.getSeriesKey());
348         assertEquals(0, item1.getSeriesIndex());
349         assertEquals(dataset, item1.getDataset());
350         assertEquals(0, item1.getDatasetIndex());
351 
352         LegendItem item2 = legendItems.get(1);
353         assertEquals("S2", item2.getLabel());
354         assertEquals("S2", item2.getSeriesKey());
355         assertEquals(1, item2.getSeriesIndex());
356         assertEquals(dataset, item2.getDataset());
357         assertEquals(0, item2.getDatasetIndex());
358     }
