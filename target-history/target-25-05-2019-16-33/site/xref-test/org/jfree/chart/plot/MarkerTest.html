349     /**
350      * Checks that a CategoryPlot deregisters listeners when clearing markers.
351      */
352     @Test
353     public void testListenersWithCategoryPlot() {
354         CategoryPlot plot = new CategoryPlot();
355         CategoryMarker marker1 = new CategoryMarker("X");
356         ValueMarker marker2 = new ValueMarker(1.0);
357         plot.addDomainMarker(marker1);
358         plot.addRangeMarker(marker2);
359         EventListener[] listeners1 = marker1.getListeners(
360                 MarkerChangeListener.class);
361         assertTrue(Arrays.asList(listeners1).contains(plot));
362         EventListener[] listeners2 = marker1.getListeners(
363                 MarkerChangeListener.class);
364         assertTrue(Arrays.asList(listeners2).contains(plot));
365         plot.clearDomainMarkers();
366         plot.clearRangeMarkers();
367         listeners1 = marker1.getListeners(MarkerChangeListener.class);
368         assertFalse(Arrays.asList(listeners1).contains(plot));
369         listeners2 = marker1.getListeners(MarkerChangeListener.class);
370         assertFalse(Arrays.asList(listeners2).contains(plot));
371     }
