531     public void testGetLegendItems() {
532         DefaultPieDataset dataset = new DefaultPieDataset();
533         dataset.setValue("Item 1", 1.0);
534         dataset.setValue("Item 2", 2.0);
535         dataset.setValue("Item 3", 0.0);
536         dataset.setValue("Item 4", null);
537 
538         PiePlot plot = new PiePlot(dataset);
539         plot.setIgnoreNullValues(false);
540         plot.setIgnoreZeroValues(false);
541         LegendItemCollection items = plot.getLegendItems();
542         assertEquals(4, items.getItemCount());
543 
544         // check that null items are ignored if requested
545         plot.setIgnoreNullValues(true);
546         items = plot.getLegendItems();
547         assertEquals(3, items.getItemCount());
548 
549         // check that zero items are ignored if requested
550         plot.setIgnoreZeroValues(true);
551         items = plot.getLegendItems();
552         assertEquals(2, items.getItemCount());
553 
554         // check that negative items are always ignored
555         dataset.setValue("Item 5", -1.0);
556         items = plot.getLegendItems();
557         assertEquals(2, items.getItemCount());
558     }
