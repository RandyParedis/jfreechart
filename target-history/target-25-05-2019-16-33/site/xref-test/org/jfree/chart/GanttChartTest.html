132     @Test

136         StandardCategoryToolTipGenerator tt

138         renderer.setSeriesToolTipGenerator(0, tt);

274     private static Date date(int day, int month, int year) {
275 
276         Calendar calendar = Calendar.getInstance();
277         calendar.set(year, month, day);
278         Date result = calendar.getTime();
279         return result;
