86      /**
87       * Confirm that cloning works.
88       */
89      @Test
90      public void testCloning() {
91          GradientPaint gp = new GradientPaint(1.0f, 2.0f, Color.RED, 3.0f, 4.0f,
92                  Color.BLUE);
93          Rectangle2D bounds1 = new Rectangle2D.Double(10.0, 20.0, 30.0, 40.0);
94          ColorBlock b1 = new ColorBlock(gp, 1.0, 2.0);
95          b1.setBounds(bounds1);
96          ColorBlock b2 = null;
97  
98          try {
99              b2 = (ColorBlock) b1.clone();
100         }
101         catch (CloneNotSupportedException e) {
102             fail(e.toString());
103         }
104         assertTrue(b1 != b2);
105         assertTrue(b1.getClass() == b2.getClass());
106         assertTrue(b1.equals(b2));
107 
108         // check independence
109         bounds1.setRect(1.0, 2.0, 3.0, 4.0);
110         assertFalse(b1.equals(b2));
111         b2.setBounds(new Rectangle2D.Double(1.0, 2.0, 3.0, 4.0));
112         assertTrue(b1.equals(b2));
113     }
