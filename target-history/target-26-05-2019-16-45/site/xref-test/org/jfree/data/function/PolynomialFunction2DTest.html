58      @Test
59      public void testConstructor() {
60          PolynomialFunction2D f = new PolynomialFunction2D(new double[] {1.0,
61                  2.0});
62          assertTrue(Arrays.equals(new double[] {1.0, 2.0}, f.getCoefficients()));
63  
64          boolean pass = false;
65          try {
66              f = new PolynomialFunction2D(null);
67          }
68          catch (IllegalArgumentException e) {
69              pass = true;
70          }
71          assertTrue(pass);
72      }

74      /**
75       * Some checks for the getCoefficients() method.
76       */
77      @Test
78      public void testGetCoefficients() {
79          PolynomialFunction2D f = new PolynomialFunction2D(new double[] {1.0,
80                  2.0});
81          double[] c = f.getCoefficients();
82          assertTrue(Arrays.equals(new double[] {1.0, 2.0}, c));
83  
84          // make sure that modifying the returned array doesn't change the
85          // function
86          c[0] = 99.9;
87          assertTrue(Arrays.equals(new double[] {1.0, 2.0}, f.getCoefficients()));
88      }
