68       */
69      @Test
70      public void testCloning() throws CloneNotSupportedException {
71          TickUnits t1 = new TickUnits();
72          t1.add(new NumberTickUnit(10, new DecimalFormat("0.00")));
73          TickUnits t2 = (TickUnits) t1.clone();
74          assertTrue(t1 != t2);
75          assertTrue(t1.getClass() == t2.getClass());
76          assertTrue(t1.equals(t2));
77      }

88          assertTrue(t1.equals(t2));
89          assertTrue(t2.equals(t1));
