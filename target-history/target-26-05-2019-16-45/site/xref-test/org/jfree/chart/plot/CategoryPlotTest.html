556     /**
557      * Confirm that cloning works.
558      */
559     @Test
560     public void testCloning() {
561         CategoryPlot p1 = new CategoryPlot();
562         p1.setRangeCrosshairPaint(new GradientPaint(1.0f, 2.0f, Color.WHITE,
563                 3.0f, 4.0f, Color.YELLOW));
564         p1.setRangeMinorGridlinePaint(new GradientPaint(2.0f, 3.0f, Color.WHITE,
565                 4.0f, 5.0f, Color.RED));
566         p1.setRangeZeroBaselinePaint(new GradientPaint(3.0f, 4.0f, Color.RED,
567                 5.0f, 6.0f, Color.WHITE));
568         CategoryPlot p2;
569         try {
570             p2 = (CategoryPlot) p1.clone();
571         }
572         catch (CloneNotSupportedException e) {
573             fail("Cloning failed.");
574             return;
575         }
576         assertTrue(p1 != p2);
577         assertTrue(p1.getClass() == p2.getClass());
578         assertTrue(p1.equals(p2));
579 
580         // check independence
581         p1.addAnnotation(new CategoryLineAnnotation("C1", 1.0, "C2", 2.0,
582                 Color.RED, new BasicStroke(1.0f)));
583         assertFalse(p1.equals(p2));
584         p2.addAnnotation(new CategoryLineAnnotation("C1", 1.0, "C2", 2.0,
585                 Color.RED, new BasicStroke(1.0f)));
586         assertTrue(p1.equals(p2));
587
