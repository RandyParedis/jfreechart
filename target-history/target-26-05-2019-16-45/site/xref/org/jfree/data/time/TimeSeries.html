113 /**
114  * Represents a sequence of zero or more data items in the form (period, value)
115  * where 'period' is some instance of a subclass of {@link RegularTimePeriod}.
116  * The time series will ensure that (a) all data items have the same type of
117  * period (for example, {@link Day}) and (b) that each period appears at
118  * most one time in the series.
119  */

287      * Sets the maximum number of items that will be retained in the series.
288      * If you add a new item to the series such that the number of items will
289      * exceed the maximum item count, then the FIRST element in the series is
290      * automatically removed, ensuring that the maximum item count is not
291      * exceeded.
292      *
293      * @param maximum  the maximum (requires &gt;= 0).
294      *
295      * @see #getMaximumItemCount()
296      */
297     public void setMaximumItemCount(int maximum) {
298         if (maximum < 0) {
299             throw new IllegalArgumentException("Negative 'maximum' argument.");
300         }
301         this.maximumItemCount = maximum;
302         int count = this.data.size();
303         if (count > maximum) {
304             delete(0, count - maximum - 1);
305         }
306     }

374     /**
375      * Finds the range of y-values that fall within the specified range of
376      * x-values (where the x-values are interpreted as milliseconds since the
377      * epoch and converted to time periods using the specified timezone).
378      * 
379      * @param xRange  the subset of x-values to use ({@code null} not
380      *     permitted).
381      * @param xAnchor  the anchor point for the x-values ({@code null}
382      *     not permitted).
383      * @param zone  the time zone ({@code null} not permitted).
384      * 
385      * @return The range of y-values.
386      * 
387      * @since 1.0.18
388      */
389     public Range findValueRange(Range xRange, TimePeriodAnchor xAnchor, 
390             TimeZone zone) {
391         Args.nullNotPermitted(xRange, "xRange");
392         Args.nullNotPermitted(xAnchor, "xAnchor");
393         Args.nullNotPermitted(zone, "zone");
394         if (this.data.isEmpty()) {
395             return null;
396         }
397         Calendar calendar = Calendar.getInstance(zone);
398         // since the items are ordered, we could be more clever here and avoid
399         // iterating over all the data
400         double lowY = Double.POSITIVE_INFINITY;
401         double highY = Double.NEGATIVE_INFINITY;
402         for (int i = 0; i < this.data.size(); i++) {
403             TimeSeriesDataItem item = (TimeSeriesDataItem) this.data.get(i);
404             long millis = item.getPeriod().getMillisecond(xAnchor, calendar);
405             if (xRange.contains(millis)) {
406                 Number n = item.getValue();
407                 if (n != null) {
408                     double v = n.doubleValue();
409                     lowY = minIgnoreNaN(lowY, v);
410                     highY = maxIgnoreNaN(highY, v);
411                 }
412             }
413         }
414         if (Double.isInfinite(lowY) && Double.isInfinite(highY)) {
415             if (lowY < highY) {
416                 return new Range(lowY, highY);
417             } else {
418                 return new Range(Double.NaN, Double.NaN);
419             }
420         }
421         return new Range(lowY, highY);
422     }

649     /**
650      * Adds a data item to the series and sends a {@link SeriesChangeEvent} to
651      * all registered listeners.
652      *
653      * @param item  the (timeperiod, value) pair ({@code null} not permitted).
654      * @param notify  notify listeners?
655      */
656     public void add(TimeSeriesDataItem item, boolean notify) {
657         Args.nullNotPermitted(item, "item");
658         item = (TimeSeriesDataItem) item.clone();
659         Class c = item.getPeriod().getClass();
660         if (this.timePeriodClass == null) {
661             this.timePeriodClass = c;
662         }
663         else if (!this.timePeriodClass.equals(c)) {
664             StringBuilder b = new StringBuilder();
665             b.append("You are trying to add data where the time period class ");
666             b.append("is ");
667             b.append(item.getPeriod().getClass().getName());
668             b.append(", but the TimeSeries is expecting an instance of ");
669             b.append(this.timePeriodClass.getName());
670             b.append(".");
671             throw new SeriesException(b.toString());
672         }
673 
674         // make the change (if it's not a duplicate time period)...
675         boolean added = false;
676         int count = getItemCount();
677         if (count == 0) {
678             this.data.add(item);
679             added = true;
680         }
681         else {
682             RegularTimePeriod last = getTimePeriod(getItemCount() - 1);
683             if (item.getPeriod().compareTo(last) > 0) {
684                 this.data.add(item);
685                 added = true;
686             }
687             else {
688                 int index = Collections.binarySearch(this.data, item);
689                 if (index < 0) {
690                     this.data.add(-index - 1, item);
691                     added = true;
692                 }
693                 else {
694                     StringBuilder b = new StringBuilder();
695                     b.append("You are attempting to add an observation for ");
696                     b.append("the time period ");
697                     b.append(item.getPeriod().toString());
698                     b.append(" but the series already contains an observation");
699                     b.append(" for that time period. Duplicates are not ");
700                     b.append("permitted.  Try using the addOrUpdate() method.");
