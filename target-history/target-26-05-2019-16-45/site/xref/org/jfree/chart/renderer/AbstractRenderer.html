120 /**
121  * Base class providing common services for renderers.  Most methods that update
122  * attributes of the renderer will fire a {@link RendererChangeEvent}, which
123  * normally means the plot that owns the renderer will receive notification that
124  * the renderer has been changed (the plot will, in turn, notify the chart).
125  *
126  * <b>Subclassing</b>
127  * If you create your own renderer that is a subclass of this, you should take
128  * care to ensure that the renderer implements cloning correctly, to ensure
129  * that {@link org.jfree.chart.JFreeChart JFreeChart} instances that use your renderer are also
130  * cloneable.  It is recommended that you also implement the
131  * {@link PublicCloneable} interface to provide simple access to the clone
132  * method.
133  */
134 public abstract class AbstractRenderer implements Cloneable, Serializable {

610     /**
611      * Notifies all registered listeners that the renderer has been modified.
612      *
613      * @param event information about the change event.
614      */
615     public void notifyListeners(RendererChangeEvent event) {
616         Object[] ls = this.listenerList.getListenerList();
617         for (int i = ls.length - 2; i >= 0; i -= 2) {
618             if (ls[i] == RendererChangeListener.class) {
619                 ((RendererChangeListener) ls[i + 1]).rendererChanged(event);
620             }
621         }
622     }

624     /**
625      * Tests this renderer for equality with another object.
626      *
627      * @param obj the object ({@code null} permitted).
628      * @return {@code true} or {@code false}.
629      */
630     @Override
631     public boolean equals(Object obj) {
632         if (obj == this) {
633             return true;
634         }
635         if (!(obj instanceof AbstractRenderer)) {
636             return false;
637         }
638         AbstractRenderer that = (AbstractRenderer) obj;
639         if (this.defaultEntityRadius != that.defaultEntityRadius) {
640             return false;
641         }
642         if (!ObjectUtils.equal(this.getVisibility(), that.getVisibility())) {
643             return false;
644         }
645         if (!ObjectUtils.equal(this.getPaint(), that.getPaint())) {
646             return false;
647         }
648         if (!ObjectUtils.equal(this.getStroke(), that.getStroke())) {
649             return false;
650         }
651         if (!ObjectUtils.equal(this.getShape(), that.getShape())) {
652             return false;
653         }
654         if (!ObjectUtils.equal(this.getItem(), that.getItem())) {
655             return false;
656         }
657         if (!ObjectUtils.equal(this.createEntitiesList,
658                 that.createEntitiesList)) {
659             return false;
660         }
661         if (this.defaultCreateEntities != that.defaultCreateEntities) {
662             return false;
663         }
664         if (!ObjectUtils.equal(this.legendTextFontMap,
665                 that.legendTextFontMap)) {
666             return false;
667         }
668         if (!ObjectUtils.equal(this.defaultLegendTextFont,
669                 that.defaultLegendTextFont)) {
670             return false;
671         }
672         if (!ObjectUtils.equal(this.legendTextPaint,
673                 that.legendTextPaint)) {
674             return false;
675         }
676         if (!PaintUtils.equal(this.defaultLegendTextPaint,
677                 that.defaultLegendTextPaint)) {
678             return false;
679         }
680         return true;
681     }

701     /**
702      * Returns an independent copy of the renderer.
703      *
704      * @return A clone.
705      * @throws CloneNotSupportedException if some component of the renderer
706      *                                    does not support cloning.
707      */
708     @Override
709     protected Object clone() throws CloneNotSupportedException {
710         AbstractRenderer clone = (AbstractRenderer) super.clone();
711 
712         if (this.renderStateVisibility != null) {
713             clone.renderStateVisibility = (RenderStateVisibility) this.renderStateVisibility.clone();
714             clone.renderStateVisibility.setAbstractRenderer(clone);
715         }
716 
717         if (this.renderStatePaint != null) {
718             clone.renderStatePaint = (RenderStatePaint) this.renderStatePaint.clone();
719             clone.renderStatePaint.setAbstractRenderer(clone);
