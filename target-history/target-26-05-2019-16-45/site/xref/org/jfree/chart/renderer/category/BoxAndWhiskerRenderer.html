445     /**
446      * Returns a legend item for a series.
447      *
448      * @param datasetIndex  the dataset index (zero-based).
449      * @param series  the series index (zero-based).
450      *
451      * @return The legend item (possibly {@code null}).
452      */
453     @Override
454     public LegendItem getLegendItem(int datasetIndex, int series) {
455 
456         CategoryPlot cp = getPlot();
457         if (cp == null) {
458             return null;
459         }
460 
461         // check that a legend item needs to be displayed...
462         if (!getVisibility().isSeriesVisible(series) || !getVisibility().isSeriesVisibleInLegend(series)) {
463             return null;
464         }
465 
466         CategoryDataset dataset = cp.getDataset(datasetIndex);
467         String label = getLegendItemLabelGenerator().generateLabel(dataset,
468                 series);
469         String description = label;
470         String toolTipText = null;
471         if (getLegendItemToolTipGenerator() != null) {
472             toolTipText = getLegendItemToolTipGenerator().generateLabel(
473                     dataset, series);
474         }
475         String urlText = null;
476         if (getLegendItemURLGenerator() != null) {
477             urlText = getLegendItemURLGenerator().generateLabel(dataset,
478                     series);
479         }
480         Shape shape = getShape().lookupLegendShape(series);
481         Paint paint = getPaint().lookupSeriesPaint(series);
482         Paint outlinePaint = getPaint().lookupSeriesOutlinePaint(series);
483         Stroke outlineStroke = getStroke().lookupSeriesOutlineStroke(series);
484         LegendItem result = new LegendItem(label, description, toolTipText,
485                 urlText, shape, paint, outlineStroke, outlinePaint);
486         result.setLabelFont(lookupLegendTextFont(series));
487         Paint labelPaint = lookupLegendTextPaint(series);
488         if (labelPaint != null) {
489             result.setLabelPaint(labelPaint);
490         }
491         result.setDataset(dataset);
492         result.setDatasetIndex(datasetIndex);
493         result.setSeriesKey(dataset.getRowKey(series));
494         result.setSeriesIndex(series);
495         return result;
496 
497     }

512     /**
513      * Initialises the renderer.  This method gets called once at the start of
514      * the process of drawing a chart.
515      *
516      * @param g2  the graphics device.
517      * @param dataArea  the area in which the data is to be plotted.
518      * @param plot  the plot.
519      * @param rendererIndex  the renderer index.
520      * @param info  collects chart rendering information for return to caller.
521      *
522      * @return The renderer state.
523      */
524     @Override
525     public CategoryItemRendererState initialise(Graphics2D g2, 
526             Rectangle2D dataArea, CategoryPlot plot, int rendererIndex,
527             PlotRenderingInfo info) {
528 
529         CategoryItemRendererState state = super.initialise(g2, dataArea, plot,
530                 rendererIndex, info);
531         // calculate the box width
532         CategoryAxis domainAxis = getDomainAxis(plot, rendererIndex);
533         CategoryDataset dataset = plot.getDataset(rendererIndex);
534         if (dataset != null) {
535             int columns = dataset.getColumnCount();
536             int rows = dataset.getRowCount();
537             double space = 0.0;
538             PlotOrientation orientation = plot.getOrientation();
539             if (orientation == PlotOrientation.HORIZONTAL) {
540                 space = dataArea.getHeight();
541             }
542             else if (orientation == PlotOrientation.VERTICAL) {
543                 space = dataArea.getWidth();
544             }
545             double maxWidth = space * getMaximumBarWidth();
546             double categoryMargin = 0.0;
547             double currentItemMargin = 0.0;
548             if (columns > 1) {
549                 categoryMargin = domainAxis.getCategoryMargin();
550             }
551             if (rows > 1) {
552                 currentItemMargin = getItemMargin();
553             }
554             double used = space * (1 - domainAxis.getLowerMargin()
555                                      - domainAxis.getUpperMargin()
556                                      - categoryMargin - currentItemMargin);
557             if ((rows * columns) > 0) {
558                 state.setBarWidth(Math.min(used / (dataset.getColumnCount()
559                         * dataset.getRowCount()), maxWidth));
560             }
561             else {
562                 state.setBarWidth(Math.min(used, maxWidth));
563             }
564         }
565         return state;
566 
567     }

569     /**
570      * Draw a single data item.
571      *
572      * @param g2  the graphics device.
573      * @param state  the renderer state.
574      * @param dataArea  the area in which the data is drawn.
575      * @param plot  the plot.
576      * @param domainAxis  the domain axis.
577      * @param rangeAxis  the range axis.
578      * @param dataset  the data (must be an instance of
579      *                 {@link BoxAndWhiskerCategoryDataset}).
580      * @param row  the row index (zero-based).
581      * @param column  the column index (zero-based).
582      * @param pass  the pass index.
583      */
584     @Override
585     public void drawItem(Graphics2D g2, CategoryItemRendererState state,
586         Rectangle2D dataArea, CategoryPlot plot, CategoryAxis domainAxis,
587         ValueAxis rangeAxis, CategoryDataset dataset, int row, int column,
588         int pass) {
589 
590         // do nothing if item is not visible
591         if (!getVisibility().getItemVisible(row, column)) {
592             return;
593         }
594 
595         if (!(dataset instanceof BoxAndWhiskerCategoryDataset)) {
596             throw new IllegalArgumentException(
597                     "BoxAndWhiskerRenderer.drawItem() : the data should be "
598                     + "of type BoxAndWhiskerCategoryDataset only.");
599         }
600 
601         PlotOrientation orientation = plot.getOrientation();
602 
603         if (orientation == PlotOrientation.HORIZONTAL) {
604             drawHorizontalItem(g2, state, dataArea, plot, domainAxis,
605                     rangeAxis, dataset, row, column);
606         }
607         else if (orientation == PlotOrientation.VERTICAL) {
608             drawVerticalItem(g2, state, dataArea, plot, domainAxis,
609                     rangeAxis, dataset, row, column);
610         }
611 
612     }

614     /**
615      * Draws the visual representation of a single data item when the plot has
616      * a horizontal orientation.
617      *
618      * @param g2  the graphics device.
619      * @param state  the renderer state.
620      * @param dataArea  the area within which the plot is being drawn.
621      * @param plot  the plot (can be used to obtain standard color
622      *              information etc).
623      * @param domainAxis  the domain axis.
624      * @param rangeAxis  the range axis.
625      * @param dataset  the dataset (must be an instance of
626      *                 {@link BoxAndWhiskerCategoryDataset}).
627      * @param row  the row index (zero-based).
628      * @param column  the column index (zero-based).
629      */
630     public void drawHorizontalItem(Graphics2D g2, 
631             CategoryItemRendererState state, Rectangle2D dataArea,
632             CategoryPlot plot, CategoryAxis domainAxis, ValueAxis rangeAxis,
633             CategoryDataset dataset, int row, int column) {
634 
635         BoxAndWhiskerCategoryDataset bawDataset
636                 = (BoxAndWhiskerCategoryDataset) dataset;
637 
638         double categoryEnd = domainAxis.getCategoryEnd(column,
639                 getColumnCount(), dataArea, plot.getDomainAxisEdge());
640         double categoryStart = domainAxis.getCategoryStart(column,
641                 getColumnCount(), dataArea, plot.getDomainAxisEdge());
642         double categoryWidth = Math.abs(categoryEnd - categoryStart);
643 
644         double yy = categoryStart;
645         int seriesCount = getRowCount();
646         int categoryCount = getColumnCount();
647 
648         if (seriesCount > 1) {
649             double seriesGap = dataArea.getHeight() * getItemMargin()
650                                / (categoryCount * (seriesCount - 1));
651             double usedWidth = (state.getBarWidth() * seriesCount)
652                                + (seriesGap * (seriesCount - 1));
653             // offset the start of the boxes if the total width used is smaller
654             // than the category width
655             double offset = (categoryWidth - usedWidth) / 2;
656             yy = yy + offset + (row * (state.getBarWidth() + seriesGap));
657         }
658         else {
659             // offset the start of the box if the box width is smaller than
660             // the category width
661             double offset = (categoryWidth - state.getBarWidth()) / 2;
662             yy = yy + offset;
663         }
664 
665         g2.setPaint(getPaint().getItemPaint(row, column));
666         Stroke s = getStroke().getItemStroke(row, column);
667         g2.setStroke(s);
668 
669         RectangleEdge location = plot.getRangeAxisEdge();
670 
671         Number xQ1 = bawDataset.getQ1Value(row, column);
672         Number xQ3 = bawDataset.getQ3Value(row, column);
673         Number xMax = bawDataset.getMaxRegularValue(row, column);
674         Number xMin = bawDataset.getMinRegularValue(row, column);
675 
676         Shape box = null;
677         if (xQ1 != null && xQ3 != null && xMax != null && xMin != null) {
678 
679             double xxQ1 = rangeAxis.valueToJava2D(xQ1.doubleValue(), dataArea,
680                     location);
681             double xxQ3 = rangeAxis.valueToJava2D(xQ3.doubleValue(), dataArea,
682                     location);
683             double xxMax = rangeAxis.valueToJava2D(xMax.doubleValue(), dataArea,
684                     location);
685             double xxMin = rangeAxis.valueToJava2D(xMin.doubleValue(), dataArea,
686                     location);
687             double yymid = yy + state.getBarWidth() / 2.0;
688             double halfW = (state.getBarWidth() / 2.0) * this.whiskerWidth;
689
