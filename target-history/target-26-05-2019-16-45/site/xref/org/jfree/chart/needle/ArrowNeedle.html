58  /**
59   * A needle in the shape of an arrow.
60   */
61  public class ArrowNeedle extends MeterNeedle implements Cloneable, 
62          Serializable {
63  
64      /** For serialization. */
65      private static final long serialVersionUID = -5334056511213782357L;
66  
67      /**
68       * A flag controlling whether or not there is an arrow at the top of the
69       * needle.
70       */
71      private boolean isArrowAtTop = true;
72  
73      /**
74       * Constructs a new arrow needle.
75       *
76       * @param isArrowAtTop  a flag that controls whether or not there is an
77       *     arrow at the top of the needle.
78       */
79      public ArrowNeedle(boolean isArrowAtTop) {
80          this.isArrowAtTop = isArrowAtTop;
81      }
82  
83      /**
84       * Draws the needle.
85       *
86       * @param g2  the graphics device.
87       * @param plotArea  the plot area.
88       * @param rotate  the rotation point.
89       * @param angle  the angle.
90       */
91      @Override
92      protected void drawNeedle(Graphics2D g2, Rectangle2D plotArea,
93              Point2D rotate, double angle) {
94  
95          Line2D shape = new Line2D.Float();
96          Shape d;
97  
98          float x = (float) (plotArea.getMinX() + (plotArea.getWidth() / 2));
99          float minY = (float) plotArea.getMinY();
100         float maxY = (float) plotArea.getMaxY();
101         shape.setLine(x, minY, x, maxY);
102 
103         GeneralPath shape1 = new GeneralPath();
104         if (this.isArrowAtTop) {
105             shape1.moveTo(x, minY);
106             minY += 4 * getSize();
107         } else {
108             shape1.moveTo(x, maxY);
109             minY = maxY - 4 * getSize();
110         }
111         shape1.lineTo(x + getSize(), minY);
112         shape1.lineTo(x - getSize(), minY);
113         shape1.closePath();
114 
115         if ((rotate != null) && (angle != 0)) {
116             getTransform().setToRotation(angle, rotate.getX(), rotate.getY());
117             d = getTransform().createTransformedShape(shape);
118         } else {
119             d = shape;
120         }
121         defaultDisplay(g2, d);
122 
123         if ((rotate != null) && (angle != 0)) {
124             d = getTransform().createTransformedShape(shape1);
125         } else {
126             d = shape1;
127         }
128         defaultDisplay(g2, d);
129 
130     }

132     /**
133      * Tests another object for equality with this object.
134      *
135      * @param obj  the object to test ({@code null} permitted).
136      *
137      * @return A boolean.
138      */
139     @Override
140     public boolean equals(Object obj) {
141         if (obj == this) {
142             return true;
143         }
144         if (!(obj instanceof ArrowNeedle)) {
145             return false;
146         }
147         if (!super.equals(obj)) {
148             return false;
149         }
150         ArrowNeedle that = (ArrowNeedle) obj;
151         if (this.isArrowAtTop != that.isArrowAtTop) {
152             return false;
153         }
154         return true;
155     }

162     @Override

164         int result = super.hashCode();
165         result = HashUtils.hashCode(result, this.isArrowAtTop);
166         return result;
167     }
